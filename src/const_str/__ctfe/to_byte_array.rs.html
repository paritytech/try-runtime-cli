15/// Converts a string slice or a byte string to a byte array.
16///
17/// # Examples
18/// ```
19/// const S: &str = "hello";
20/// const B: &[u8; 6] = b"hello\0";
21/// assert_eq!(const_str::to_byte_array!(S), [b'h', b'e', b'l', b'l', b'o']);
22/// assert_eq!(const_str::to_byte_array!(B), [b'h', b'e', b'l', b'l', b'o', b'\0']);
23/// ```
24///
25#[macro_export]
26macro_rules! to_byte_array {
27    ($s: expr) => {{
28        const OUTPUT_LEN: usize = $s.len();
29        $crate::__ctfe::ToByteArray($s).const_eval::<OUTPUT_LEN>()
30    }};
31}
