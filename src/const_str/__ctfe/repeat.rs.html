7impl Repeat<&str> {
8    pub const fn const_eval<const N: usize>(&self) -> StrBuf<N> {
9        let buf = bytes_repeat(self.0.as_bytes(), self.1);
10        unsafe { StrBuf::new_unchecked(buf) }
11    }
12}
13
14const fn bytes_repeat<const N: usize>(bytes: &[u8], n: usize) -> [u8; N] {
15    constfn_assert!(bytes.len().checked_mul(n).is_some());
16    constfn_assert!(bytes.len() * n == N);
17    let mut buf = [0; N];
18    let mut i = 0;
19    let mut j = 0;
20    while i < n {
21        let mut k = 0;
22        while k < bytes.len() {
23            buf[j] = bytes[k];
24            j += 1;
25            k += 1;
26        }
27        i += 1;
28    }
29    buf
30}

32/// Creates a new string slice by repeating a string slice n times.
33///
34/// # Examples
35///
36/// ```
37/// const S: &str = "abc";
38/// const SSSS: &str = const_str::repeat!(S, 4);
39/// assert_eq!(SSSS, "abcabcabcabc");
40/// ```
41///
42#[macro_export]
43macro_rules! repeat {
44    ($s: expr, $n: expr) => {{
45        const INPUT: &str = $s;
46        const N: usize = $n;
47        const OUTPUT_LEN: usize = INPUT.len() * N;
48        const OUTPUT_BUF: $crate::__ctfe::StrBuf<OUTPUT_LEN> =
49            $crate::__ctfe::Repeat(INPUT, N).const_eval();
50        OUTPUT_BUF.as_str()
51    }};
52}
