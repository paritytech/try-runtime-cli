3pub struct StrBuf<const N: usize>([u8; N]);
4
5impl<const N: usize> StrBuf<N> {
6    /// # Safety
7    /// `buf` must contain valid utf-8 bytes.
8    pub const unsafe fn new_unchecked(buf: [u8; N]) -> Self {
9        #[cfg(debug_assertions)]
10        {
11            constfn_assert!(crate::utf8::run_utf8_validation(&buf).is_ok());
12        }
13        Self(buf)
14    }
15
16    // const since 1.55
17    pub const fn as_str(&self) -> &str {
18        unsafe { core::str::from_utf8_unchecked(&self.0) }
19    }
20
21    pub const fn as_bytes(&self) -> &[u8] {
22        &self.0
23    }
24
25    pub const fn from_str(s: &str) -> Self {
26        let buf = crate::bytes::clone::<N>(s.as_bytes());
27        unsafe { Self::new_unchecked(buf) }
28    }
29}

31pub const fn from_utf8<const N: usize>(v: &[u8]) -> StrBuf<N> {
32    constfn_assert!(v.len() == N);
33
34    match crate::utf8::run_utf8_validation(v) {
35        Ok(()) => {
36            let mut buf = [0; N];
37            let mut i = 0;
38            while i < v.len() {
39                buf[i] = v[i];
40                i += 1;
41            }
42            unsafe { StrBuf::new_unchecked(buf) }
43        }
44        Err(_) => {
45            constfn_panic!("invalid utf-8 sequence")
46        }
47    }
48}

50/// Converts a byte string to a string slice
51///
52/// # Examples
53/// ```
54/// const BYTE_PATH: &[u8] = b"/tmp/file";
55/// const PATH: &str = const_str::from_utf8!(BYTE_PATH);
56///
57/// assert_eq!(PATH, "/tmp/file");
58/// ```
59///
60#[macro_export]
61macro_rules! from_utf8 {
62    ($s: expr) => {{
63        use ::core::primitive::{str, u8, usize};
64        const INPUT: &[u8] = $s;
65        const N: usize = INPUT.len();
66        const OUTPUT_BUF: $crate::__ctfe::StrBuf<N> = $crate::__ctfe::from_utf8(INPUT);
67        OUTPUT_BUF.as_str()
68    }};
69}
