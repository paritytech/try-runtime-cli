1use futures::FutureExt;
2use std::task::{Context, Poll, Waker};
3use std::time::Duration;

5use futures_timer::Delay;

7/// Default value chosen at `<https://github.com/libp2p/rust-libp2p/pull/4838#discussion_r1490184754>`.
8pub(crate) const DEFAULT_AUTOMATIC_THROTTLE: Duration = Duration::from_millis(500);
9
10#[derive(Debug)]
11pub(crate) struct Status {
12    /// If the user did not disable periodic bootstrap (by providing `None` for `periodic_interval`)
13    /// this is the periodic interval and the delay of the current period. When `Delay` finishes,
14    /// a bootstrap will be triggered and the `Delay` will be reset.
15    interval_and_delay: Option<(Duration, Delay)>,
16
17    /// Configured duration to wait before triggering a bootstrap when a new peer
18    /// is inserted in the routing table. `None` if automatic bootstrap is disabled.
19    automatic_throttle: Option<Duration>,
20    /// Timer that will be set (if automatic bootstrap is not disabled) when a new peer is inserted
21    /// in the routing table. When it finishes, it will trigger a bootstrap and will be set to `None`
22    /// again. If an other new peer is inserted in the routing table before this timer finishes,
23    /// the timer is reset.
24    throttle_timer: Option<ThrottleTimer>,
25
26    /// Number of bootstrap requests currently in progress. We ensure neither periodic bootstrap
27    /// or automatic bootstrap trigger new requests when there is still some running.
28    current_bootstrap_requests: usize,
29    /// Waker to wake up the `poll` method if progress is ready to be made.
30    waker: Option<Waker>,
31}
32
33impl Status {
34    pub(crate) fn new(
35        periodic_interval: Option<Duration>,
36        automatic_throttle: Option<Duration>,
37    ) -> Self {
38        Self {
39            interval_and_delay: periodic_interval.map(|interval| (interval, Delay::new(interval))),
40            waker: None,
41            automatic_throttle,
42            throttle_timer: None,
43            current_bootstrap_requests: 0,
44        }
45    }
46
47    /// Trigger a bootstrap now or after the configured `automatic_throttle` if configured.
48    pub(crate) fn trigger(&mut self) {
49        // Registering `self.throttle_timer` means scheduling a bootstrap.
50        // A bootstrap will be triggered when `self.throttle_timer` finishes.
51        // A `throttle_timer` is useful to not trigger a batch of bootstraps when a
52        // batch of peers is inserted into the routing table.
53        if let Some(throttle_duration) = self.automatic_throttle {
54            self.throttle_timer = Some(throttle_duration.into());
55        } else {
56            // The user disabled bootstrapping on new peer in the routing table.
57        }
58
59        // Waking up the waker that could have been registered.
60        if let Some(waker) = self.waker.take() {
61            waker.wake()
62        }
63    }
64
65    pub(crate) fn reset_timers(&mut self) {
66        // Canceling the `throttle_timer` if any and resetting the `delay` if any.
67        self.throttle_timer = None;
68
69        if let Some((interval, delay)) = self.interval_and_delay.as_mut() {
70            delay.reset(*interval);
71        }
72    }
73
74    pub(crate) fn on_started(&mut self) {
75        // No periodic or automatic bootstrap will be triggered as long as
76        // `self.current_bootstrap_requests > 0` but the user could still manually
77        // trigger a bootstrap.
78        self.current_bootstrap_requests += 1;
79
80        // Resetting the Status timers since a bootstrap request is being triggered right now.
81        self.reset_timers();
82    }
83
84    pub(crate) fn on_finish(&mut self) {
85        if let Some(value) = self.current_bootstrap_requests.checked_sub(1) {
86            self.current_bootstrap_requests = value;
87        } else {
88            debug_assert!(
89                false,
90                "Could not decrement current_bootstrap_requests because it's already 0"
91            );
92        }
93
94        // Waking up the waker that could have been registered.
95        if let Some(waker) = self.waker.take() {
96            waker.wake();
97        }
98    }
99
100    pub(crate) fn poll_next_bootstrap(&mut self, cx: &mut Context<'_>) -> Poll<()> {
101        if self.current_bootstrap_requests > 0 {
102            // Some bootstrap request(s) is(are) currently running.
103            self.waker = Some(cx.waker().clone());
104            return Poll::Pending;
105        }
106
107        if let Some(throttle_delay) = &mut self.throttle_timer {
108            // A `throttle_timer` has been registered. It means one or more peers have been
109            // inserted into the routing table and that a bootstrap request should be triggered.
110            // However, to not risk cascading bootstrap requests, we wait a little time to ensure
111            // the user will not add more peers in the routing table in the next "throttle_timer" remaining.
112            if throttle_delay.poll_unpin(cx).is_ready() {
113                // The `throttle_timer` is finished, triggering bootstrap right now.
114                // The call to `on_started` will reset `throttle_delay`.
115                return Poll::Ready(());
116            }
117
118            // The `throttle_timer` is not finished but the periodic interval for triggering bootstrap might be reached.
119        } else {
120            // No new peer has recently been inserted into the routing table or automatic bootstrap is disabled.
121        }
122
123        // Checking if the user has enabled the periodic bootstrap feature.
124        if let Some((_, delay)) = self.interval_and_delay.as_mut() {
125            if let Poll::Ready(()) = delay.poll_unpin(cx) {
126                // It is time to run the periodic bootstrap.
127                // The call to `on_started` will reset `delay`.
128                return Poll::Ready(());
129            }
130        } else {
131            // The user disabled periodic bootstrap.
132        }
133
134        // Registering the `waker` so that we can wake up when calling `on_new_peer_in_routing_table`.
135        self.waker = Some(cx.waker().clone());
136        Poll::Pending
137    }
138
139    #[cfg(test)]
140    async fn next(&mut self) {
141        std::future::poll_fn(|cx| self.poll_next_bootstrap(cx)).await
142    }
143}

145/// Simple enum to indicate when the throttle timer resolves.
146/// A dedicated `Immediate` variant is necessary because creating
147/// `Delay::new(Duration::ZERO)` does not always actually resolve
148/// immediately.
149#[derive(Debug)]
150enum ThrottleTimer {
151    Immediate,
152    Delay(Delay),
153}
154
155impl From<Duration> for ThrottleTimer {
156    fn from(value: Duration) -> Self {
157        if value.is_zero() {
158            Self::Immediate
159        } else {
160            Self::Delay(Delay::new(value))
161        }
162    }
163}
164
165impl futures::Future for ThrottleTimer {
166    type Output = ();
167
168    fn poll(self: std::pin::Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
169        match self.get_mut() {
170            Self::Immediate => Poll::Ready(()),
171            Self::Delay(delay) => delay.poll_unpin(cx),
172        }
173    }
174}

181    const MS_5: Duration = Duration::from_millis(5);
182    const MS_100: Duration = Duration::from_millis(100);

184    fn do_bootstrap(status: &mut Status) {
185        status.on_started();
186        status.on_finish();
187    }
188
189    async fn await_and_do_bootstrap(status: &mut Status) {
190        status.next().await;
191        do_bootstrap(status);
192    }

194    #[async_std::test]
195    async fn immediate_automatic_bootstrap_is_triggered_immediately() {
196        let mut status = Status::new(Some(Duration::from_secs(1)), Some(Duration::ZERO));
197
198        await_and_do_bootstrap(&mut status).await; // Wait for periodic bootstrap
199
200        assert!(
201            status.next().now_or_never().is_none(),
202            "bootstrap to not be triggered immediately because periodic bootstrap is in ~1s"
203        );
204
205        status.trigger(); // Connected to a new peer though!
206        assert!(
207            status.next().now_or_never().is_some(),
208            "bootstrap to be triggered immediately because we connected to a new peer"
209        );
210
211        assert!(
212            async_std::future::timeout(Duration::from_millis(500), status.next())
213                .await
214                .is_ok(),
215            "bootstrap to be triggered in less then the configured delay because we connected to a new peer"
216        );
217    }
