30/// A peer iterator for a dynamically changing list of peers, sorted by increasing
31/// distance to a chosen target.
32#[derive(Debug, Clone)]
33pub struct ClosestPeersIter {
34    config: ClosestPeersIterConfig,
35
36    /// The target whose distance to any peer determines the position of
37    /// the peer in the iterator.
38    target: KeyBytes,
39
40    /// The internal iterator state.
41    state: State,
42
43    /// The closest peers to the target, ordered by increasing distance.
44    closest_peers: BTreeMap<Distance, Peer>,
45
46    /// The number of peers for which the iterator is currently waiting for results.
47    num_waiting: usize,
48}
49
50/// Configuration for a `ClosestPeersIter`.
51#[derive(Debug, Clone)]
52pub struct ClosestPeersIterConfig {
53    /// Allowed level of parallelism.
54    ///
55    /// The `Î±` parameter in the Kademlia paper. The maximum number of peers that
56    /// the iterator is allowed to wait for in parallel while iterating towards the closest
57    /// nodes to a target. Defaults to `ALPHA_VALUE`.
58    pub parallelism: NonZeroUsize,
59
60    /// Number of results (closest peers) to search for.
61    ///
62    /// The number of closest peers for which the iterator must obtain successful results
63    /// in order to finish successfully. Defaults to `K_VALUE`.
64    pub num_results: NonZeroUsize,
65
66    /// The timeout for a single peer.
67    ///
68    /// If a successful result is not reported for a peer within this timeout
69    /// window, the iterator considers the peer unresponsive and will not wait for
70    /// the peer when evaluating the termination conditions, until and unless a
71    /// result is delivered. Defaults to `10` seconds.
72    pub peer_timeout: Duration,
73}
74
75impl Default for ClosestPeersIterConfig {
76    fn default() -> Self {
77        ClosestPeersIterConfig {
78            parallelism: ALPHA_VALUE,
79            num_results: K_VALUE,
80            peer_timeout: Duration::from_secs(10),
81        }
82    }
83}

85impl ClosestPeersIter {
86    /// Creates a new iterator with a default configuration.
87    pub fn new<I>(target: KeyBytes, known_closest_peers: I) -> Self
88    where
89        I: IntoIterator<Item = Key<PeerId>>,
90    {
91        Self::with_config(
92            ClosestPeersIterConfig::default(),
93            target,
94            known_closest_peers,
95        )
96    }
97
98    /// Creates a new iterator with the given configuration.
99    pub fn with_config<I, T>(
100        config: ClosestPeersIterConfig,
101        target: T,
102        known_closest_peers: I,
103    ) -> Self
104    where
105        I: IntoIterator<Item = Key<PeerId>>,
106        T: Into<KeyBytes>,
107    {
108        let target = target.into();
109
110        // Initialise the closest peers to start the iterator with.
111        let closest_peers = BTreeMap::from_iter(
112            known_closest_peers
113                .into_iter()
114                .map(|key| {
115                    let distance = key.distance(&target);
116                    let state = PeerState::NotContacted;
117                    (distance, Peer { key, state })
118                })
119                .take(K_VALUE.into()),
120        );
121
122        // The iterator initially makes progress by iterating towards the target.
123        let state = State::Iterating { no_progress: 0 };
124
125        ClosestPeersIter {
126            config,
127            target,
128            state,
129            closest_peers,
130            num_waiting: 0,
131        }
132    }
133
134    /// Callback for delivering the result of a successful request to a peer.
135    ///
136    /// Delivering results of requests back to the iterator allows the iterator to make
137    /// progress. The iterator is said to make progress either when the given
138    /// `closer_peers` contain a peer closer to the target than any peer seen so far,
139    /// or when the iterator did not yet accumulate `num_results` closest peers and
140    /// `closer_peers` contains a new peer, regardless of its distance to the target.
141    ///
142    /// If the iterator is currently waiting for a result from `peer`,
143    /// the iterator state is updated and `true` is returned. In that
144    /// case, after calling this function, `next` should eventually be
145    /// called again to obtain the new state of the iterator.
146    ///
147    /// If the iterator is finished, it is not currently waiting for a
148    /// result from `peer`, or a result for `peer` has already been reported,
149    /// calling this function has no effect and `false` is returned.
150    pub fn on_success<I>(&mut self, peer: &PeerId, closer_peers: I) -> bool
151    where
152        I: IntoIterator<Item = PeerId>,
153    {
154        if let State::Finished = self.state {
155            return false;
156        }
157
158        let key = Key::from(*peer);
159        let distance = key.distance(&self.target);
160
161        // Mark the peer as succeeded.
162        match self.closest_peers.entry(distance) {
163            Entry::Vacant(..) => return false,
164            Entry::Occupied(mut e) => match e.get().state {
165                PeerState::Waiting(..) => {
166                    debug_assert!(self.num_waiting > 0);
167                    self.num_waiting -= 1;
168                    e.get_mut().state = PeerState::Succeeded;
169                }
170                PeerState::Unresponsive => {
171                    e.get_mut().state = PeerState::Succeeded;
172                }
173                PeerState::NotContacted | PeerState::Failed | PeerState::Succeeded => return false,
174            },
175        }
176
177        let mut cur_range = distance;
178        let num_results = self.config.num_results.get();
179        // furthest_peer is the furthest peer in range among the closest_peers
180        let furthest_peer = self
181            .closest_peers
182            .iter()
183            .enumerate()
184            .nth(num_results - 1)
185            .map(|(_, peer)| peer)
186            .or_else(|| self.closest_peers.iter().last());
187        if let Some((dist, _)) = furthest_peer {
188            cur_range = *dist;
189        }
190
191        // Incorporate the reported closer peers into the iterator.
192        //
193        // The iterator makes progress if:
194        //     1, the iterator did not yet accumulate enough closest peers.
195        //   OR
196        //     2, any of the new peers is closer to the target than any peer seen so far
197        //        (i.e. is the first entry after being incorporated)
198        let mut progress = self.closest_peers.len() < self.config.num_results.get();
199        for peer in closer_peers {
200            let key = peer.into();
201            let distance = self.target.distance(&key);
202            let peer = Peer {
203                key,
204                state: PeerState::NotContacted,
205            };
206
207            let is_first_insert = match self.closest_peers.entry(distance) {
208                Entry::Occupied(_) => false,
209                Entry::Vacant(entry) => {
210                    entry.insert(peer);
211                    true
212                }
213            };
214
215            progress = (is_first_insert && distance < cur_range) || progress;
216        }
217
218        // Update the iterator state.
219        self.state = match self.state {
220            State::Iterating { no_progress } => {
221                let no_progress = if progress { 0 } else { no_progress + 1 };
222                if no_progress >= self.config.parallelism.get() {
223                    State::Stalled
224                } else {
225                    State::Iterating { no_progress }
226                }
227            }
228            State::Stalled => {
229                if progress {
230                    State::Iterating { no_progress: 0 }
231                } else {
232                    State::Stalled
233                }
234            }
235            State::Finished => State::Finished,
236        };
237
238        true
239    }
240
241    /// Callback for informing the iterator about a failed request to a peer.
242    ///
243    /// If the iterator is currently waiting for a result from `peer`,
244    /// the iterator state is updated and `true` is returned. In that
245    /// case, after calling this function, `next` should eventually be
246    /// called again to obtain the new state of the iterator.
247    ///
248    /// If the iterator is finished, it is not currently waiting for a
249    /// result from `peer`, or a result for `peer` has already been reported,
250    /// calling this function has no effect and `false` is returned.
251    pub fn on_failure(&mut self, peer: &PeerId) -> bool {
252        if let State::Finished = self.state {
253            return false;
254        }
255
256        let key = Key::from(*peer);
257        let distance = key.distance(&self.target);
258
259        match self.closest_peers.entry(distance) {
260            Entry::Vacant(_) => return false,
261            Entry::Occupied(mut e) => match e.get().state {
262                PeerState::Waiting(_) => {
263                    debug_assert!(self.num_waiting > 0);
264                    self.num_waiting -= 1;
265                    e.get_mut().state = PeerState::Failed
266                }
267                PeerState::Unresponsive => e.get_mut().state = PeerState::Failed,
268                PeerState::NotContacted | PeerState::Failed | PeerState::Succeeded => return false,
269            },
270        }
271
272        true
273    }
274
275    /// Returns the list of peers for which the iterator is currently waiting
276    /// for results.
277    pub fn waiting(&self) -> impl Iterator<Item = &PeerId> {
278        self.closest_peers
279            .values()
280            .filter_map(|peer| match peer.state {
281                PeerState::Waiting(..) => Some(peer.key.preimage()),
282                _ => None,
283            })
284    }
285
286    /// Returns the number of peers for which the iterator is currently
287    /// waiting for results.
288    pub fn num_waiting(&self) -> usize {
289        self.num_waiting
290    }
291
292    /// Returns true if the iterator is waiting for a response from the given peer.
293    pub fn is_waiting(&self, peer: &PeerId) -> bool {
294        self.waiting().any(|p| peer == p)
295    }

297    /// Advances the state of the iterator, potentially getting a new peer to contact.
298    pub fn next(&mut self, now: Instant) -> PeersIterState<'_> {
299        if let State::Finished = self.state {
300            return PeersIterState::Finished;
301        }
302
303        // Count the number of peers that returned a result. If there is a
304        // request in progress to one of the `num_results` closest peers, the
305        // counter is set to `None` as the iterator can only finish once
306        // `num_results` closest peers have responded (or there are no more
307        // peers to contact, see `num_waiting`).
308        let mut result_counter = Some(0);
309
310        // Check if the iterator is at capacity w.r.t. the allowed parallelism.
311        let at_capacity = self.at_capacity();
312
313        for peer in self.closest_peers.values_mut() {
314            match peer.state {
315                PeerState::Waiting(timeout) => {
316                    if now >= timeout {
317                        // Unresponsive peers no longer count towards the limit for the
318                        // bounded parallelism, though they might still be ongoing and
319                        // their results can still be delivered to the iterator.
320                        debug_assert!(self.num_waiting > 0);
321                        self.num_waiting -= 1;
322                        peer.state = PeerState::Unresponsive
323                    } else if at_capacity {
324                        // The iterator is still waiting for a result from a peer and is
325                        // at capacity w.r.t. the maximum number of peers being waited on.
326                        return PeersIterState::WaitingAtCapacity;
327                    } else {
328                        // The iterator is still waiting for a result from a peer and the
329                        // `result_counter` did not yet reach `num_results`. Therefore
330                        // the iterator is not yet done, regardless of already successful
331                        // queries to peers farther from the target.
332                        result_counter = None;
333                    }
334                }
335
336                PeerState::Succeeded => {
337                    if let Some(ref mut cnt) = result_counter {
338                        *cnt += 1;
339                        // If `num_results` successful results have been delivered for the
340                        // closest peers, the iterator is done.
341                        if *cnt >= self.config.num_results.get() {
342                            self.state = State::Finished;
343                            return PeersIterState::Finished;
344                        }
345                    }
346                }
347
348                PeerState::NotContacted => {
349                    if !at_capacity {
350                        let timeout = now + self.config.peer_timeout;
351                        peer.state = PeerState::Waiting(timeout);
352                        self.num_waiting += 1;
353                        return PeersIterState::Waiting(Some(Cow::Borrowed(peer.key.preimage())));
354                    } else {
355                        return PeersIterState::WaitingAtCapacity;
356                    }
357                }
358
359                PeerState::Unresponsive | PeerState::Failed => {
360                    // Skip over unresponsive or failed peers.
361                }
362            }
363        }
364
365        if self.num_waiting > 0 {
366            // The iterator is still waiting for results and not at capacity w.r.t.
367            // the allowed parallelism, but there are no new peers to contact
368            // at the moment.
369            PeersIterState::Waiting(None)
370        } else {
371            // The iterator is finished because all available peers have been contacted
372            // and the iterator is not waiting for any more results.
373            self.state = State::Finished;
374            PeersIterState::Finished
375        }
376    }
377
378    /// Immediately transitions the iterator to [`PeersIterState::Finished`].
379    pub fn finish(&mut self) {
380        self.state = State::Finished
381    }
382
383    /// Checks whether the iterator has finished.
384    pub fn is_finished(&self) -> bool {
385        self.state == State::Finished
386    }
387
388    /// Consumes the iterator, returning the closest peers.
389    pub fn into_result(self) -> impl Iterator<Item = PeerId> {
390        self.closest_peers
391            .into_iter()
392            .filter_map(|(_, peer)| {
393                if let PeerState::Succeeded = peer.state {
394                    Some(peer.key.into_preimage())
395                } else {
396                    None
397                }
398            })
399            .take(self.config.num_results.get())
400    }
401
402    /// Checks if the iterator is at capacity w.r.t. the permitted parallelism.
403    ///
404    /// While the iterator is stalled, up to `num_results` parallel requests
405    /// are allowed. This is a slightly more permissive variant of the
406    /// requirement that the initiator "resends the FIND_NODE to all of the
407    /// k closest nodes it has not already queried".
408    fn at_capacity(&self) -> bool {
409        match self.state {
410            State::Stalled => {
411                self.num_waiting
412                    >= usize::max(self.config.num_results.get(), self.config.parallelism.get())
413            }
414            State::Iterating { .. } => self.num_waiting >= self.config.parallelism.get(),
415            State::Finished => true,
416        }
417    }
418}

423/// Internal state of the iterator.
424#[derive(Debug, PartialEq, Eq, Copy, Clone)]
425enum State {
426    /// The iterator is making progress by iterating towards `num_results` closest
427    /// peers to the target with a maximum of `parallelism` peers for which the
428    /// iterator is waiting for results at a time.
429    ///
430    /// > **Note**: When the iterator switches back to `Iterating` after being
431    /// > `Stalled`, it may temporarily be waiting for more than `parallelism`
432    /// > results from peers, with new peers only being considered once
433    /// > the number pending results drops below `parallelism`.
434    Iterating {
435        /// The number of consecutive results that did not yield a peer closer
436        /// to the target. When this number reaches `parallelism` and no new
437        /// peer was discovered or at least `num_results` peers are known to
438        /// the iterator, it is considered `Stalled`.
439        no_progress: usize,
440    },
441
442    /// A iterator is stalled when it did not make progress after `parallelism`
443    /// consecutive successful results (see `on_success`).
444    ///
445    /// While the iterator is stalled, the maximum allowed parallelism for pending
446    /// results is increased to `num_results` in an attempt to finish the iterator.
447    /// If the iterator can make progress again upon receiving the remaining
448    /// results, it switches back to `Iterating`. Otherwise it will be finished.
449    Stalled,
450
451    /// The iterator is finished.
452    ///
453    /// A iterator finishes either when it has collected `num_results` results
454    /// from the closest peers (not counting those that failed or are unresponsive)
455    /// or because the iterator ran out of peers that have not yet delivered
456    /// results (or failed).
457    Finished,
458}

460/// Representation of a peer in the context of a iterator.
461#[derive(Debug, Clone)]
462struct Peer {
463    key: Key<PeerId>,
464    state: PeerState,
465}
466
467/// The state of a single `Peer`.
468#[derive(Debug, Copy, Clone)]
469enum PeerState {
470    /// The peer has not yet been contacted.
471    ///
472    /// This is the starting state for every peer.
473    NotContacted,
474
475    /// The iterator is waiting for a result from the peer.
476    Waiting(Instant),
477
478    /// A result was not delivered for the peer within the configured timeout.
479    ///
480    /// The peer is not taken into account for the termination conditions
481    /// of the iterator until and unless it responds.
482    Unresponsive,
483
484    /// Obtaining a result from the peer has failed.
485    ///
486    /// This is a final state, reached as a result of a call to `on_failure`.
487    Failed,
488
489    /// A successful result from the peer has been delivered.
490    ///
491    /// This is a final state, reached as a result of a call to `on_success`.
492    Succeeded,
493}

589    #[test]
590    fn termination_and_parallelism() {
591        fn prop(mut iter: ClosestPeersIter, seed: Seed) {
592            let now = Instant::now();
593            let mut rng = StdRng::from_seed(seed.0);
594
595            let mut expected = iter
596                .closest_peers
597                .values()
598                .map(|e| e.key)
599                .collect::<Vec<_>>();
600            let num_known = expected.len();
601            let max_parallelism = usize::min(iter.config.parallelism.get(), num_known);
602
603            let target = iter.target;
604            let mut remaining;
605            let mut num_failures = 0;
606
607            'finished: loop {
608                if expected.is_empty() {
609                    break;
610                }
611                // Split off the next up to `parallelism` expected peers.
612                else if expected.len() < max_parallelism {
613                    remaining = Vec::new();
614                } else {
615                    remaining = expected.split_off(max_parallelism);
616                }
617
618                // Advance for maximum parallelism.
619                for k in expected.iter() {
620                    match iter.next(now) {
621                        PeersIterState::Finished => break 'finished,
622                        PeersIterState::Waiting(Some(p)) => assert_eq!(&*p, k.preimage()),
623                        PeersIterState::Waiting(None) => panic!("Expected another peer."),
624                        PeersIterState::WaitingAtCapacity => {
625                            panic!("Unexpectedly reached capacity.")
626                        }
627                    }
628                }
629                let num_waiting = iter.num_waiting();
630                assert_eq!(num_waiting, expected.len());
631
632                // Check the bounded parallelism.
633                if iter.at_capacity() {
634                    assert_eq!(iter.next(now), PeersIterState::WaitingAtCapacity)
635                }
636
637                // Report results back to the iterator with a random number of "closer"
638                // peers or an error, thus finishing the "in-flight requests".
639                for (i, k) in expected.iter().enumerate() {
640                    if rng.gen_bool(0.75) {
641                        let num_closer = rng.gen_range(0..iter.config.num_results.get() + 1);
642                        let closer_peers = random_peers(num_closer, &mut rng);
643                        remaining.extend(closer_peers.iter().cloned().map(Key::from));
644                        iter.on_success(k.preimage(), closer_peers);
645                    } else {
646                        num_failures += 1;
647                        iter.on_failure(k.preimage());
648                    }
649                    assert_eq!(iter.num_waiting(), num_waiting - (i + 1));
650                }
651
652                // Re-sort the remaining expected peers for the next "round".
653                remaining.sort_by_key(|k| target.distance(&k));
654
655                expected = remaining
656            }
657
658            // The iterator must be finished.
659            assert_eq!(iter.next(now), PeersIterState::Finished);
660            assert_eq!(iter.state, State::Finished);
661
662            // Determine if all peers have been contacted by the iterator. This _must_ be
663            // the case if the iterator finished with fewer than the requested number
664            // of results.
665            let all_contacted = iter
666                .closest_peers
667                .values()
668                .all(|e| !matches!(e.state, PeerState::NotContacted | PeerState::Waiting { .. }));
669
670            let target = iter.target;
671            let num_results = iter.config.num_results;
672            let result = iter.into_result();
673            let closest = result.map(Key::from).collect::<Vec<_>>();
674
675            assert!(sorted(&target, &closest));
676
677            if closest.len() < num_results.get() {
678                // The iterator returned fewer results than requested. Therefore
679                // either the initial number of known peers must have been
680                // less than the desired number of results, or there must
681                // have been failures.
682                assert!(num_known < num_results.get() || num_failures > 0);
683                // All peers must have been contacted.
684                assert!(all_contacted, "Not all peers have been contacted.");
685            } else {
686                assert_eq!(num_results.get(), closest.len(), "Too  many results.");
687            }
688        }
689
690        QuickCheck::new()
691            .tests(10)
692            .quickcheck(prop as fn(_, _) -> _)
693    }

695    #[test]
696    fn no_duplicates() {
697        fn prop(mut iter: ClosestPeersIter, closer: ArbitraryPeerId) -> bool {
698            let now = Instant::now();
699
700            let closer = vec![closer.0];
701
702            // A first peer reports a "closer" peer.
703            let peer1 = match iter.next(now) {
704                PeersIterState::Waiting(Some(p)) => p.into_owned(),
705                _ => panic!("No peer."),
706            };
707            iter.on_success(&peer1, closer.clone());
708            // Duplicate result from te same peer.
709            iter.on_success(&peer1, closer.clone());
710
711            // If there is a second peer, let it also report the same "closer" peer.
712            match iter.next(now) {
713                PeersIterState::Waiting(Some(p)) => {
714                    let peer2 = p.into_owned();
715                    assert!(iter.on_success(&peer2, closer.clone()))
716                }
717                PeersIterState::Finished => {}
718                _ => panic!("Unexpectedly iter state."),
719            };
720
721            // The "closer" peer must only be in the iterator once.
722            let n = iter
723                .closest_peers
724                .values()
725                .filter(|e| e.key.preimage() == &closer[0])
726                .count();
727            assert_eq!(n, 1);
728
729            true
730        }
731
732        QuickCheck::new()
733            .tests(10)
734            .quickcheck(prop as fn(_, _) -> _)
735    }

737    #[test]
738    fn timeout() {
739        fn prop(mut iter: ClosestPeersIter) -> bool {
740            let mut now = Instant::now();
741            let peer = iter
742                .closest_peers
743                .values()
744                .next()
745                .unwrap()
746                .key
747                .into_preimage();
748
749            // Poll the iterator for the first peer to be in progress.
750            match iter.next(now) {
751                PeersIterState::Waiting(Some(id)) => assert_eq!(&*id, &peer),
752                _ => panic!(),
753            }
754
755            // Artificially advance the clock.
756            now += iter.config.peer_timeout;
757
758            // Advancing the iterator again should mark the first peer as unresponsive.
759            let _ = iter.next(now);
760            match &iter.closest_peers.values().next().unwrap() {
761                Peer {
762                    key,
763                    state: PeerState::Unresponsive,
764                } => {
765                    assert_eq!(key.preimage(), &peer);
766                }
767                Peer { state, .. } => panic!("Unexpected peer state: {state:?}"),
768            }
769
770            let finished = iter.is_finished();
771            iter.on_success(&peer, iter::empty());
772            let closest = iter.into_result().collect::<Vec<_>>();
773
774            if finished {
775                // Delivering results when the iterator already finished must have
776                // no effect.
777                assert_eq!(Vec::<PeerId>::new(), closest)
778            } else {
779                // Unresponsive peers can still deliver results while the iterator
780                // is not finished.
781                assert_eq!(vec![peer], closest)
782            }
783            true
784        }
785
786        QuickCheck::new().tests(10).quickcheck(prop as fn(_) -> _)
787    }
