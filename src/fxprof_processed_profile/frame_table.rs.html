16pub struct FrameTable {
17    addresses: Vec<Option<u32>>,
18    categories: Vec<CategoryHandle>,
19    subcategories: Vec<Subcategory>,

23}

25impl FrameTable {
26    pub fn new() -> Self {
27        Default::default()
28    }

30    pub fn index_for_frame(
31        &mut self,
32        string_table: &mut ThreadStringTable,
33        resource_table: &mut ResourceTable,
34        func_table: &mut FuncTable,
35        native_symbol_table: &mut NativeSymbols,
36        global_libs: &GlobalLibTable,
37        frame: InternalFrame,
38    ) -> usize {
39        let addresses = &mut self.addresses;
40        let funcs = &mut self.funcs;
41        let native_symbols = &mut self.native_symbols;
42        let categories = &mut self.categories;
43        let subcategories = &mut self.subcategories;
44        *self
45            .internal_frame_to_frame_index
46            .entry(frame.clone())
47            .or_insert_with(|| {
48                let frame_index = addresses.len();
49                let (address, location_string_index, native_symbol, resource) = match frame.location
50                {
51                    InternalFrameLocation::UnknownAddress(address) => {
52                        let location_string = format!("0x{address:x}");
53                        let s = string_table.index_for_string(&location_string);
54                        (None, s, None, None)
55                    }
56                    InternalFrameLocation::AddressInLib(address, lib_index) => {
57                        let res =
58                            resource_table.resource_for_lib(lib_index, global_libs, string_table);
59                        let lib = global_libs.get_lib(lib_index).unwrap();
60                        let native_symbol_and_name =
61                            lib.symbol_table.as_deref().and_then(|symbol_table| {
62                                let symbol = symbol_table.lookup(address)?;
63                                Some(
64                                    native_symbol_table.symbol_index_and_string_index_for_symbol(
65                                        lib_index,
66                                        symbol,
67                                        string_table,
68                                    ),
69                                )
70                            });
71                        let (native_symbol, s) = match native_symbol_and_name {
72                            Some((native_symbol, name_string_index)) => {
73                                (Some(native_symbol), name_string_index)
74                            }
75                            None => {
76                                let location_string = format!("0x{address:x}");
77                                (None, string_table.index_for_string(&location_string))
78                            }
79                        };
80                        (Some(address), s, native_symbol, Some(res))
81                    }
82                    InternalFrameLocation::Label(string_index) => (None, string_index, None, None),
83                };
84                let func_index =
85                    func_table.index_for_func(location_string_index, resource, frame.flags);
86                let CategoryPairHandle(category, subcategory_index) = frame.category_pair;
87                let subcategory = match subcategory_index {
88                    Some(index) => Subcategory::Normal(index),
89                    None => Subcategory::Other(category),
90                };
91                addresses.push(address);
92                categories.push(category);
93                subcategories.push(subcategory);
94                funcs.push(func_index);
95                native_symbols.push(native_symbol);
96                frame_index
97            })
98    }

108struct SerializableFrameTable<'a> {
109    table: &'a FrameTable,
110    categories: &'a [Category],
111}
112
113impl<'a> Serialize for SerializableFrameTable<'a> {
114    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
115        let len = self.table.addresses.len();
116        let mut map = serializer.serialize_map(None)?;
117        map.serialize_entry("length", &len)?;
118        map.serialize_entry(
119            "address",
120            &SerializableFrameTableAddressColumn(&self.table.addresses),
121        )?;
122        map.serialize_entry("inlineDepth", &SerializableSingleValueColumn(0u32, len))?;
123        map.serialize_entry("category", &self.table.categories)?;
124        map.serialize_entry(
125            "subcategory",
126            &SerializableSubcategoryColumn(&self.table.subcategories, self.categories),
127        )?;
128        map.serialize_entry("func", &self.table.funcs)?;
129        map.serialize_entry("nativeSymbol", &self.table.native_symbols)?;
130        map.serialize_entry("innerWindowID", &SerializableSingleValueColumn((), len))?;
131        map.serialize_entry("implementation", &SerializableSingleValueColumn((), len))?;
132        map.serialize_entry("line", &SerializableSingleValueColumn((), len))?;
133        map.serialize_entry("column", &SerializableSingleValueColumn((), len))?;
134        map.serialize_entry("optimizations", &SerializableSingleValueColumn((), len))?;
135        map.end()
136    }
137}

139struct SerializableFrameTableAddressColumn<'a>(&'a [Option<u32>]);
140
141impl<'a> Serialize for SerializableFrameTableAddressColumn<'a> {
142    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
143        let mut seq = serializer.serialize_seq(Some(self.0.len()))?;
144        for address in self.0 {
145            match address {
146                Some(address) => seq.serialize_element(&address)?,
147                None => seq.serialize_element(&-1)?,
148            }
149        }
150        seq.end()
151    }
152}
