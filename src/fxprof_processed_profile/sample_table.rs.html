3use crate::cpu_delta::CpuDelta;

6#[derive(Debug, Clone, Default)]
7pub struct SampleTable {
8    sample_weights: Vec<i32>,
9    sample_timestamps: Vec<Timestamp>,
10    sample_stack_indexes: Vec<Option<usize>>,
11    sample_cpu_deltas: Vec<CpuDelta>,
12}
13
14impl SampleTable {
15    pub fn new() -> Self {
16        Default::default()
17    }
18
19    pub fn add_sample(
20        &mut self,
21        timestamp: Timestamp,
22        stack_index: Option<usize>,
23        cpu_delta: CpuDelta,
24        weight: i32,
25    ) {
26        self.sample_weights.push(weight);
27        self.sample_timestamps.push(timestamp);
28        self.sample_stack_indexes.push(stack_index);
29        self.sample_cpu_deltas.push(cpu_delta);
30    }
31
32    pub fn modify_last_sample(&mut self, timestamp: Timestamp, weight: i32) {
33        *self.sample_weights.last_mut().unwrap() += weight;
34        *self.sample_timestamps.last_mut().unwrap() = timestamp;
35    }
36}
37
38impl Serialize for SampleTable {
39    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
40        let len = self.sample_timestamps.len();
41        let mut map = serializer.serialize_map(None)?;
42        map.serialize_entry("length", &len)?;
43        map.serialize_entry("stack", &self.sample_stack_indexes)?;
44        map.serialize_entry("time", &self.sample_timestamps)?;
45        map.serialize_entry("weight", &self.sample_weights)?;
46        map.serialize_entry("weightType", &"samples")?;
47        map.serialize_entry("threadCPUDelta", &self.sample_cpu_deltas)?;
48        map.end()
49    }
