57/// Memory store.
58pub struct MemoryStore {
59    /// Local peer ID. Used to track local providers.
60    local_peer_id: PeerId,
61    /// Configuration.
62    config: MemoryStoreConfig,
63    /// Records.
64    records: HashMap<Key, Record>,
65    /// Provider records.
66    provider_keys: HashMap<Key, Vec<ProviderRecord>>,
67    /// Local providers.
68    local_providers: HashMap<Key, (ContentProvider, Quorum)>,
69    /// Futures to signal it's time to republish a local provider.

99    /// Try to get record from local store for `key`.
100    pub fn get(&mut self, key: &Key) -> Option<&Record> {
101        let is_expired = self
102            .records
103            .get(key)
104            .is_some_and(|record| record.is_expired(std::time::Instant::now()));
105
106        if is_expired {
107            self.records.remove(key);
108            None
109        } else {
110            self.records.get(key)
111        }
112    }
113
114    /// Store record.
115    pub fn put(&mut self, record: Record) {
116        if record.value.len() >= self.config.max_record_size_bytes {
117            tracing::warn!(
118                target: LOG_TARGET,
119                key = ?record.key,
120                publisher = ?record.publisher,
121                size = record.value.len(),
122                max_size = self.config.max_record_size_bytes,
123                "discarding a DHT record that exceeds the configured size limit",
124            );
125            return;
126        }
127
128        let len = self.records.len();
129        match self.records.entry(record.key.clone()) {
130            Entry::Occupied(mut entry) => {
131                // Lean towards the new record.
132                if let (Some(stored_record_ttl), Some(new_record_ttl)) =
133                    (entry.get().expires, record.expires)
134                {
135                    if stored_record_ttl > new_record_ttl {
136                        return;
137                    }
138                }
139
140                entry.insert(record);
141            }
142
143            Entry::Vacant(entry) => {
144                if len >= self.config.max_records {
145                    tracing::warn!(
146                        target: LOG_TARGET,
147                        max_records = self.config.max_records,
148                        "discarding a DHT record, because maximum memory store size reached",
149                    );
150                    return;
151                }
152
153                entry.insert(record);
154            }
155        }
156    }

158    /// Try to get providers from local store for `key`.
159    ///
160    /// Returns a non-empty list of providers, if any.
161    pub fn get_providers(&mut self, key: &Key) -> Vec<ContentProvider> {
162        let drop_key = self.provider_keys.get_mut(key).is_some_and(|providers| {
163            let now = std::time::Instant::now();
164            providers.retain(|p| !p.is_expired(now));
165
166            providers.is_empty()
167        });
168
169        if drop_key {
170            self.provider_keys.remove(key);
171
172            Vec::default()
173        } else {
174            self.provider_keys
175                .get(key)
176                .cloned()
177                .unwrap_or_else(Vec::default)
178                .into_iter()
179                .map(|p| ContentProvider {
180                    peer: p.provider,
181                    addresses: p.addresses,
182                })
183                .collect()
184        }
185    }

187    /// Try to add a provider for `key`. If there are already `max_providers_per_key` for
188    /// this `key`, the new provider is only inserted if its closer to `key` than
189    /// the furthest already inserted provider. The furthest provider is then discarded.
190    ///
191    /// Returns `true` if the provider was added, `false` otherwise.
192    ///
193    /// `quorum` is only relevant for local providers.
194    pub fn put_provider(&mut self, key: Key, provider: ContentProvider) -> bool {
195        // Make sure we have no more than `max_provider_addresses`.
196        let provider_record = {
197            let mut record = ProviderRecord {
198                key,
199                provider: provider.peer,
200                addresses: provider.addresses,
201                expires: std::time::Instant::now() + self.config.provider_ttl,
202            };
203            record.addresses.truncate(self.config.max_provider_addresses);
204            record
205        };
206
207        let can_insert_new_key = self.provider_keys.len() < self.config.max_provider_keys;
208
209        match self.provider_keys.entry(provider_record.key.clone()) {
210            Entry::Vacant(entry) =>
211                if can_insert_new_key {
212                    entry.insert(vec![provider_record]);
213
214                    true
215                } else {
216                    tracing::warn!(
217                        target: LOG_TARGET,
218                        max_provider_keys = self.config.max_provider_keys,
219                        "discarding a provider record, because the provider key limit reached",
220                    );
221
222                    false
223                },
224            Entry::Occupied(mut entry) => {
225                let providers = entry.get_mut();
226
227                // Providers under every key are sorted by distance from the provided key, with
228                // equal distances meaning peer IDs (more strictly, their hashes)
229                // are equal.
230                let provider_position =
231                    providers.binary_search_by(|p| p.distance().cmp(&provider_record.distance()));
232
233                match provider_position {
234                    Ok(i) => {
235                        // Update the provider in place.
236                        providers[i] = provider_record.clone();
237
238                        true
239                    }
240                    Err(i) => {
241                        // `Err(i)` contains the insertion point.
242                        if i == self.config.max_providers_per_key {
243                            tracing::trace!(
244                                target: LOG_TARGET,
245                                key = ?provider_record.key,
246                                provider = ?provider_record.provider,
247                                max_providers_per_key = self.config.max_providers_per_key,
248                                "discarding a provider record, because it's further than \
249                                 existing `max_providers_per_key`",
250                            );
251
252                            false
253                        } else {
254                            if providers.len() == self.config.max_providers_per_key {
255                                providers.pop();
256                            }
257
258                            providers.insert(i, provider_record.clone());
259
260                            true
261                        }
262                    }
263                }
264            }
265        }
266    }

268    /// Try to add ourself as a provider for `key`.
269    ///
270    /// Returns `true` if the provider was added, `false` otherwise.
271    pub fn put_local_provider(&mut self, key: Key, quorum: Quorum) -> bool {
272        let provider = ContentProvider {
273            peer: self.local_peer_id,
274            // For local providers addresses are populated when replying to `GET_PROVIDERS`
275            // requests.
276            addresses: vec![],
277        };
278
279        if self.put_provider(key.clone(), provider.clone()) {
280            let refresh_interval = self.config.provider_refresh_interval;
281            self.local_providers.insert(key.clone(), (provider, quorum));
282            self.pending_provider_refresh.push(Box::pin(async move {
283                tokio::time::sleep(refresh_interval).await;
284                key
285            }));
286
287            true
288        } else {
289            false
290        }
291    }

293    /// Remove local provider for `key`.
294    pub fn remove_local_provider(&mut self, key: Key) {
295        if self.local_providers.remove(&key).is_none() {
296            tracing::warn!(?key, "trying to remove nonexistent local provider",);
297            return;
298        };
299
300        match self.provider_keys.entry(key.clone()) {
301            Entry::Vacant(_) => {
302                tracing::error!(?key, "local provider key not found during removal",);
303                debug_assert!(false);
304            }
305            Entry::Occupied(mut entry) => {
306                let providers = entry.get_mut();
307
308                // Providers are sorted by distance.
309                let local_provider_distance =
310                    KademliaKey::from(self.local_peer_id).distance(&KademliaKey::new(key.clone()));
311                let provider_position =
312                    providers.binary_search_by(|p| p.distance().cmp(&local_provider_distance));
313
314                match provider_position {
315                    Ok(i) => {
316                        providers.remove(i);
317                    }
318                    Err(_) => {
319                        tracing::error!(?key, "local provider not found during removal",);
320                        debug_assert!(false);
321                        return;
322                    }
323                }
324
325                if providers.is_empty() {
326                    entry.remove();
327                }
328            }
329        };
330    }
331
332    /// Poll next action from the store.
333    pub async fn next_action(&mut self) -> Option<MemoryStoreAction> {
334        // [`FuturesStream`] never terminates, so `and_then()` below is always triggered.
335        self.pending_provider_refresh.next().await.and_then(|key| {
336            if let Some((provider, quorum)) = self.local_providers.get(&key).cloned() {
337                tracing::trace!(
338                    target: LOG_TARGET,
339                    ?key,
340                    "refresh provider"
341                );
342
343                Some(MemoryStoreAction::RefreshProvider {
344                    provided_key: key,
345                    provider,
346                    quorum,
347                })
348            } else {
349                tracing::trace!(
350                    target: LOG_TARGET,
351                    ?key,
352                    "it's time to refresh a provider, but we do not provide this key anymore",
353                );
354
355                None
356            }
357        })
358    }

361#[derive(Debug)]
362pub struct MemoryStoreConfig {
363    /// Maximum number of records to store.
364    pub max_records: usize,
365
366    /// Maximum size of a record in bytes.
367    pub max_record_size_bytes: usize,
368
369    /// Maximum number of provider keys this node stores.
370    pub max_provider_keys: usize,
371
372    /// Maximum number of cached addresses per provider.
373    pub max_provider_addresses: usize,
374
375    /// Maximum number of providers per key. Only providers with peer IDs closest to the key are
376    /// kept.
377    pub max_providers_per_key: usize,
378
379    /// Local providers republish interval.
380    pub provider_refresh_interval: Duration,
381
382    /// Provider record TTL.
383    pub provider_ttl: Duration,
384}
385
386impl Default for MemoryStoreConfig {
387    fn default() -> Self {
388        Self {
389            max_records: DEFAULT_MAX_RECORDS,
390            max_record_size_bytes: DEFAULT_MAX_RECORD_SIZE_BYTES,
391            max_provider_keys: DEFAULT_MAX_PROVIDER_KEYS,
392            max_provider_addresses: DEFAULT_MAX_PROVIDER_ADDRESSES,
393            max_providers_per_key: DEFAULT_MAX_PROVIDERS_PER_KEY,
394            provider_refresh_interval: DEFAULT_PROVIDER_REFRESH_INTERVAL,
395            provider_ttl: DEFAULT_PROVIDER_TTL,
396        }
397    }
398}

675        let providers = peer_ids

679                addresses: vec![multiaddr!(Ip4([127, 0, 0, 1]), Tcp(10000u16))],

689            let mut providers = providers;
690            providers.sort_by(|p1, p2| {
691                KademliaKey::from(p1.peer)
692                    .distance(&target)
693                    .cmp(&KademliaKey::from(p2.peer).distance(&target))
694            });
695            providers
696        };

699
700        let provider0_new = ContentProvider {

713                } else {
714                    p
715                }
