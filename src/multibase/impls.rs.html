8macro_rules! derive_base_encoding {
9    ( $(#[$doc:meta] $type:ident, $encoding:expr;)* ) => {
10        $(
11            #[$doc]
12            #[derive(PartialEq, Eq, Clone, Copy, Debug)]
13            pub(crate) struct $type;
14
15            impl BaseCodec for $type {
16                fn encode<I: AsRef<[u8]>>(input: I) -> String {
17                    $encoding.encode(input.as_ref())
18                }
19
20                fn decode<I: AsRef<str>>(input: I) -> Result<Vec<u8>> {
21                    Ok($encoding.decode(input.as_ref().as_bytes())?)
22                }
23            }
24        )*
25    };
26}

48pub(crate) trait BaseCodec {
49    /// Encode with the given byte slice.
50    fn encode<I: AsRef<[u8]>>(input: I) -> String;
51
52    /// Decode with the given string.
53    fn decode<I: AsRef<str>>(input: I) -> Result<Vec<u8>>;
54}
55
56/// Identity, 8-bit binary (encoder and decoder keeps data unmodified).
57#[derive(PartialEq, Eq, Clone, Copy, Debug)]
58pub(crate) struct Identity;
59
60impl BaseCodec for Identity {
61    fn encode<I: AsRef<[u8]>>(input: I) -> String {
62        String::from_utf8(input.as_ref().to_vec()).expect("input must be valid UTF-8 bytes")
63    }
64
65    fn decode<I: AsRef<str>>(input: I) -> Result<Vec<u8>> {
66        Ok(input.as_ref().as_bytes().to_vec())
67    }
68}

130/// Base36, [0-9a-z] no padding (alphabet: abcdefghijklmnopqrstuvwxyz0123456789).
131#[derive(PartialEq, Eq, Clone, Copy, Debug)]
132pub(crate) struct Base36Lower;
133
134impl BaseCodec for Base36Lower {
135    fn encode<I: AsRef<[u8]>>(input: I) -> String {
136        base_x::encode(encoding::BASE36_LOWER, input.as_ref())
137    }
138
139    fn decode<I: AsRef<str>>(input: I) -> Result<Vec<u8>> {
140        // The input is case insensitive, hence lowercase it
141        let lowercased = input.as_ref().to_ascii_lowercase();
142        Ok(base_x::decode(encoding::BASE36_LOWER, &lowercased)?)
143    }
144}
