21// Note: `CacheRegion` cannot have more than four enum variants. This is because
22// `crate::{sync,unsync}::DeqNodes` uses a `tagptr::TagNonNull<DeqNode<T>, 2>`
23// pointer, where the 2-bit tag is `CacheRegion`.
24#[derive(Clone, Copy, Debug, Eq)]
25pub(crate) enum CacheRegion {
26    Window = 0,
27    MainProbation = 1,
28    MainProtected = 2,
29    Other = 3,
30}
31
32impl From<usize> for CacheRegion {
33    fn from(n: usize) -> Self {
34        match n {
35            0 => Self::Window,
36            1 => Self::MainProbation,
37            2 => Self::MainProtected,
38            3 => Self::Other,
39            _ => panic!("No such CacheRegion variant for {n}"),
40        }
41    }
42}
43
44impl CacheRegion {
45    pub(crate) fn name(self) -> &'static str {
46        match self {
47            Self::Window => "window",
48            Self::MainProbation => "main probation",
49            Self::MainProtected => "main protected",
50            Self::Other => "other",
51        }
52    }
53}
54
55impl PartialEq<Self> for CacheRegion {
56    fn eq(&self, other: &Self) -> bool {
57        core::mem::discriminant(self) == core::mem::discriminant(other)
58    }
59}
60
61impl PartialEq<usize> for CacheRegion {
62    fn eq(&self, other: &usize) -> bool {
63        *self as usize == *other
64    }
65}

67#[derive(Clone, Debug)]
68pub(crate) struct HousekeeperConfig {
69    /// The timeout duration for the `run_pending_tasks` method. This is a safe-guard
70    /// to prevent cache read/write operations (that may call `run_pending_tasks`
71    /// internally) from being blocked for a long time when the user wrote a slow
72    /// eviction listener closure.
73    ///
74    /// Used only when the eviction listener closure is set for the cache instance.
75    ///
76    /// Default: `DEFAULT_MAINTENANCE_TASK_TIMEOUT_MILLIS`
77    pub(crate) maintenance_task_timeout: Duration,
78    /// The maximum repeat count for receiving operation logs from the read and write
79    /// log channels. Default: `MAX_LOG_SYNC_REPEATS`.
80    pub(crate) max_log_sync_repeats: u32,
81    /// The batch size of entries to be processed by each internal eviction method.
82    /// Default: `EVICTION_BATCH_SIZE`.
83    pub(crate) eviction_batch_size: u32,
84}
85
86impl Default for HousekeeperConfig {
87    fn default() -> Self {
88        Self {
89            maintenance_task_timeout: Duration::from_millis(
90                DEFAULT_MAINTENANCE_TASK_TIMEOUT_MILLIS,
91            ),
92            max_log_sync_repeats: DEFAULT_MAX_LOG_SYNC_REPEATS as u32,
93            eviction_batch_size: DEFAULT_EVICTION_BATCH_SIZE,
94        }
95    }
96}
97
98impl HousekeeperConfig {
99    #[cfg(test)]
100    pub(crate) fn new(
101        maintenance_task_timeout: Option<Duration>,
102        max_log_sync_repeats: Option<u32>,
103        eviction_batch_size: Option<u32>,
104    ) -> Self {
105        Self {
106            maintenance_task_timeout: maintenance_task_timeout.unwrap_or(Duration::from_millis(
107                DEFAULT_MAINTENANCE_TASK_TIMEOUT_MILLIS,
108            )),
109            max_log_sync_repeats: max_log_sync_repeats
110                .unwrap_or(DEFAULT_MAX_LOG_SYNC_REPEATS as u32),
111            eviction_batch_size: eviction_batch_size.unwrap_or(DEFAULT_EVICTION_BATCH_SIZE),
112        }
113    }
114}
