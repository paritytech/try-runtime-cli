1use std::sync::atomic::{self, AtomicBool, AtomicU16, AtomicU32, Ordering};
2
3use super::{AccessTime, KeyHash};
4use crate::common::time::{AtomicInstant, Instant};
5
6#[derive(Debug)]
7pub(crate) struct EntryInfo<K> {
8    key_hash: KeyHash<K>,
9    /// `is_admitted` indicates that the entry has been admitted to the cache. When
10    /// `false`, it means the entry is _temporary_ admitted to the cache or evicted
11    /// from the cache (so it should not have LRU nodes).
12    is_admitted: AtomicBool,
13    /// `entry_gen` (entry generation) is incremented every time the entry is updated
14    /// in the concurrent hash table.
15    entry_gen: AtomicU16,
16    /// `policy_gen` (policy generation) is incremented every time entry's `WriteOp`
17    /// is applied to the cache policies including the access-order queue (the LRU
18    /// deque).
19    policy_gen: AtomicU16,
20    /// `expiry_gen` (expiry generation) is incremented every time the entry's
21    /// expiration time is modified. Used to detect stale timer node pointers.
22    expiry_gen: AtomicU32,
23    last_accessed: AtomicInstant,
24    last_modified: AtomicInstant,
25    expiration_time: AtomicInstant,
26    policy_weight: AtomicU32,
27}
28
29impl<K> EntryInfo<K> {
30    #[inline]
31    pub(crate) fn new(key_hash: KeyHash<K>, timestamp: Instant, policy_weight: u32) -> Self {
32        #[cfg(feature = "unstable-debug-counters")]
33        super::debug_counters::InternalGlobalDebugCounters::entry_info_created();
34
35        Self {
36            key_hash,
37            is_admitted: AtomicBool::default(),
38            // `entry_gen` starts at 1 and `policy_gen` start at 0.
39            entry_gen: AtomicU16::new(1),
40            policy_gen: AtomicU16::new(0),
41            expiry_gen: AtomicU32::new(0),
42            last_accessed: AtomicInstant::new(timestamp),
43            last_modified: AtomicInstant::new(timestamp),
44            expiration_time: AtomicInstant::default(),
45            policy_weight: AtomicU32::new(policy_weight),
46        }
47    }
48
49    #[inline]
50    pub(crate) fn key_hash(&self) -> &KeyHash<K> {
51        &self.key_hash
52    }
53
54    #[inline]
55    pub(crate) fn is_admitted(&self) -> bool {
56        self.is_admitted.load(Ordering::Acquire)
57    }
58
59    #[inline]
60    pub(crate) fn set_admitted(&self, value: bool) {
61        self.is_admitted.store(value, Ordering::Release);
62    }
63
64    /// Returns `true` if the `ValueEntry` having this `EntryInfo` is dirty.
65    ///
66    /// Dirty means that the entry has been updated in the concurrent hash table but
67    /// not yet in the cache policies such as access-order queue.
68    #[inline]
69    pub(crate) fn is_dirty(&self) -> bool {
70        let result =
71            self.entry_gen.load(Ordering::Relaxed) != self.policy_gen.load(Ordering::Relaxed);
72        atomic::fence(Ordering::Acquire);
73        result
74    }
75
76    #[inline]
77    pub(crate) fn entry_gen(&self) -> u16 {
78        self.entry_gen.load(Ordering::Acquire)
79    }
80
81    /// Increments the entry generation and returns the new value.
82    #[inline]
83    pub(crate) fn incr_entry_gen(&self) -> u16 {
84        // NOTE: This operation wraps around on overflow.
85        let prev = self.entry_gen.fetch_add(1, Ordering::AcqRel);
86        // Need to add `1` to the previous value to get the current value.
87        prev.wrapping_add(1)
88    }
89
90    /// Sets the policy generation to the given value.
91    #[inline]
92    pub(crate) fn set_policy_gen(&self, value: u16) {
93        let g = &self.policy_gen;
94        loop {
95            let current = g.load(Ordering::Acquire);
96
97            // Do not set the given value if it is smaller than the current value of
98            // `policy_gen`. Note that the current value may have been wrapped
99            // around. If the value is much larger than the current value, it is
100            // likely that the value of `policy_gen` has been wrapped around.
101            if current >= value || value.wrapping_sub(current) > u16::MAX / 2 {
102                break;
103            }
104
105            // Try to set the value.
106            if g.compare_exchange_weak(current, value, Ordering::AcqRel, Ordering::Acquire)
107                .is_ok()
108            {
109                break;
110            }
111        }
112    }
113
114    #[inline]
115    pub(crate) fn policy_weight(&self) -> u32 {
116        self.policy_weight.load(Ordering::Acquire)
117    }
118
119    pub(crate) fn set_policy_weight(&self, size: u32) {
120        self.policy_weight.store(size, Ordering::Release);
121    }
122
123    #[inline]
124    pub(crate) fn expiration_time(&self) -> Option<Instant> {
125        self.expiration_time.instant()
126    }
127
128    /// Returns the new expiry generation.
129    pub(crate) fn set_expiration_time(&self, time: Option<Instant>) -> u32 {
130        let new_gen = self
131            .expiry_gen
132            .fetch_add(1, Ordering::AcqRel)
133            .wrapping_add(1);
134        if let Some(t) = time {
135            self.expiration_time.set_instant(t);
136        } else {
137            self.expiration_time.clear();
138        }
139        new_gen
140    }
141
142    #[inline]
143    pub(crate) fn expiry_gen(&self) -> u32 {
144        self.expiry_gen.load(Ordering::Acquire)
145    }
146}
