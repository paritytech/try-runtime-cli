28fn to_kebab_str<'a>(s: &'a str, desc: &str, offset: usize) -> Result<&'a KebabStr> {
29    match KebabStr::new(s) {
30        Some(s) => Ok(s),
31        None => {
32            if s.is_empty() {
33                bail!(offset, "{desc} name cannot be empty");
34            }
35
36            bail!(offset, "{desc} name `{s}` is not in kebab case");
37        }
38    }

58    pub types: Vec<ComponentAnyTypeId>,
59    pub funcs: Vec<ComponentFuncTypeId>,
60    pub values: Vec<(ComponentValType, bool)>,
61    pub instances: Vec<ComponentInstanceTypeId>,

65    pub import_names: IndexSet<ComponentName>,
66    pub exports: IndexMap<String, ComponentEntityType>,
67    pub export_names: IndexSet<ComponentName>,
68
69    has_start: bool,
70    type_info: TypeInfo,
71
72    /// A mapping of imported resources in this component.
73    ///
74    /// This mapping represents all "type variables" imported into the
75    /// component, or resources. This could be resources imported directly as
76    /// a top-level type import or additionally transitively through other
77    /// imported instances.
78    ///
79    /// The mapping element here is a "path" which is a list of indexes into
80    /// the import map that will be generated for this component. Each index
81    /// is an index into an `IndexMap`, and each list is guaranteed to have at
82    /// least one element.
83    ///
84    /// An example of this map is:
85    ///
86    /// ```wasm
87    /// (component
88    ///     ;; [0] - the first import
89    ///     (import "r" (type (sub resource)))
90    ///
91    ///     ;; [1] - the second import
92    ///     (import "r2" (type (sub resource)))
93    ///
94    ///     (import "i" (instance
95    ///         ;; [2, 0] - the third import, and the first export the instance
96    ///         (export "r3" (type (sub resource)))
97    ///         ;; [2, 1] - the third import, and the second export the instance
98    ///         (export "r4" (type (sub resource)))
99    ///     ))
100    ///
101    ///     ;; ...
102    /// )
103    /// ```
104    ///
105    /// The `Vec<usize>` here can be thought of as `Vec<String>` but a
106    /// (hopefully) more efficient representation.
107    ///
108    /// Finally note that this map is listed as an "append only" map because all
109    /// insertions into it should always succeed. Any insertion which overlaps
110    /// with a previous entry indicates a bug in the validator which needs to be
111    /// corrected via other means.
112    //
113    // TODO: make these `SkolemResourceId` and then go fix all the compile
114    // errors, don't add skolem things into the type area
115    imported_resources: IndexMapAppendOnly<ResourceId, Vec<usize>>,
116
117    /// A mapping of "defined" resources in this component, or those which
118    /// are defined within the instantiation of this component.
119    ///
120    /// Defined resources, as the name implies, can sort of be thought of as
121    /// "these are defined within the component". Note though that the means by
122    /// which a local definition can occur are not simply those defined in the
123    /// component but also in its transitively instantiated components
124    /// internally. This means that this set closes over many transitive
125    /// internal items in addition to those defined immediately in the component
126    /// itself.
127    ///
128    /// The `Option<ValType>` in this mapping is whether or not the underlying
129    /// representation of the resource is known to this component. Immediately
130    /// defined resources, for example, will have `Some(I32)` here. Resources
131    /// that come from transitively defined components, for example, will have
132    /// `None`. In the type context all entries here are `None`.
133    ///
134    /// Note that like `imported_resources` all insertions into this map are
135    /// expected to succeed to it's declared as append-only.
136    defined_resources: IndexMapAppendOnly<ResourceId, Option<ValType>>,
137
138    /// A mapping of explicitly exported resources from this component in
139    /// addition to the path that they're exported at.
140    ///
141    /// For more information on the path here see the documentation for
142    /// `imported_resources`. Note that the indexes here index into the
143    /// list of exports of this component.
144    explicit_resources: IndexMap<ResourceId, Vec<usize>>,
145
146    /// The set of types which are considered "exported" from this component.
147    ///
148    /// This is added to whenever a type export is found, or an instance export
149    /// which itself contains a type export. This additionally includes all
150    /// imported types since those are suitable for export as well.
151    ///
152    /// This set is consulted whenever an exported item is added since all
153    /// referenced types must be members of this set.
154    exported_types: Set<ComponentAnyTypeId>,
155
156    /// Same as `exported_types`, but for imports.
157    imported_types: Set<ComponentAnyTypeId>,
158
159    /// The set of top-level resource exports and their names.
160    ///
161    /// This context is used to validate method names such as `[method]foo.bar`
162    /// to ensure that `foo` is an exported resource and that the type mentioned
163    /// in a function type is actually named `foo`.
164    ///
165    /// Note that imports/exports have disjoint contexts to ensure that they're
166    /// validated correctly. Namely you can't retroactively attach methods to an
167    /// import, for example.
168    toplevel_exported_resources: ComponentNameContext,
169
170    /// Same as `toplevel_exported_resources`, but for imports.
171    toplevel_imported_resources: ComponentNameContext,
172}

313    pub(crate) fn check_lower(&self, offset: usize) -> Result<&Self> {
314        if self.post_return.is_some() {
315            bail!(
316                offset,
317                "canonical option `post-return` cannot be specified for lowerings"
318            );
319        }
320
321        if let Concurrency::Async { callback: Some(_) } = self.concurrency {
322            bail!(
323                offset,
324                "canonical option `callback` cannot be specified for lowerings"
325            );
326        }
327
328        if self.gc && self.core_type.is_none() {
329            bail!(
330                offset,
331                "cannot specify `gc` without also specifying a `core-type` for lowerings"
332            )
333        }
334
335        Ok(self)
336    }

338    pub(crate) fn check_lift(
339        &mut self,
340        types: &TypeList,
341        state: &ComponentState,
342        core_ty_id: CoreTypeId,
343        offset: usize,
344    ) -> Result<&Self> {
345        debug_assert!(matches!(
346            types[core_ty_id].composite_type.inner,
347            CompositeInnerType::Func(_)
348        ));
349
350        if let Some(idx) = self.post_return {
351            let post_return_func_ty = types[state.core_function_at(idx, offset)?].unwrap_func();
352            let core_ty = types[core_ty_id].unwrap_func();
353            if post_return_func_ty.params() != core_ty.results()
354                || !post_return_func_ty.results().is_empty()
355            {
356                bail!(
357                    offset,
358                    "canonical option `post-return` uses a core function with an incorrect signature"
359                );
360            }
361        }
362
363        match self.concurrency {
364            Concurrency::Sync => {}
365
366            Concurrency::Async { callback: None } if !state.features.cm_async_stackful() => {
367                bail!(offset, "requires the async stackful feature")
368            }
369            Concurrency::Async { callback: None } => {}
370
371            Concurrency::Async {
372                callback: Some(idx),
373            } => {
374                let func_ty = types[state.core_function_at(idx, offset)?].unwrap_func();
375                if func_ty.params() != [ValType::I32; 3] && func_ty.params() != [ValType::I32] {
376                    return Err(BinaryReaderError::new(
377                        "canonical option `callback` uses a core function with an incorrect signature",
378                        offset,
379                    ));
380                }
381            }
382        }
383
384        if self.core_type.is_some() {
385            bail!(
386                offset,
387                "canonical option `core-type` is not allowed in `canon lift`"
388            )
389        }
390        self.core_type = Some(core_ty_id);
391
392        Ok(self)
393    }

395    pub(crate) fn check_core_type(
396        &self,
397        types: &mut TypeAlloc,
398        actual: FuncType,
399        offset: usize,
400    ) -> Result<CoreTypeId> {
401        if let Some(declared_id) = self.core_type {
402            let declared = types[declared_id].unwrap_func();
403
404            if actual.params() != declared.params() {
405                bail!(
406                    offset,
407                    "declared core type has `{:?}` parameter types, but actual lowering has \
408                     `{:?}` parameter types",
409                    declared.params(),
410                    actual.params(),
411                );
412            }
413
414            if actual.results() != declared.results() {
415                bail!(
416                    offset,
417                    "declared core type has `{:?}` result types, but actual lowering has \
418                     `{:?}` result types",
419                    declared.results(),
420                    actual.results(),
421                );
422            }
423
424            Ok(declared_id)
425        } else {
426            Ok(types.intern_func_type(actual, offset))
427        }
428    }

478        components: &mut [Self],

545    pub fn add_type(
546        components: &mut Vec<Self>,
547        ty: crate::ComponentType,
548        types: &mut TypeAlloc,
549        offset: usize,
550        check_limit: bool,
551    ) -> Result<()> {
552        assert!(!components.is_empty());
553
554        fn current(components: &mut Vec<ComponentState>) -> &mut ComponentState {
555            components.last_mut().unwrap()
556        }
557
558        let id = match ty {
559            crate::ComponentType::Defined(ty) => {
560                let ty = current(components).create_defined_type(ty, types, offset)?;
561                types.push(ty).into()
562            }
563            crate::ComponentType::Func(ty) => {
564                let ty = current(components).create_function_type(ty, types, offset)?;
565                types.push(ty).into()
566            }
567            crate::ComponentType::Component(decls) => {
568                let ty = Self::create_component_type(components, decls.into_vec(), types, offset)?;
569                types.push(ty).into()
570            }
571            crate::ComponentType::Instance(decls) => {
572                let ty = Self::create_instance_type(components, decls.into_vec(), types, offset)?;
573                types.push(ty).into()
574            }
575            crate::ComponentType::Resource { rep, dtor } => {
576                let component = current(components);
577
578                // Resource types cannot be declared in a type context, only
579                // within a component context.
580                if component.kind != ComponentKind::Component {
581                    bail!(
582                        offset,
583                        "resources can only be defined within a concrete component"
584                    );
585                }
586
587                // Current MVP restriction of the component model.
588                if rep != ValType::I32 {
589                    bail!(offset, "resources can only be represented by `i32`");
590                }
591
592                // If specified validate that the destructor is both a valid
593                // function and has the correct signature.
594                if let Some(dtor) = dtor {
595                    let ty = component.core_function_at(dtor, offset)?;
596                    let ty = types[ty].composite_type.unwrap_func();
597                    if ty.params() != [rep] || ty.results() != [] {
598                        bail!(
599                            offset,
600                            "core function {dtor} has wrong signature for a destructor"
601                        );
602                    }
603                }
604
605                // As this is the introduction of a resource create a fresh new
606                // identifier for the resource. This is then added into the
607                // list of defined resources for this component, notably with a
608                // rep listed to enable getting access to various intrinsics
609                // such as `resource.rep`.
610                let id = types.alloc_resource_id();
611                component.defined_resources.insert(id.resource(), Some(rep));
612                id.into()
613            }
614        };
615
616        let current = current(components);
617        if check_limit {
618            check_max(current.type_count(), 1, MAX_WASM_TYPES, "types", offset)?;
619        }
620        current.types.push(id);
621
622        Ok(())
623    }

652    fn add_entity(
653        &mut self,
654        ty: &mut ComponentEntityType,
655        name_and_kind: Option<(&str, ExternKind)>,
656        types: &mut TypeAlloc,
657        offset: usize,
658    ) -> Result<()> {
659        let kind = name_and_kind.map(|(_, k)| k);
660        let (len, max, desc) = match ty {
661            ComponentEntityType::Module(id) => {
662                self.core_modules.push(*id);
663                (self.core_modules.len(), MAX_WASM_MODULES, "modules")
664            }
665            ComponentEntityType::Component(id) => {
666                self.components.push(*id);
667                (self.components.len(), MAX_WASM_COMPONENTS, "components")
668            }
669            ComponentEntityType::Instance(id) => {
670                match kind {
671                    Some(ExternKind::Import) => self.prepare_instance_import(id, types),
672                    Some(ExternKind::Export) => self.prepare_instance_export(id, types),
673                    None => {}
674                }
675                self.instances.push(*id);
676                (self.instance_count(), MAX_WASM_INSTANCES, "instances")
677            }
678            ComponentEntityType::Func(id) => {
679                self.funcs.push(*id);
680                (self.function_count(), MAX_WASM_FUNCTIONS, "functions")
681            }
682            ComponentEntityType::Value(ty) => {
683                self.check_value_support(offset)?;
684                let value_used = match kind {
685                    Some(ExternKind::Import) | None => false,
686                    Some(ExternKind::Export) => true,
687                };
688                self.values.push((*ty, value_used));
689                (self.values.len(), MAX_WASM_VALUES, "values")
690            }
691            ComponentEntityType::Type {
692                created,
693                referenced,
694            } => {
695                self.types.push(*created);
696
697                // Extra logic here for resources being imported and exported.
698                // Note that if `created` is the same as `referenced` then this
699                // is the original introduction of the resource which is where
700                // `self.{imported,defined}_resources` are updated.
701                if let ComponentAnyTypeId::Resource(id) = *created {
702                    match kind {
703                        Some(ExternKind::Import) => {
704                            // A fresh new resource is being imported into a
705                            // component. This arises from the import section of
706                            // a component or from the import declaration in a
707                            // component type. In both cases a new imported
708                            // resource is injected with a fresh new identifier
709                            // into our state.
710                            if created == referenced {
711                                self.imported_resources
712                                    .insert(id.resource(), vec![self.imports.len()]);
713                            }
714                        }
715
716                        Some(ExternKind::Export) => {
717                            // A fresh resource is being exported from this
718                            // component. This arises as part of the
719                            // declaration of a component type, for example. In
720                            // this situation brand new resource identifier is
721                            // allocated and a definition is added, unlike the
722                            // import case where an imported resource is added.
723                            // Notably the representation of this new resource
724                            // is unknown so it's listed as `None`.
725                            if created == referenced {
726                                self.defined_resources.insert(id.resource(), None);
727                            }
728
729                            // If this is a type export of a resource type then
730                            // update the `explicit_resources` list. A new
731                            // export path is about to be created for this
732                            // resource and this keeps track of that.
733                            self.explicit_resources
734                                .insert(id.resource(), vec![self.exports.len()]);
735                        }
736
737                        None => {}
738                    }
739                }
740                (self.types.len(), MAX_WASM_TYPES, "types")
741            }
742        };
743
744        check_max(len, 0, max, desc, offset)?;
745
746        // Before returning perform the final validation of the type of the item
747        // being imported/exported. This will ensure that everything is
748        // appropriately named with respect to type definitions, resources, etc.
749        if let Some((name, kind)) = name_and_kind {
750            if !self.validate_and_register_named_types(Some(name), kind, ty, types) {
751                bail!(
752                    offset,
753                    "{} not valid to be used as {}",
754                    ty.desc(),
755                    kind.desc()
756                );
757            }
758        }
759        Ok(())
760    }

762    /// Validates that the `ty` referenced only refers to named types internally
763    /// and then inserts anything necessary, if applicable, to the defined sets
764    /// within this component.
765    ///
766    /// This function will validate that `ty` only refers to named types. For
767    /// example if it's a record then all of its fields must refer to named
768    /// types. This consults either `self.imported_types` or
769    /// `self.exported_types` as specified by `kind`. Note that this is not
770    /// inherently recursive itself but it ends up being recursive since if
771    /// recursive members were named then all their components must also be
772    /// named. Consequently this check stops at the "one layer deep" position,
773    /// or more accurately the position where types must be named (e.g. tuples
774    /// aren't required to be named).
775    fn validate_and_register_named_types(
776        &mut self,
777        toplevel_name: Option<&str>,
778        kind: ExternKind,
779        ty: &ComponentEntityType,
780        types: &TypeAlloc,
781    ) -> bool {
782        if let ComponentEntityType::Type { created, .. } = ty {
783            // If this is a top-level resource then register it in the
784            // appropriate context so later validation of method-like-names
785            // works out.
786            if let Some(name) = toplevel_name {
787                if let ComponentAnyTypeId::Resource(id) = *created {
788                    let cx = match kind {
789                        ExternKind::Import => &mut self.toplevel_imported_resources,
790                        ExternKind::Export => &mut self.toplevel_exported_resources,
791                    };
792                    cx.register(name, id);
793                }
794            }
795        }
796
797        match self.kind {
798            ComponentKind::Component | ComponentKind::ComponentType => {}
799            ComponentKind::InstanceType => return true,
800        }
801        let set = match kind {
802            ExternKind::Import => &self.imported_types,
803            ExternKind::Export => &self.exported_types,
804        };
805        match ty {
806            // When a type is imported or exported than any recursive type
807            // referred to by that import/export must additionally be exported
808            // or imported. Here this walks the "first layer" of the type which
809            // delegates to `TypeAlloc::type_named_type_id` to determine whether
810            // the components of the type being named here are indeed all they
811            // themselves named.
812            ComponentEntityType::Type {
813                created,
814                referenced,
815            } => {
816                if !self.all_valtypes_named(types, *referenced, set) {
817                    return false;
818                }
819                match kind {
820                    // Imported types are both valid for import and valid for
821                    // export.
822                    ExternKind::Import => {
823                        self.imported_types.insert(*created);
824                        self.exported_types.insert(*created);
825                    }
826                    ExternKind::Export => {
827                        self.exported_types.insert(*created);
828                    }
829                }
830
831                true
832            }
833
834            // Instances are slightly nuanced here. The general idea is that if
835            // an instance is imported, then any type exported by the instance
836            // is then also exported. Additionally for exports. To get this to
837            // work out this arm will recursively call
838            // `validate_and_register_named_types` which means that types are
839            // inserted into `self.{imported,exported}_types` as-we-go rather
840            // than all at once.
841            //
842            // This then recursively validates that all items in the instance
843            // itself are valid to import/export, recursive instances are
844            // captured, and everything is appropriately added to the right
845            // imported/exported set.
846            ComponentEntityType::Instance(i) => types[*i]
847                .exports
848                .iter()
849                .all(|(_name, ty)| self.validate_and_register_named_types(None, kind, ty, types)),
850
851            // All types referred to by a function must be named.
852            ComponentEntityType::Func(id) => self.all_valtypes_named_in_func(types, *id, set),
853
854            ComponentEntityType::Value(ty) => types.type_named_valtype(ty, set),
855
856            // Components/modules are always "closed" or "standalone" and don't
857            // need validation with respect to their named types.
858            ComponentEntityType::Component(_) | ComponentEntityType::Module(_) => true,
859        }
860    }

862    fn all_valtypes_named(
863        &self,
864        types: &TypeAlloc,
865        id: ComponentAnyTypeId,
866        set: &Set<ComponentAnyTypeId>,
867    ) -> bool {
868        match id {
869            // Resource types, in isolation, are always valid to import or
870            // export since they're either attached to an import or being
871            // exported.
872            //
873            // Note that further validation of this happens in `finish`, too.
874            ComponentAnyTypeId::Resource(_) => true,
875
876            // Component types are validated as they are constructed,
877            // so all component types are valid to export if they've
878            // already been constructed.
879            ComponentAnyTypeId::Component(_) => true,
880
881            ComponentAnyTypeId::Defined(id) => self.all_valtypes_named_in_defined(types, id, set),
882            ComponentAnyTypeId::Func(id) => self.all_valtypes_named_in_func(types, id, set),
883            ComponentAnyTypeId::Instance(id) => self.all_valtypes_named_in_instance(types, id, set),
884        }
885    }

887    fn all_valtypes_named_in_instance(
888        &self,
889        types: &TypeAlloc,
890        id: ComponentInstanceTypeId,
891        set: &Set<ComponentAnyTypeId>,
892    ) -> bool {
893        // Instances must recursively have all referenced types named.
894        let ty = &types[id];
895        ty.exports.values().all(|ty| match ty {
896            ComponentEntityType::Module(_) => true,
897            ComponentEntityType::Func(id) => self.all_valtypes_named_in_func(types, *id, set),
898            ComponentEntityType::Type { created: id, .. } => {
899                self.all_valtypes_named(types, *id, set)
900            }
901            ComponentEntityType::Value(ComponentValType::Type(id)) => {
902                self.all_valtypes_named_in_defined(types, *id, set)
903            }
904            ComponentEntityType::Instance(id) => {
905                self.all_valtypes_named_in_instance(types, *id, set)
906            }
907            ComponentEntityType::Component(_)
908            | ComponentEntityType::Value(ComponentValType::Primitive(_)) => return true,
909        })
910    }

912    fn all_valtypes_named_in_defined(
913        &self,
914        types: &TypeAlloc,
915        id: ComponentDefinedTypeId,
916        set: &Set<ComponentAnyTypeId>,
917    ) -> bool {
918        let ty = &types[id];
919        match ty {
920            // These types do not contain anything which must be
921            // named.
922            ComponentDefinedType::Primitive(_)
923            | ComponentDefinedType::Flags(_)
924            | ComponentDefinedType::Enum(_) => true,
925
926            // Referenced types of all these aggregates must all be
927            // named.
928            ComponentDefinedType::Record(r) => {
929                r.fields.values().all(|t| types.type_named_valtype(t, set))
930            }
931            ComponentDefinedType::Tuple(r) => {
932                r.types.iter().all(|t| types.type_named_valtype(t, set))
933            }
934            ComponentDefinedType::Variant(r) => r
935                .cases
936                .values()
937                .filter_map(|t| t.ty.as_ref())
938                .all(|t| types.type_named_valtype(t, set)),
939            ComponentDefinedType::Result { ok, err } => {
940                ok.as_ref()
941                    .map(|t| types.type_named_valtype(t, set))
942                    .unwrap_or(true)
943                    && err
944                        .as_ref()
945                        .map(|t| types.type_named_valtype(t, set))
946                        .unwrap_or(true)
947            }
948            ComponentDefinedType::List(ty)
949            | ComponentDefinedType::FixedSizeList(ty, _)
950            | ComponentDefinedType::Option(ty) => types.type_named_valtype(ty, set),
951
952            // The resource referred to by own/borrow must be named.
953            ComponentDefinedType::Own(id) | ComponentDefinedType::Borrow(id) => {
954                set.contains(&ComponentAnyTypeId::from(*id))
955            }
956
957            ComponentDefinedType::Future(ty) => ty
958                .as_ref()
959                .map(|ty| types.type_named_valtype(ty, set))
960                .unwrap_or(true),
961            ComponentDefinedType::Stream(ty) => ty
962                .as_ref()
963                .map(|ty| types.type_named_valtype(ty, set))
964                .unwrap_or(true),
965        }
966    }

983    /// Updates the type `id` specified, an identifier for a component instance
984    /// type, to be imported into this component.
985    ///
986    /// Importing an instance type into a component specially handles the
987    /// defined resources registered in the instance type. Notably all
988    /// defined resources are "freshened" into brand new type variables and
989    /// these new variables are substituted within the type. This is what
990    /// creates a new `TypeId` and may update the `id` specified.
991    ///
992    /// One side effect of this operation, for example, is that if an instance
993    /// type is used twice to import two different instances then the instances
994    /// do not share resource types despite sharing the same original instance
995    /// type.
996    fn prepare_instance_import(&mut self, id: &mut ComponentInstanceTypeId, types: &mut TypeAlloc) {
997        let ty = &types[*id];
998
999        // No special treatment for imports of instances which themselves have
1000        // no defined resources
1001        if ty.defined_resources.is_empty() {
1002            return;
1003        }
1004
1005        let mut new_ty = ComponentInstanceType {
1006            // Copied from the input verbatim
1007            info: ty.info,
1008
1009            // Copied over as temporary storage for now, and both of these are
1010            // filled out and expanded below.
1011            exports: ty.exports.clone(),
1012            explicit_resources: ty.explicit_resources.clone(),
1013
1014            // Explicitly discard this field since the
1015            // defined resources are lifted into `self`
1016            defined_resources: Default::default(),
1017        };
1018
1019        // Create brand new resources for all defined ones in the instance.
1020        let resources = (0..ty.defined_resources.len())
1021            .map(|_| types.alloc_resource_id())
1022            .collect::<IndexSet<_>>();
1023
1024        // Build a map from the defined resources in `ty` to those in `new_ty`.
1025        //
1026        // As part of this same loop the new resources, which were previously
1027        // defined in `ty`, now become imported variables in `self`. Their
1028        // path for where they're imported is updated as well with
1029        // `self.next_import_index` as the import-to-be soon.
1030        let mut mapping = Remapping::default();
1031        let ty = &types[*id];
1032        for (old, new) in ty.defined_resources.iter().zip(&resources) {
1033            let prev = mapping.resources.insert(*old, new.resource());
1034            assert!(prev.is_none());
1035
1036            let mut base = vec![self.imports.len()];
1037            base.extend(ty.explicit_resources[old].iter().copied());
1038            self.imported_resources.insert(new.resource(), base);
1039        }
1040
1041        // Using the old-to-new resource mapping perform a substitution on
1042        // the `exports` and `explicit_resources` fields of `new_ty`
1043        for ty in new_ty.exports.values_mut() {
1044            types.remap_component_entity(ty, &mut mapping);
1045        }
1046        for (id, path) in mem::take(&mut new_ty.explicit_resources) {
1047            let id = *mapping.resources.get(&id).unwrap_or(&id);
1048            new_ty.explicit_resources.insert(id, path);
1049        }
1050
1051        // Now that `new_ty` is complete finish its registration and then
1052        // update `id` on the way out.
1053        *id = types.push_ty(new_ty);
1054    }

1056    /// Prepares an instance type, pointed to `id`, for being exported as a
1057    /// concrete instance from `self`.
1058    ///
1059    /// This will internally perform any resource "freshening" as required and
1060    /// then additionally update metadata within `self` about resources being
1061    /// exported or defined.
1062    fn prepare_instance_export(&mut self, id: &mut ComponentInstanceTypeId, types: &mut TypeAlloc) {
1063        // Exports of an instance mean that the enclosing context
1064        // is inheriting the resources that the instance
1065        // encapsulates. This means that the instance type
1066        // recorded for this export will itself have no
1067        // defined resources.
1068        let ty = &types[*id];
1069
1070        // Check to see if `defined_resources` is non-empty, and if so then
1071        // "freshen" all the resources and inherit them to our own defined
1072        // resources, updating `id` in the process.
1073        //
1074        // Note though that this specifically is not rewriting the resources of
1075        // exported instances. The `defined_resources` set on instance types is
1076        // a little subtle (see its documentation for more info), but the
1077        // general idea is that for a concrete instance it's always empty. Only
1078        // for instance type definitions does it ever have elements in it.
1079        //
1080        // That means that if this set is non-empty then what's happening is
1081        // that we're in a type context an exporting an instance of a previously
1082        // specified type. In this case all resources are required to be
1083        // "freshened" to ensure that multiple exports of the same type all
1084        // export different types of resources.
1085        //
1086        // And finally note that this operation empties out the
1087        // `defined_resources` set of the type that is registered for the
1088        // instance, as this export is modeled as producing a concrete instance.
1089        if !ty.defined_resources.is_empty() {
1090            let mut new_ty = ty.clone();
1091            let mut mapping = Remapping::default();
1092            for old in mem::take(&mut new_ty.defined_resources) {
1093                let new = types.alloc_resource_id();
1094                mapping.resources.insert(old, new.resource());
1095                self.defined_resources.insert(new.resource(), None);
1096            }
1097            for ty in new_ty.exports.values_mut() {
1098                types.remap_component_entity(ty, &mut mapping);
1099            }
1100            for (id, path) in mem::take(&mut new_ty.explicit_resources) {
1101                let id = mapping.resources.get(&id).copied().unwrap_or(id);
1102                new_ty.explicit_resources.insert(id, path);
1103            }
1104            *id = types.push_ty(new_ty);
1105        }
1106
1107        // Any explicit resources in the instance are now additionally explicit
1108        // in this component since it's exported.
1109        //
1110        // The path to each explicit resources gets one element prepended which
1111        // is `self.next_export_index`, the index of the export about to be
1112        // generated.
1113        let ty = &types[*id];
1114        for (id, path) in ty.explicit_resources.iter() {
1115            let mut new_path = vec![self.exports.len()];
1116            new_path.extend(path);
1117            self.explicit_resources.insert(*id, new_path);
1118        }
1119    }

1254    fn lift_function(
1255        &mut self,
1256        core_func_index: u32,
1257        type_index: u32,
1258        options: &[CanonicalOption],
1259        types: &mut TypeAlloc,
1260        offset: usize,
1261    ) -> Result<()> {
1262        let ty = self.function_type_at(type_index, types, offset)?;
1263        let core_ty_id = self.core_function_at(core_func_index, offset)?;
1264
1265        // Lifting a function is for an export, so match the expected canonical ABI
1266        // export signature
1267        let mut options = self.check_options(types, options, offset)?;
1268        options.check_lift(types, self, core_ty_id, offset)?;
1269        let func_ty = ty.lower(types, &options, Abi::Lift, offset)?;
1270        let lowered_core_ty_id = func_ty.intern(types, offset);
1271
1272        if core_ty_id == lowered_core_ty_id {
1273            self.funcs
1274                .push(self.types[type_index as usize].unwrap_func());
1275            return Ok(());
1276        }
1277
1278        let ty = types[core_ty_id].unwrap_func();
1279        let lowered_ty = types[lowered_core_ty_id].unwrap_func();
1280
1281        if lowered_ty.params() != ty.params() {
1282            bail!(
1283                offset,
1284                "lowered parameter types `{:?}` do not match parameter types `{:?}` of \
1285                     core function {core_func_index}",
1286                lowered_ty.params(),
1287                ty.params()
1288            );
1289        }
1290
1291        if lowered_ty.results() != ty.results() {
1292            bail!(
1293                offset,
1294                "lowered result types `{:?}` do not match result types `{:?}` of \
1295                     core function {core_func_index}",
1296                lowered_ty.results(),
1297                ty.results()

1399            info: TypeInfo::new(),
1400            params: result
1401                .iter()
1402                .map(|ty| {
1403                    Ok((
1404                        KebabString::new("v").unwrap(),
1405                        match ty {
1406                            crate::ComponentValType::Primitive(ty) => {
1407                                ComponentValType::Primitive(*ty)
1408                            }

1410                                ComponentValType::Type(self.defined_type_at(*index, offset)?)
1411                            }
1412                        },
1413                    ))
1414                })
1415                .collect::<Result<_>>()?,
1416            result: None,
1417        };
1418

1420        if options.realloc.is_some() {
1421            bail!(offset, "cannot specify `realloc` option on `task.return`")

1425                offset,
1426                "cannot specify `post-return` option on `task.return`"
1427            )
1428        }
1429        options.check_lower(offset)?;
1430        options.require_sync(offset, "task.return")?;
1431

1433        let ty_id = func_ty.intern(types, offset);

2357    fn check_options(
2358        &self,
2359        types: &TypeList,
2360        options: &[CanonicalOption],
2361        offset: usize,
2362    ) -> Result<CanonicalOptions> {
2363        fn display(option: CanonicalOption) -> &'static str {
2364            match option {
2365                CanonicalOption::UTF8 => "utf8",
2366                CanonicalOption::UTF16 => "utf16",
2367                CanonicalOption::CompactUTF16 => "latin1-utf16",
2368                CanonicalOption::Memory(_) => "memory",
2369                CanonicalOption::Realloc(_) => "realloc",
2370                CanonicalOption::PostReturn(_) => "post-return",
2371                CanonicalOption::Async => "async",
2372                CanonicalOption::Callback(_) => "callback",
2373                CanonicalOption::CoreType(_) => "core-type",
2374                CanonicalOption::Gc => "gc",
2375            }
2376        }
2377
2378        let mut encoding = None;
2379        let mut memory = None;
2380        let mut realloc = None;
2381        let mut post_return = None;
2382        let mut is_async = false;
2383        let mut callback = None;
2384        let mut core_type = None;
2385        let mut gc = false;
2386
2387        for option in options {
2388            match option {
2389                CanonicalOption::UTF8 | CanonicalOption::UTF16 | CanonicalOption::CompactUTF16 => {
2390                    match encoding {
2391                        Some(existing) => {
2392                            bail!(
2393                                offset,
2394                                "canonical encoding option `{existing}` conflicts with option `{}`",
2395                                display(*option),
2396                            )
2397                        }
2398                        None => {
2399                            encoding = Some(match option {
2400                                CanonicalOption::UTF8 => StringEncoding::Utf8,
2401                                CanonicalOption::UTF16 => StringEncoding::Utf16,
2402                                CanonicalOption::CompactUTF16 => StringEncoding::CompactUtf16,
2403                                _ => unreachable!(),
2404                            });
2405                        }
2406                    }
2407                }
2408                CanonicalOption::Memory(idx) => {
2409                    memory = match memory {
2410                        None => {
2411                            self.cabi_memory_at(*idx, offset)?;
2412                            Some(*idx)
2413                        }
2414                        Some(_) => {
2415                            return Err(BinaryReaderError::new(
2416                                "canonical option `memory` is specified more than once",
2417                                offset,
2418                            ));
2419                        }
2420                    }
2421                }
2422                CanonicalOption::Realloc(idx) => {
2423                    realloc = match realloc {
2424                        None => {
2425                            let ty_id = self.core_function_at(*idx, offset)?;
2426                            let func_ty = types[ty_id].unwrap_func();
2427                            if func_ty.params()
2428                                != [ValType::I32, ValType::I32, ValType::I32, ValType::I32]
2429                                || func_ty.results() != [ValType::I32]
2430                            {
2431                                return Err(BinaryReaderError::new(
2432                                    "canonical option `realloc` uses a core function with an incorrect signature",
2433                                    offset,
2434                                ));
2435                            }
2436                            Some(*idx)
2437                        }
2438                        Some(_) => {
2439                            return Err(BinaryReaderError::new(
2440                                "canonical option `realloc` is specified more than once",
2441                                offset,
2442                            ));
2443                        }
2444                    }
2445                }
2446                CanonicalOption::PostReturn(idx) => {
2447                    post_return = match post_return {
2448                        None => Some(*idx),
2449                        Some(_) => {
2450                            return Err(BinaryReaderError::new(
2451                                "canonical option `post-return` is specified more than once",
2452                                offset,
2453                            ));
2454                        }
2455                    }
2456                }
2457                CanonicalOption::Async => {
2458                    if is_async {
2459                        return Err(BinaryReaderError::new(
2460                            "canonical option `async` is specified more than once",
2461                            offset,
2462                        ));
2463                    } else {
2464                        if !self.features.cm_async() {
2465                            bail!(
2466                                offset,
2467                                "canonical option `async` requires the component model async feature"
2468                            );
2469                        }
2470
2471                        is_async = true;
2472                    }
2473                }
2474                CanonicalOption::Callback(idx) => {
2475                    callback = match callback {
2476                        None => Some(*idx),
2477                        Some(_) => {
2478                            return Err(BinaryReaderError::new(
2479                                "canonical option `callback` is specified more than once",
2480                                offset,
2481                            ));
2482                        }
2483                    }
2484                }
2485                CanonicalOption::CoreType(idx) => {
2486                    core_type = match core_type {
2487                        None => {
2488                            if !self.features.cm_gc() {
2489                                bail!(
2490                                    offset,
2491                                    "canonical option `core type` requires the component model gc feature"
2492                                )
2493                            }
2494                            let ty = match self.core_type_at(*idx, offset)? {
2495                                ComponentCoreTypeId::Sub(ty) => ty,
2496                                ComponentCoreTypeId::Module(_) => {
2497                                    return Err(BinaryReaderError::new(
2498                                        "canonical option `core type` must reference a core function \
2499                                     type",
2500                                        offset,
2501                                    ));
2502                                }
2503                            };
2504                            match &types[ty].composite_type.inner {
2505                                CompositeInnerType::Func(_) => {}
2506                                CompositeInnerType::Array(_)
2507                                | CompositeInnerType::Struct(_)
2508                                | CompositeInnerType::Cont(_) => {
2509                                    return Err(BinaryReaderError::new(
2510                                        "canonical option `core type` must reference a core function \
2511                                     type",
2512                                        offset,
2513                                    ));
2514                                }
2515                            }
2516                            Some(ty)
2517                        }
2518                        Some(_) => {
2519                            return Err(BinaryReaderError::new(
2520                                "canonical option `core type` is specified more than once",
2521                                offset,
2522                            ));
2523                        }
2524                    };
2525                }
2526                CanonicalOption::Gc => {
2527                    if gc {
2528                        return Err(BinaryReaderError::new(
2529                            "canonical option `gc` is specified more than once",
2530                            offset,
2531                        ));
2532                    }
2533                    if !self.features.cm_gc() {
2534                        return Err(BinaryReaderError::new(
2535                            "canonical option `gc` requires the `cm-gc` feature",
2536                            offset,
2537                        ));
2538                    }
2539                    gc = true;
2540                }
2541            }
2542        }
2543

2545
2546        let concurrency = match (is_async, callback, post_return.is_some()) {
2547            (false, Some(_), _) => {
2548                bail!(offset, "cannot specify callback without async")
2549            }
2550            (true, _, true) => {
2551                bail!(offset, "cannot specify post-return function in async")
2552            }
2553            (false, None, _) => Concurrency::Sync,
2554            (true, callback, false) => Concurrency::Async { callback },
2555        };
2556
2557        if !gc && core_type.is_some() {
2558            bail!(offset, "cannot specify `core-type` without `gc`")
2559        }
2560
2561        Ok(CanonicalOptions {
2562            string_encoding,
2563            memory,
2564            realloc,
2565            post_return,
2566            concurrency,
2567            core_type,
2568            gc,
2569        })
2570    }

2658                    referenced,
2659                    created,
2660                }
2661            }
2662            ComponentExternalKind::Instance => {
2663                ComponentEntityType::Instance(self.instance_at(export.index, offset)?)
2664            }
2665            ComponentExternalKind::Component => {
2666                ComponentEntityType::Component(self.component_at(export.index, offset)?)
2667            }
2668        };

2671            Some(ty) => self.check_type_ref(ty, types, offset)?,
2672            None => return Ok(actual),
2673        };

2676            .component_entity_type(&actual, &ascribed, offset)

2970    fn instantiate_component(
2971        &mut self,
2972        component_index: u32,
2973        component_args: Vec<crate::ComponentInstantiationArg>,
2974        types: &mut TypeAlloc,
2975        offset: usize,
2976    ) -> Result<ComponentInstanceTypeId> {
2977        let component_type_id = self.component_at(component_index, offset)?;
2978        let mut args = IndexMap::default();
2979
2980        // Populate the arguments
2981        for component_arg in component_args {
2982            let ty = match component_arg.kind {
2983                ComponentExternalKind::Module => {
2984                    ComponentEntityType::Module(self.module_at(component_arg.index, offset)?)
2985                }
2986                ComponentExternalKind::Component => {
2987                    ComponentEntityType::Component(self.component_at(component_arg.index, offset)?)
2988                }
2989                ComponentExternalKind::Instance => {
2990                    ComponentEntityType::Instance(self.instance_at(component_arg.index, offset)?)
2991                }
2992                ComponentExternalKind::Func => {
2993                    ComponentEntityType::Func(self.function_at(component_arg.index, offset)?)
2994                }
2995                ComponentExternalKind::Value => {
2996                    self.check_value_support(offset)?;
2997                    ComponentEntityType::Value(*self.value_at(component_arg.index, offset)?)
2998                }
2999                ComponentExternalKind::Type => {
3000                    let ty = self.component_type_at(component_arg.index, offset)?;
3001                    ComponentEntityType::Type {
3002                        referenced: ty,
3003                        created: ty,
3004                    }
3005                }
3006            };
3007            match args.entry(component_arg.name.to_string()) {
3008                Entry::Occupied(e) => {
3009                    bail!(
3010                        offset,
3011                        "instantiation argument `{name}` conflicts with previous argument `{prev}`",
3012                        prev = e.key(),
3013                        name = component_arg.name
3014                    );
3015                }
3016                Entry::Vacant(e) => {
3017                    e.insert(ty);
3018                }
3019            }
3020        }
3021
3022        // Here comes the fun part of the component model, we're instantiating
3023        // the component with type `component_type_id` with the `args`
3024        // specified. Easy enough!
3025        //
3026        // This operation, however, is one of the lynchpins of safety in the
3027        // component model. Additionally what this ends up implementing ranges
3028        // from "well just check the types are equal" to "let's have a
3029        // full-blown ML-style module type system in the component model". There
3030        // are primarily two major tricky pieces to the component model which
3031        // make this operation, instantiating components, hard:
3032        //
3033        // 1. Components can import and exports other components. This means
3034        //    that arguments to instantiation are along the lines of functions
3035        //    being passed to functions or similar. Effectively this means that
3036        //    the term "variance" comes into play with either contravariance
3037        //    or covariance depending on where you are in typechecking. This is
3038        //    one of the main rationales, however, that this check below is a
3039        //    check for subtyping as opposed to exact type equivalence. For
3040        //    example an instance that exports something is a subtype of an
3041        //    instance that exports nothing. Components get a bit trick since
3042        //    they both have imports and exports. My way of thinking about it
3043        //    is "who's asking for what". If you're asking for imports then
3044        //    I need to at least supply those imports, but I can possibly
3045        //    supply more. If you're asking for a thing which you'll give a set
3046        //    of imports, then I can give you something which takes less imports
3047        //    because what you give still suffices. (things like that). The
3048        //    real complication with components, however, comes with...
3049        //
3050        // 2. Resources. Resources in the component model are akin to "abstract
3051        //    types". They're not abstract in the sense that they have no
3052        //    representation, they're always backed by a 32-bit integer right
3053        //    now. Instead they're abstract in the sense that some components
3054        //    aren't allowed to understand the representation of a resource.
3055        //    For example if you import a resource you can't get the underlying
3056        //    internals of it. Furthermore the resource is strictly tracked
3057        //    within the component with `own` and `borrow` runtime semantics.
3058        //    The hardest part about resources, though, is handling them as
3059        //    part of instantiation and subtyping.
3060        //
3061        //    For example one major aspect of resources is that if a component
3062        //    exports a resource then each instantiation of the component
3063        //    produces a fresh resource type. This means that the type recorded
3064        //    for the instantiation here can't simply be "I instantiated
3065        //    component X" since in such a situation the type of all
3066        //    instantiations would be the same, which they aren't.
3067        //
3068        //    This sort of subtlety comes up quite frequently for resources.
3069        //    This file contains references to `imported_resources` and
3070        //    `defined_resources` for example which refer to the formal
3071        //    nature of components and their abstract variables. Specifically
3072        //    for instantiation though we're eventually faced with the problem
3073        //    of subtype checks where resource subtyping is defined as "does
3074        //    your id equal mine". Naively implemented that means anything with
3075        //    resources isn't subtypes of anything else since resource ids are
3076        //    unique between components. Instead what actually needs to happen
3077        //    is types need to be substituted.
3078        //
3079        // Much of the complexity here is not actually apparent here in this
3080        // literal one function. Instead it's spread out across validation
3081        // in this file and type-checking in the `types.rs` module. Note that
3082        // the "spread out" nature isn't because we're bad maintainers
3083        // (hopefully), but rather it's quite infectious how many parts need
3084        // to handle resources and account for defined/imported variables.
3085        //
3086        // For example only one subtyping method is called here where `args` is
3087        // passed in. This method is quite recursive in its nature though and
3088        // will internally touch all the fields that this file maintains to
3089        // end up putting into various bits and pieces of type information.
3090        //
3091        // Unfortunately there's probably not really a succinct way to read
3092        // this method and understand everything. If you've written ML module
3093        // type systems this will probably look quite familiar, but otherwise
3094        // the whole system is not really easily approachable at this time. It's
3095        // hoped in the future that there's a formalism to refer to which will
3096        // make things more clear as the code would be able to reference this
3097        // hypothetical formalism. Until that's the case, though, these
3098        // comments are hopefully enough when augmented with communication with
3099        // the authors.
3100
3101        let component_type = &types[component_type_id];
3102        let mut exports = component_type.exports.clone();
3103        let mut info = TypeInfo::new();
3104        for (_, ty) in component_type.exports.iter() {
3105            info.combine(ty.info(types), offset)?;
3106        }
3107
3108        // Perform the subtype check that `args` matches the imports of
3109        // `component_type_id`. The result of this subtype check is the
3110        // production of a mapping of resource types from the imports to the
3111        // arguments provided. This is a substitution map which is then used
3112        // below to perform a substitution into the exports of the instance
3113        // since the types of the exports are now in terms of whatever was
3114        // supplied as imports.
3115        let mut mapping = SubtypeCx::new(types, types).open_instance_type(
3116            &args,
3117            component_type_id,
3118            ExternKind::Import,
3119            offset,
3120        )?;
3121
3122        // Part of the instantiation of a component is that all of its
3123        // defined resources become "fresh" on each instantiation. This
3124        // means that each instantiation of a component gets brand new type
3125        // variables representing its defined resources, modeling that each
3126        // instantiation produces distinct types. The freshening is performed
3127        // here by allocating new ids and inserting them into `mapping`.
3128        //
3129        // Note that technically the `mapping` from subtyping should be applied
3130        // first and then the mapping for freshening should be applied
3131        // afterwards. The keys of the map from subtyping are the imported
3132        // resources from this component which are disjoint from its defined
3133        // resources. That means it should be possible to place everything
3134        // into one large map which maps from:
3135        //
3136        // * the component's imported resources go to whatever was explicitly
3137        //   supplied in the import map
3138        // * the component's defined resources go to fresh new resources
3139        //
3140        // These two remapping operations can then get folded into one by
3141        // placing everything in the same `mapping` and using that for a remap
3142        // only once.
3143        let fresh_defined_resources = (0..component_type.defined_resources.len())
3144            .map(|_| types.alloc_resource_id().resource())
3145            .collect::<IndexSet<_>>();
3146        let component_type = &types[component_type_id];
3147        for ((old, _path), new) in component_type
3148            .defined_resources
3149            .iter()
3150            .zip(&fresh_defined_resources)
3151        {
3152            let prev = mapping.resources.insert(*old, *new);
3153            assert!(prev.is_none());
3154        }
3155
3156        // Perform the remapping operation over all the exports that will be
3157        // listed for the final instance type. Note that this is performed
3158        // both for all the export types in addition to the explicitly exported
3159        // resources list.
3160        //
3161        // Note that this is a crucial step of the instantiation process which
3162        // is intentionally transforming the type of a component based on the
3163        // variables provided by imports and additionally ensuring that all
3164        // references to the component's defined resources are rebound to the
3165        // fresh ones introduced just above.
3166        for entity in exports.values_mut() {
3167            types.remap_component_entity(entity, &mut mapping);
3168        }
3169        let component_type = &types[component_type_id];
3170        let explicit_resources = component_type
3171            .explicit_resources
3172            .iter()
3173            .map(|(id, path)| {
3174                (
3175                    mapping.resources.get(id).copied().unwrap_or(*id),
3176                    path.clone(),
3177                )
3178            })
3179            .collect::<IndexMap<_, _>>();
3180
3181        // Technically in the last formalism that was consulted in writing this
3182        // implementation there are two further steps that are part of the
3183        // instantiation process:
3184        //
3185        // 1. The set of defined resources from the instance created, which are
3186        //    added to the outer component, is the subset of the instance's
3187        //    original defined resources and the free variables of the exports.
3188        //
3189        // 2. Each element of this subset is required to be "explicit in" the
3190        //    instance, or otherwise explicitly exported somewhere within the
3191        //    instance.
3192        //
3193        // With the syntactic structure of the component model, however, neither
3194        // of these conditions should be necessary. The main reason for this is
3195        // that this function is specifically dealing with instantiation of
3196        // components which should already have these properties validated
3197        // about them. Subsequently we shouldn't have to re-check them.
3198        //
3199        // In debug mode, however, do a sanity check.
3200        if cfg!(debug_assertions) {
3201            let mut free = IndexSet::default();
3202            for ty in exports.values() {
3203                types.free_variables_component_entity(ty, &mut free);
3204            }
3205            assert!(fresh_defined_resources.is_subset(&free));
3206            for resource in fresh_defined_resources.iter() {
3207                assert!(explicit_resources.contains_key(resource));
3208            }
3209        }
3210
3211        // And as the final step of the instantiation process all of the
3212        // new defined resources from this component instantiation are moved
3213        // onto `self`. Note that concrete instances never have defined
3214        // resources (see more comments in `instantiate_exports`) so the
3215        // `defined_resources` listing in the final type is always empty. This
3216        // represents how by having a concrete instance the definitions
3217        // referred to in that instance are now problems for the outer
3218        // component rather than the inner instance since the instance is bound
3219        // to the component.
3220        //
3221        // All defined resources here have no known representation, so they're
3222        // all listed with `None`. Also note that none of the resources were
3223        // exported yet so `self.explicit_resources` is not updated yet. If
3224        // this instance is exported, however, it'll consult the type's
3225        // `explicit_resources` array and use that appropriately.
3226        for resource in fresh_defined_resources {
3227            self.defined_resources.insert(resource, None);
3228        }
3229
3230        Ok(types.push_ty(ComponentInstanceType {
3231            info,
3232            defined_resources: Default::default(),
3233            explicit_resources,
3234            exports,
3235        }))
3236    }

3238    fn instantiate_component_exports(
3239        &mut self,
3240        exports: Vec<crate::ComponentExport>,
3241        types: &mut TypeAlloc,
3242        offset: usize,
3243    ) -> Result<ComponentInstanceTypeId> {
3244        let mut info = TypeInfo::new();
3245        let mut inst_exports = IndexMap::default();
3246        let mut explicit_resources = IndexMap::default();
3247        let mut export_names = IndexSet::default();
3248
3249        // NB: It's intentional that this context is empty since no indices are
3250        // introduced in the bag-of-exports construct which means there's no
3251        // way syntactically to register something inside of this.
3252        let names = ComponentNameContext::default();
3253
3254        for export in exports {
3255            assert!(export.ty.is_none());
3256            let ty = match export.kind {
3257                ComponentExternalKind::Module => {
3258                    ComponentEntityType::Module(self.module_at(export.index, offset)?)
3259                }
3260                ComponentExternalKind::Component => {
3261                    ComponentEntityType::Component(self.component_at(export.index, offset)?)
3262                }
3263                ComponentExternalKind::Instance => {
3264                    let ty = self.instance_at(export.index, offset)?;
3265
3266                    // When an instance is exported from an instance then
3267                    // all explicitly exported resources on the sub-instance are
3268                    // now also listed as exported resources on the outer
3269                    // instance, just with one more element in their path.
3270                    explicit_resources.extend(types[ty].explicit_resources.iter().map(
3271                        |(id, path)| {

3274                            (*id, new_path)
3275                        },
3276                    ));
3277                    ComponentEntityType::Instance(ty)
3278                }
3279                ComponentExternalKind::Func => {
3280                    ComponentEntityType::Func(self.function_at(export.index, offset)?)
3281                }
3282                ComponentExternalKind::Value => {
3283                    self.check_value_support(offset)?;
3284                    ComponentEntityType::Value(*self.value_at(export.index, offset)?)
3285                }
3286                ComponentExternalKind::Type => {
3287                    let ty = self.component_type_at(export.index, offset)?;
3288                    // If this is an export of a resource type be sure to
3289                    // record that in the explicit list with the appropriate
3290                    // path because if this instance ends up getting used
3291                    // it'll count towards the "explicit in" check.
3292                    if let ComponentAnyTypeId::Resource(id) = ty {
3293                        explicit_resources.insert(id.resource(), vec![inst_exports.len()]);
3294                    }
3295                    ComponentEntityType::Type {
3296                        referenced: ty,
3297                        // The created type index here isn't used anywhere
3298                        // in index spaces because a "bag of exports"
3299                        // doesn't build up its own index spaces. Just fill
3300                        // in the same index here in this case as what's
3301                        // referenced.
3302                        created: ty,
3303                    }
3304                }
3305            };

3316                &self.features,

3320        Ok(types.push_ty(ComponentInstanceType {
3321            info,
3322            explicit_resources,
3323            exports: inst_exports,
3324
3325            // NB: the list of defined resources for this instance itself
3326            // is always empty. Even if this instance exports resources,
3327            // it's empty.
3328            //
3329            // The reason for this is a bit subtle. The general idea, though, is
3330            // that the defined resources list here is only used for instance
3331            // types that are sort of "floating around" and haven't actually
3332            // been attached to something yet. For example when an instance type
3333            // is simply declared it can have defined resources introduced
3334            // through `(export "name" (type (sub resource)))`. These
3335            // definitions, however, are local to the instance itself and aren't
3336            // defined elsewhere.
3337            //
3338            // Here, though, no new definitions were introduced. The instance
3339            // created here is a "bag of exports" which could only refer to
3340            // preexisting items. This means that inherently no new resources
3341            // were created so there's nothing to put in this list. Any
3342            // resources referenced by the instance must be bound by the outer
3343            // component context or further above.
3344            //
3345            // Furthermore, however, actual instances of instances, which this
3346            // is, aren't allowed to have defined resources. Instead the
3347            // resources would have to be injected into the outer component
3348            // enclosing the instance. That means that even if bag-of-exports
3349            // could declare a new resource then the resource would be moved
3350            // from here to `self.defined_resources`. This doesn't exist at this
3351            // time, though, so this still remains empty and
3352            // `self.defined_resources` remains unperturbed.
3353            defined_resources: Default::default(),
3354        }))
3355    }

3633    fn alias_type(
3634        components: &mut [Self],
3635        count: u32,
3636        index: u32,
3637        types: &mut TypeAlloc,
3638        offset: usize,
3639    ) -> Result<()> {
3640        let component = Self::check_alias_count(components, count, offset)?;
3641        let ty = component.component_type_at(index, offset)?;
3642
3643        // If `count` "crossed a component boundary", meaning that it went from
3644        // one component to another, then this must additionally verify that
3645        // `ty` has no free variables with respect to resources. This is
3646        // intended to preserve the property for components where each component
3647        // is an isolated unit that can theoretically be extracted from other
3648        // components. If resources from other components were allowed to leak
3649        // in then it would prevent that.
3650        //
3651        // This check is done by calculating the `pos` within `components` that
3652        // our target `component` above was selected at. Once this is acquired
3653        // the component to the "right" is checked, and if that's a component
3654        // then it's considered as crossing a component boundary meaning the
3655        // free variables check runs.
3656        //
3657        // The reason this works is that in the list of `ComponentState` types
3658        // it's guaranteed that any `is_type` components are contiguous at the
3659        // end of the array. This means that if state one level deeper than the
3660        // target of this alias is a `!is_type` component, then the target must
3661        // be a component as well. If the one-level deeper state `is_type` then
3662        // the target is either a type or a component, both of which are valid
3663        // (as aliases can reach the enclosing component and have as many free
3664        // variables as they want).
3665        let pos_after_component = components.len() - (count as usize);
3666        if let Some(component) = components.get(pos_after_component) {
3667            if component.kind == ComponentKind::Component {
3668                let mut free = IndexSet::default();
3669                types.free_variables_any_type_id(ty, &mut free);
3670                if !free.is_empty() {
3671                    bail!(
3672                        offset,
3673                        "cannot alias outer type which transitively refers \
3674                         to resources not defined in the current component"
3675                    );
3676                }
3677            }
3678        }
3679
3680        let current = components.last_mut().unwrap();
3681        check_max(current.type_count(), 1, MAX_WASM_TYPES, "types", offset)?;
3682
3683        current.types.push(ty);
3684
3685        Ok(())
3686    }

4156    /// Completes the translation of this component, performing final
4157    /// validation of its structure.
4158    ///
4159    /// This method is required to be called for translating all components.
4160    /// Internally this will convert local data structures into a
4161    /// `ComponentType` which is suitable to use to describe the type of this
4162    /// component.
4163    pub fn finish(&mut self, types: &TypeAlloc, offset: usize) -> Result<ComponentType> {
4164        let mut ty = ComponentType {
4165            // Inherit some fields based on translation of the component.
4166            info: self.type_info,
4167            imports: self.imports.clone(),
4168            exports: self.exports.clone(),
4169
4170            // This is filled in as a subset of `self.defined_resources`
4171            // depending on what's actually used by the exports. See the
4172            // bottom of this function.
4173            defined_resources: Default::default(),
4174
4175            // These are inherited directly from what was calculated for this
4176            // component.
4177            imported_resources: mem::take(&mut self.imported_resources)
4178                .into_iter()
4179                .collect(),
4180            explicit_resources: mem::take(&mut self.explicit_resources),
4181        };
4182
4183        // Collect all "free variables", or resources, from the imports of this
4184        // component. None of the resources defined within this component can
4185        // be used as part of the exports. This set is then used to reject any
4186        // of `self.defined_resources` which show up.
4187        let mut free = IndexSet::default();
4188        for ty in ty.imports.values() {
4189            types.free_variables_component_entity(ty, &mut free);
4190        }
4191        for (resource, _path) in self.defined_resources.iter() {
4192            // FIXME: this error message is quite opaque and doesn't indicate
4193            // more contextual information such as:
4194            //
4195            // * what was the exported resource found in the imports
4196            // * which import was the resource found within
4197            //
4198            // These are possible to calculate here if necessary, however.
4199            if free.contains(resource) {
4200                bail!(offset, "local resource type found in imports");
4201            }
4202        }
4203
4204        // The next step in validation a component, with respect to resources,
4205        // is to minimize the set of defined resources to only those that
4206        // are actually used by the exports. This weeds out resources that are
4207        // defined, used within a component, and never exported, for example.
4208        //
4209        // The free variables of all exports are inserted into the `free` set
4210        // (which is reused from the imports after clearing it). The defined
4211        // resources calculated for this component are then inserted into this
4212        // type's list of defined resources if it's contained somewhere in
4213        // the free variables.
4214        //
4215        // Note that at the same time all defined resources must be exported,
4216        // somehow, transitively from this component. The `explicit_resources`
4217        // map is consulted for this purpose which lists all explicitly
4218        // exported resources in the component, regardless from whence they
4219        // came. If not present in this map then it's not exported and an error
4220        // is returned.
4221        //
4222        // NB: the "types are exported" check is probably sufficient nowadays
4223        // that the check of the `explicit_resources` map is probably not
4224        // necessary, but it's left here for completeness and out of an
4225        // abundance of caution.
4226        free.clear();
4227        for ty in ty.exports.values() {
4228            types.free_variables_component_entity(ty, &mut free);
4229        }
4230        for (id, _rep) in mem::take(&mut self.defined_resources) {
4231            if !free.contains(&id) {
4232                continue;
4233            }
4234
4235            let path = match ty.explicit_resources.get(&id).cloned() {
4236                Some(path) => path,
4237                // FIXME: this error message is quite opaque and doesn't
4238                // indicate more contextual information such as:
4239                //
4240                // * which resource wasn't found in an export
4241                // * which export has a reference to the resource
4242                //
4243                // These are possible to calculate here if necessary, however.
4244                None => bail!(
4245                    offset,
4246                    "local resource type found in export but not exported itself"
4247                ),
4248            };
4249
4250            ty.defined_resources.push((id, path));
4251        }
4252
4253        Ok(ty)
4254    }

4291    /// Registers that the resource `id` is named `name` within this context.
4292    fn register(&mut self, name: &str, id: AliasableResourceId) {
4293        let idx = self.all_resource_names.len();
4294        let prev = self.resource_name_map.insert(id, idx);
4295        assert!(
4296            prev.is_none(),

4302    fn validate_extern(
4303        &self,
4304        name: &str,
4305        kind: ExternKind,
4306        ty: &ComponentEntityType,
4307        types: &TypeAlloc,
4308        offset: usize,
4309        kind_names: &mut IndexSet<ComponentName>,
4310        items: &mut IndexMap<String, ComponentEntityType>,
4311        info: &mut TypeInfo,
4312        features: &WasmFeatures,
4313    ) -> Result<()> {
4314        // First validate that `name` is even a valid kebab name, meaning it's
4315        // in kebab-case, is an ID, etc.
4316        let kebab = ComponentName::new_with_features(name, offset, *features)
4317            .with_context(|| format!("{} name `{name}` is not a valid extern name", kind.desc()))?;
4318
4319        if let ExternKind::Export = kind {
4320            match kebab.kind() {
4321                ComponentNameKind::Label(_)
4322                | ComponentNameKind::AsyncLabel(_)
4323                | ComponentNameKind::Method(_)
4324                | ComponentNameKind::AsyncMethod(_)
4325                | ComponentNameKind::Static(_)
4326                | ComponentNameKind::AsyncStatic(_)
4327                | ComponentNameKind::Constructor(_)
4328                | ComponentNameKind::Interface(_) => {}
4329
4330                ComponentNameKind::Hash(_)
4331                | ComponentNameKind::Url(_)
4332                | ComponentNameKind::Dependency(_) => {
4333                    bail!(offset, "name `{name}` is not a valid export name")
4334                }
4335            }
4336        }
4337
4338        // Validate that the kebab name, if it has structure such as
4339        // `[method]a.b`, is indeed valid with respect to known resources.
4340        self.validate(&kebab, ty, types, offset, features)
4341            .with_context(|| format!("{} name `{kebab}` is not valid", kind.desc()))?;
4342
4343        // Top-level kebab-names must all be unique, even between both imports
4344        // and exports ot a component. For those names consult the `kebab_names`
4345        // set.
4346        if let Some(prev) = kind_names.replace(kebab.clone()) {
4347            bail!(
4348                offset,
4349                "{} name `{kebab}` conflicts with previous name `{prev}`",
4350                kind.desc()
4351            );
4352        }
4353
4354        // Otherwise all strings must be unique, regardless of their name, so
4355        // consult the `items` set to ensure that we're not for example
4356        // importing the same interface ID twice.
4357        match items.entry(name.to_string()) {
4358            Entry::Occupied(e) => {
4359                bail!(
4360                    offset,
4361                    "{kind} name `{name}` conflicts with previous name `{prev}`",
4362                    kind = kind.desc(),
4363                    prev = e.key(),
4364                );
4365            }
4366            Entry::Vacant(e) => {
4367                e.insert(*ty);
4368                info.combine(ty.info(types), offset)?;
4369            }
4370        }
4371        Ok(())
4372    }

4374    /// Validates that the `name` provided is allowed to have the type `ty`.
4375    fn validate(
4376        &self,
4377        name: &ComponentName,
4378        ty: &ComponentEntityType,
4379        types: &TypeAlloc,
4380        offset: usize,
4381        features: &WasmFeatures,
4382    ) -> Result<()> {
4383        let func = || {
4384            let id = match ty {
4385                ComponentEntityType::Func(id) => *id,
4386                _ => bail!(offset, "item is not a func"),
4387            };
4388            Ok(&types[id])
4389        };
4390        match name.kind() {
4391            ComponentNameKind::AsyncLabel(_)
4392            | ComponentNameKind::AsyncMethod(_)
4393            | ComponentNameKind::AsyncStatic(_) => {
4394                if !features.cm_async() {
4395                    bail!(
4396                        offset,
4397                        "async kebab-names require the component model async feature"
4398                    );
4399                }
4400            }
4401            _ => {}
4402        }
4403
4404        match name.kind() {
4405            // No validation necessary for these styles of names
4406            ComponentNameKind::Label(_)
4407            | ComponentNameKind::AsyncLabel(_)
4408            | ComponentNameKind::Interface(_)
4409            | ComponentNameKind::Url(_)
4410            | ComponentNameKind::Dependency(_)
4411            | ComponentNameKind::Hash(_) => {}
4412
4413            // Constructors must return `(own $resource)` and the `$resource`
4414            // must be named within this context to match `rname`
4415            ComponentNameKind::Constructor(rname) => {
4416                let ty = func()?;
4417                let ty = match ty.result {
4418                    Some(result) => result,
4419                    None => bail!(offset, "function should return one value"),
4420                };
4421                let resource = match ty {
4422                    ComponentValType::Primitive(_) => None,
4423                    ComponentValType::Type(ty) => match &types[ty] {
4424                        ComponentDefinedType::Own(id) => Some(id),
4425                        _ => None,
4426                    },
4427                };
4428                let resource = match resource {
4429                    Some(id) => id,
4430                    None => bail!(offset, "function should return `(own $T)`"),
4431                };
4432                self.validate_resource_name(*resource, rname, offset)?;
4433            }
4434
4435            // Methods must take `(param "self" (borrow $resource))` as the
4436            // first argument where `$resources` matches the name `resource` as
4437            // named in this context.
4438            ComponentNameKind::Method(name) | ComponentNameKind::AsyncMethod(name) => {
4439                let ty = func()?;
4440                if ty.params.len() == 0 {
4441                    bail!(offset, "function should have at least one argument");
4442                }
4443                let (pname, pty) = &ty.params[0];
4444                if pname.as_str() != "self" {
4445                    bail!(
4446                        offset,
4447                        "function should have a first argument called `self`",
4448                    );
4449                }
4450                let id = match pty {
4451                    ComponentValType::Primitive(_) => None,
4452                    ComponentValType::Type(ty) => match &types[*ty] {
4453                        ComponentDefinedType::Borrow(id) => Some(id),
4454                        _ => None,
4455                    },
4456                };
4457                let id = match id {
4458                    Some(id) => id,
4459                    None => bail!(
4460                        offset,
4461                        "function should take a first argument of `(borrow $T)`"
4462                    ),
4463                };
4464                self.validate_resource_name(*id, name.resource(), offset)?;
4465            }
4466
4467            // Static methods don't have much validation beyond that they must
4468            // be a function and the resource name referred to must already be
4469            // in this context.
4470            ComponentNameKind::Static(name) | ComponentNameKind::AsyncStatic(name) => {
4471                func()?;
4472                if !self.all_resource_names.contains(name.resource().as_str()) {
4473                    bail!(offset, "static resource name is not known in this context");
4474                }
4475            }
4476        }
4477
4478        Ok(())
4479    }

4481    fn validate_resource_name(
4482        &self,
4483        id: AliasableResourceId,
4484        name: &KebabStr,
4485        offset: usize,
4486    ) -> Result<()> {
4487        let expected_name_idx = match self.resource_name_map.get(&id) {
4488            Some(idx) => *idx,
4489            None => {
4490                bail!(
4491                    offset,
4492                    "resource used in function does not have a name in this context"
4493                )
4494            }
4495        };
4496        let expected_name = &self.all_resource_names[expected_name_idx];
4497        if name.as_str() != expected_name {
4498            bail!(
4499                offset,
4500                "function does not match expected \
4501                         resource name `{expected_name}`"
4502            );
4503        }
4504        Ok(())
4505    }
