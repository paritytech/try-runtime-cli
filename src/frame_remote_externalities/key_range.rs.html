18//! Key range management for parallel storage fetching.
19
20use sp_core::storage::StorageKey;
21use std::{
22	collections::VecDeque,
23	sync::{Arc, Mutex},
24};
25
26/// Default page size for fetching keys.
27pub(crate) const DEFAULT_PAGE_SIZE: u32 = 1000;
28
29/// Represents a range of keys to fetch from the remote node.
30#[derive(Debug, Clone)]
31pub(crate) struct KeyRange {
32	/// The starting key of this range (inclusive for initial ranges, exclusive for continuations).
33	pub(crate) start_key: StorageKey,
34	/// The ending key of this range (exclusive), or None for open-ended range.
35	pub(crate) end_key: Option<StorageKey>,
36	/// The common prefix for this range.
37	pub(crate) prefix: StorageKey,
38	/// Page size for fetching keys in this range (decreases on failure).
39	pub(crate) page_size: u32,
40	/// If true, start_key was already fetched and should be excluded from results.
41	pub(crate) exclude_start_key: bool,
42}
43
44impl KeyRange {
45	pub(crate) fn new(
46		start_key: StorageKey,
47		end_key: Option<StorageKey>,
48		prefix: StorageKey,
49	) -> Self {
50		Self { start_key, end_key, prefix, page_size: DEFAULT_PAGE_SIZE, exclude_start_key: false }
51	}
52
53	pub(crate) fn continuation(
54		start_key: StorageKey,
55		end_key: Option<StorageKey>,
56		prefix: StorageKey,
57	) -> Self {
58		Self { start_key, end_key, prefix, page_size: DEFAULT_PAGE_SIZE, exclude_start_key: true }
59	}
60
61	/// Returns a new KeyRange with halved page size (minimum 10).
62	pub(crate) fn with_halved_page_size(&self) -> Self {
63		Self { page_size: (self.page_size / 2).max(10), ..self.clone() }
64	}
65
66	/// Filter keys returned from RPC to only include keys within this range.
67	///
68	/// Returns a tuple of:
69	/// - The filtered keys
70	/// - Whether this should be considered a "full batch" (triggers subdivision)
71	///
72	/// A batch is only considered "full" if AFTER filtering it still has keys.
73	/// This prevents infinite subdivision when continuation ranges filter out the start_key.
74	pub(crate) fn filter_keys(&self, keys: Vec<StorageKey>) -> (Vec<StorageKey>, bool) {
75		let rpc_returned_full_batch = keys.len() == self.page_size as usize;
76
77		let filtered: Vec<_> = keys
78			.into_iter()
79			.filter(|key| {
80				// Exclude start_key if this is a continuation range
81				if self.exclude_start_key && key == &self.start_key {
82					return false;
83				}
84				// Exclude keys at or beyond end_key
85				self.end_key.as_ref().map_or(true, |end| key < end)
86			})
87			.collect();
88
89		// Only consider it a full batch if we still have keys after filtering
90		let is_full_batch = rpc_returned_full_batch && !filtered.is_empty();
91
92		(filtered, is_full_batch)
93	}
94}

96/// Work queue for distributing key ranges to workers.
97pub(crate) type WorkQueue = Arc<Mutex<VecDeque<KeyRange>>>;
98
99/// Generate 16 key ranges for parallel fetching by dividing the key space.
100///
101/// Creates ranges by appending one nibble (4 bits) to the prefix.
102/// This gives us: 0x00, 0x10, 0x20, ..., 0xF0
103pub(crate) fn gen_key_ranges(prefix: &StorageKey) -> Vec<KeyRange> {
104	let prefix_bytes = prefix.as_ref().to_vec();
105	let mut ranges = Vec::with_capacity(16);
106
107	for i in 0u8..16u8 {
108		let mut start_key = prefix_bytes.clone();
109		start_key.push(i << 4); // Shift nibble to upper 4 bits
110
111		let end_key = if i < 15 {
112			let mut end = prefix_bytes.clone();
113			end.push((i + 1) << 4); // Next nibble
114			Some(StorageKey(end))
115		} else {
116			None
117		};
118
119		ranges.push(KeyRange::new(StorageKey(start_key), end_key, prefix.clone()));
120	}
121
122	ranges
123}
124
125/// Initialize a work queue with ranges for each prefix.
126pub(crate) fn initialize_work_queue(prefixes: &[StorageKey]) -> WorkQueue {
127	let mut queue = VecDeque::new();
128
129	for prefix in prefixes {
130		let ranges = gen_key_ranges(prefix);
131		queue.extend(ranges);
132	}
133
134	Arc::new(Mutex::new(queue))
135}

137/// Subdivide the key space AFTER the last_key into up to 16 new ranges.
138///
139/// Takes the last two keys from a batch to find where they diverge, then creates
140/// ranges based on incrementing the nibble at the divergence point.
141pub(crate) fn subdivide_remaining_range(
142	second_last_key: &StorageKey,
143	last_key: &StorageKey,
144	end_key: Option<&StorageKey>,
145	prefix: &StorageKey,
146) -> Vec<KeyRange> {
147	let second_last_bytes = second_last_key.as_ref();
148	let last_key_bytes = last_key.as_ref();
149
150	// Find the first byte position where the two keys diverge
151	let divergence_pos = second_last_bytes
152		.iter()
153		.zip(last_key_bytes.iter())
154		.position(|(a, b)| a != b)
155		.unwrap_or(second_last_bytes.len().min(last_key_bytes.len()));
156
157	let mut subdivision_nibble = None;
158
159	for subdivision_pos in divergence_pos..last_key_bytes.len() {
160		let byte = last_key_bytes[subdivision_pos];
161		let nibble = byte >> 4;
162
163		if nibble < 15 {
164			subdivision_nibble = Some((subdivision_pos, nibble));
165			break;
166		}
167	}
168
169	let mut ranges = Vec::new();
170
171	// If we found a position where we can subdivide
172	if let Some((pos, current_nibble)) = subdivision_nibble {
173		let subdivision_prefix = &last_key_bytes[..pos];
174
175		let mut end = subdivision_prefix.to_vec();
176		end.push((current_nibble + 1) << 4);
177		// Use continuation() since last_key was already fetched
178		ranges.push(KeyRange::continuation(
179			last_key.clone(),
180			Some(StorageKey(end)),
181			prefix.clone(),
182		));
183
184		// Create ranges for each nibble from (current_nibble + 1) to 0xF
185		for nibble in (current_nibble + 1)..16u8 {
186			let mut start = subdivision_prefix.to_vec();
187			start.push(nibble << 4);
188			let start_key = StorageKey(start);
189
190			// Check if this range starts at or after the end
191			if end_key.map_or(false, |ek| start_key >= *ek) {
192				break;
193			}
194
195			let chunk_end_key = if nibble < 15 {
196				let mut end = subdivision_prefix.to_vec();
197				end.push((nibble + 1) << 4);
198				let computed_end = StorageKey(end);
199
200				Some(match end_key {
201					Some(actual_end) if &computed_end > actual_end => actual_end.clone(),
202					_ => computed_end,
203				})
204			} else {
205				end_key.cloned()
206			};
207
208			// These are fresh ranges, not continuations
209			ranges.push(KeyRange::new(start_key, chunk_end_key, prefix.clone()));
210		}
211	} else if end_key.as_ref().map_or(true, |ek| last_key <= ek) {
212		// We are not yet past the end - use continuation since last_key was already fetched
213		ranges.push(KeyRange::continuation(last_key.clone(), end_key.cloned(), prefix.clone()));
214	}
215
216	ranges
217}

227	#[test]
228	fn gen_key_ranges_creates_16_non_overlapping_ranges() {
229		let prefix = key(&[0xAB, 0xCD]);
230		let ranges = gen_key_ranges(&prefix);
231
232		assert_eq!(ranges.len(), 16);
233
234		// Check first range
235		assert_eq!(ranges[0].start_key, key(&[0xAB, 0xCD, 0x00]));
236		assert_eq!(ranges[0].end_key, Some(key(&[0xAB, 0xCD, 0x10])));
237		assert!(!ranges[0].exclude_start_key);
238
239		// Check middle range
240		assert_eq!(ranges[8].start_key, key(&[0xAB, 0xCD, 0x80]));
241		assert_eq!(ranges[8].end_key, Some(key(&[0xAB, 0xCD, 0x90])));
242
243		// Check last range (open-ended)
244		assert_eq!(ranges[15].start_key, key(&[0xAB, 0xCD, 0xF0]));
245		assert_eq!(ranges[15].end_key, None);
246
247		// Verify ranges are contiguous (no gaps, no overlaps)
248		for i in 0..15 {
249			assert_eq!(ranges[i].end_key, Some(ranges[i + 1].start_key.clone()));
250		}
251	}

389	fn filter_keys_prevents_infinite_subdivision_on_empty_result() {
390		// This test verifies the bug fix: when a continuation range returns only
391		// the start_key (which gets filtered out), is_full_batch must be false
392		// to prevent infinite subdivision loops.
393		let start = key(&[0xAB, 0x12, 0x34]);
394		let prefix = key(&[0xAB]);
395
396		let mut range = KeyRange::continuation(start.clone(), None, prefix);
397		range.page_size = 1; // Simulate a range with page_size=1
398
399		// RPC returns exactly page_size keys (a "full batch"), but it's only the start_key
400		let keys = vec![start.clone()];
401
402		let (filtered, is_full_batch) = range.filter_keys(keys);
403
404		// After filtering, the page is empty
405		assert!(filtered.is_empty(), "start_key should be filtered out");
406		// CRITICAL: is_full_batch must be false to prevent infinite subdivision
407		assert!(!is_full_batch, "is_full_batch should be false when filtered result is empty");
408	}
