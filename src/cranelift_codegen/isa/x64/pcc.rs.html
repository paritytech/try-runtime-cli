12fn undefined_result(
13    ctx: &FactContext,
14    vcode: &mut VCode<Inst>,
15    dst: Writable<Gpr>,
16    reg_bits: u16,
17    result_bits: u16,
18) -> PccResult<()> {
19    check_output(ctx, vcode, dst.to_writable_reg(), &[], |_vcode| {
20        clamp_range(ctx, reg_bits, result_bits, None)
21    })
22}

24fn ensure_no_fact(vcode: &VCode<Inst>, reg: Reg) -> PccResult<()> {
25    if vcode.vreg_fact(reg.into()).is_some() {
26        Err(PccError::UnsupportedFact)
27    } else {
28        Ok(())
29    }
30}

32/// Flow-state between facts.
33#[derive(Clone, Debug, Default)]
34pub(crate) struct FactFlowState {
35    cmp_flags: Option<(Fact, Fact)>,
36}
37
38pub(crate) fn check(
39    ctx: &FactContext,
40    vcode: &mut VCode<Inst>,
41    inst_idx: InsnIndex,
42    state: &mut FactFlowState,
43) -> PccResult<()> {
44    trace!("Checking facts on inst: {:?}", vcode[inst_idx]);
45
46    // We only persist flag state for one instruction, because we
47    // can't exhaustively enumerate all flags-effecting ops; so take
48    // the `cmp_state` here and perhaps use it below but don't let it
49    // remain.
50    let _cmp_flags = state.cmp_flags.take();
51
52    match vcode[inst_idx] {
53        Inst::Args { .. } => {
54            // Defs on the args have "axiomatic facts": we trust the
55            // ABI code to pass through the values unharmed, so the
56            // facts given to us in the CLIF should still be true.
57            Ok(())
58        }
59
60        Inst::CheckedSRemSeq {
61            dst_quotient,
62            dst_remainder,
63            ..
64        } => {
65            undefined_result(ctx, vcode, dst_quotient, 64, 64)?;
66            undefined_result(ctx, vcode, dst_remainder, 64, 64)?;
67            Ok(())
68        }
69
70        Inst::CheckedSRemSeq8 { dst, .. } => undefined_result(ctx, vcode, dst, 64, 64),
71
72        Inst::MovFromPReg { dst, .. } => undefined_result(ctx, vcode, dst, 64, 64),
73        Inst::MovToPReg { .. } => Ok(()),
74
75        Inst::XmmCmove { dst, .. } => ensure_no_fact(vcode, dst.to_writable_reg().to_reg()),
76
77        Inst::StackProbeLoop { tmp, .. } => ensure_no_fact(vcode, tmp.to_reg()),
78
79        // NOTE: it's assumed that all of these cases perform 128-bit loads, but this hasn't been
80        // verified. The effect of this will be spurious PCC failures when these instructions are
81        // involved.
82        Inst::XmmRmREvex { dst, ref src2, .. }
83        | Inst::XmmUnaryRmRImmEvex {
84            dst, src: ref src2, ..
85        }
86        | Inst::XmmUnaryRmREvex {
87            dst, src: ref src2, ..
88        }
89        | Inst::XmmRmREvex3 {
90            dst,
91            src3: ref src2,
92            ..
93        } => {
94            match <&RegMem>::from(src2) {
95                RegMem::Mem { addr } => {
96                    check_load(ctx, None, addr, vcode, I8X16, 128)?;
97                }
98                RegMem::Reg { .. } => {}
99            }
100            ensure_no_fact(vcode, dst.to_writable_reg().to_reg())
101        }
102
103        Inst::CvtUint64ToFloatSeq {
104            dst,
105            tmp_gpr1,
106            tmp_gpr2,
107            ..
108        } => {
109            ensure_no_fact(vcode, dst.to_writable_reg().to_reg())?;
110            ensure_no_fact(vcode, tmp_gpr1.to_writable_reg().to_reg())?;
111            ensure_no_fact(vcode, tmp_gpr2.to_writable_reg().to_reg())?;
112            Ok(())
113        }
114
115        Inst::CvtFloatToSintSeq {
116            dst,
117            tmp_gpr,
118            tmp_xmm,
119            ..
120        } => {
121            undefined_result(ctx, vcode, dst, 64, 64)?;
122            ensure_no_fact(vcode, tmp_gpr.to_writable_reg().to_reg())?;
123            ensure_no_fact(vcode, tmp_xmm.to_writable_reg().to_reg())?;
124            Ok(())
125        }
126
127        Inst::CvtFloatToUintSeq {
128            dst,
129            tmp_gpr,
130            tmp_xmm,
131            tmp_xmm2,
132            ..
133        } => {
134            undefined_result(ctx, vcode, dst, 64, 64)?;
135            ensure_no_fact(vcode, tmp_gpr.to_writable_reg().to_reg())?;
136            ensure_no_fact(vcode, tmp_xmm.to_writable_reg().to_reg())?;
137            ensure_no_fact(vcode, tmp_xmm2.to_writable_reg().to_reg())?;
138            Ok(())
139        }
140
141        Inst::XmmMinMaxSeq { dst, .. } => ensure_no_fact(vcode, dst.to_writable_reg().to_reg()),
142
143        Inst::CallKnown { .. }
144        | Inst::ReturnCallKnown { .. }
145        | Inst::JmpKnown { .. }
146        | Inst::WinchJmpIf { .. }
147        | Inst::JmpCond { .. }
148        | Inst::JmpCondOr { .. }
149        | Inst::TrapIf { .. }
150        | Inst::TrapIfAnd { .. }
151        | Inst::TrapIfOr { .. } => Ok(()),
152        Inst::Rets { .. } => Ok(()),
153
154        Inst::ReturnCallUnknown { .. } => Ok(()),
155
156        Inst::CallUnknown { ref info } => match &info.dest {
157            RegMem::Mem { addr } => {
158                check_load(ctx, None, addr, vcode, I64, 64)?;
159                Ok(())
160            }
161            RegMem::Reg { .. } => Ok(()),
162        },
163
164        Inst::JmpTableSeq { tmp1, tmp2, .. } => {
165            ensure_no_fact(vcode, tmp1.to_reg())?;
166            ensure_no_fact(vcode, tmp2.to_reg())?;
167            Ok(())
168        }
169
170        Inst::LoadExtName { dst, .. } => {
171            ensure_no_fact(vcode, *dst.to_reg())?;
172            Ok(())
173        }
174
175        Inst::AtomicRmwSeq {
176            ref mem,
177            temp,
178            dst_old,
179            ..
180        } => {
181            ensure_no_fact(vcode, *dst_old.to_reg())?;
182            ensure_no_fact(vcode, *temp.to_reg())?;
183            check_store(ctx, None, mem, vcode, I64)?;
184            Ok(())
185        }
186
187        Inst::Atomic128RmwSeq {
188            ref mem,
189            temp_low,
190            temp_high,
191            dst_old_low,
192            dst_old_high,
193            ..
194        } => {
195            ensure_no_fact(vcode, *dst_old_low.to_reg())?;
196            ensure_no_fact(vcode, *dst_old_high.to_reg())?;
197            ensure_no_fact(vcode, *temp_low.to_reg())?;
198            ensure_no_fact(vcode, *temp_high.to_reg())?;
199            check_store(ctx, None, mem, vcode, I128)?;
200            Ok(())
201        }
202
203        Inst::Atomic128XchgSeq {
204            ref mem,
205            dst_old_low,
206            dst_old_high,
207            ..
208        } => {
209            ensure_no_fact(vcode, *dst_old_low.to_reg())?;
210            ensure_no_fact(vcode, *dst_old_high.to_reg())?;
211            check_store(ctx, None, mem, vcode, I128)?;
212            Ok(())
213        }
214
215        Inst::XmmUninitializedValue { dst } => {
216            ensure_no_fact(vcode, dst.to_writable_reg().to_reg())
217        }
218
219        Inst::GprUninitializedValue { dst } => {
220            ensure_no_fact(vcode, dst.to_writable_reg().to_reg())
221        }
222
223        Inst::ElfTlsGetAddr { dst, .. } | Inst::MachOTlsGetAddr { dst, .. } => {
224            ensure_no_fact(vcode, dst.to_writable_reg().to_reg())
225        }
226        Inst::CoffTlsGetAddr { dst, tmp, .. } => {
227            ensure_no_fact(vcode, dst.to_writable_reg().to_reg())?;
228            ensure_no_fact(vcode, tmp.to_writable_reg().to_reg())?;
229            Ok(())
230        }
231
232        Inst::Unwind { .. } | Inst::DummyUse { .. } => Ok(()),
233
234        Inst::StackSwitchBasic { .. } => Err(PccError::UnimplementedInst),
235
236        Inst::External { .. } => Ok(()), // TODO: unsure what to do about this!
237    }
238}

240fn check_load(
241    ctx: &FactContext,
242    dst: Option<Writable<Reg>>,
243    src: &SyntheticAmode,
244    vcode: &VCode<Inst>,
245    ty: Type,
246    to_bits: u16,
247) -> PccResult<Option<Fact>> {
248    let result_fact = dst.and_then(|dst| vcode.vreg_fact(dst.to_reg().into()));
249    let from_bits = u16::try_from(ty.bits()).unwrap();
250    check_mem(
251        ctx,
252        src,
253        vcode,
254        ty,
255        LoadOrStore::Load {
256            result_fact,
257            from_bits,
258            to_bits,
259        },
260    )
261}
262
263fn check_store(
264    ctx: &FactContext,
265    data: Option<Reg>,
266    dst: &SyntheticAmode,
267    vcode: &VCode<Inst>,
268    ty: Type,
269) -> PccResult<()> {
270    let stored_fact = data.and_then(|data| vcode.vreg_fact(data.into()));
271    check_mem(ctx, dst, vcode, ty, LoadOrStore::Store { stored_fact }).map(|_| ())
272}
273
274fn check_mem<'a>(
275    ctx: &FactContext,
276    amode: &SyntheticAmode,
277    vcode: &VCode<Inst>,
278    ty: Type,
279    op: LoadOrStore<'a>,
280) -> PccResult<Option<Fact>> {
281    let addr = match amode {
282        SyntheticAmode::Real(amode) if amode.get_flags().checked() => {
283            compute_addr(ctx, vcode, amode, 64).ok_or(PccError::MissingFact)?
284        }
285        _ => return Ok(None),
286    };
287
288    match op {
289        LoadOrStore::Load {
290            result_fact,
291            from_bits,
292            to_bits,
293        } => {
294            let loaded_fact = clamp_range(ctx, to_bits, from_bits, ctx.load(&addr, ty)?.cloned())?;
295            trace!(
296                "loaded_fact = {:?} result_fact = {:?}",
297                loaded_fact, result_fact
298            );
299            if ctx.subsumes_fact_optionals(loaded_fact.as_ref(), result_fact) {
300                Ok(loaded_fact.clone())
301            } else {
302                Err(PccError::UnsupportedFact)
303            }
304        }
305        LoadOrStore::Store { stored_fact } => {
306            ctx.store(&addr, ty, stored_fact)?;
307            Ok(None)
308        }
309    }
310}

312fn compute_addr(ctx: &FactContext, vcode: &VCode<Inst>, amode: &Amode, bits: u16) -> Option<Fact> {
313    trace!("compute_addr: {:?}", amode);
314    match *amode {
315        Amode::ImmReg { simm32, base, .. } => {
316            let base = get_fact_or_default(vcode, base, bits);
317            trace!("base = {:?}", base);
318            let simm32: i64 = simm32.into();
319            let simm32: u64 = simm32 as u64;
320            let offset = Fact::constant(bits, simm32);
321            let sum = ctx.add(&base, &offset, bits)?;
322            trace!("sum = {:?}", sum);
323            Some(sum)
324        }
325        Amode::ImmRegRegShift {
326            simm32,
327            base,
328            index,
329            shift,
330            ..
331        } => {
332            let base = get_fact_or_default(vcode, base.into(), bits);
333            let index = get_fact_or_default(vcode, index.into(), bits);
334            trace!("base = {:?} index = {:?}", base, index);
335            let shifted = ctx.shl(&index, bits, shift.into())?;
336            let sum = ctx.add(&base, &shifted, bits)?;
337            let simm32: i64 = simm32.into();
338            let simm32: u64 = simm32 as u64;
339            let offset = Fact::constant(bits, simm32);
340            let sum = ctx.add(&sum, &offset, bits)?;
341            trace!("sum = {:?}", sum);
342            Some(sum)
343        }
344        Amode::RipRelative { .. } => None,
345    }
