25/// Get the default GC compiler.
26pub fn gc_compiler(func_env: &mut FuncEnvironment<'_>) -> WasmResult<Box<dyn GcCompiler>> {
27    // If this function requires a GC compiler, that is not too bad of an
28    // over-approximation for it requiring a GC heap.
29    func_env.needs_gc_heap = true;
30
31    match func_env.tunables.collector {
32        #[cfg(feature = "gc-drc")]
33        Some(Collector::DeferredReferenceCounting) => Ok(Box::new(drc::DrcCompiler::default())),
34        #[cfg(not(feature = "gc-drc"))]
35        Some(Collector::DeferredReferenceCounting) => Err(wasm_unsupported!(
36            "the DRC collector is unavailable because the `gc-drc` feature \
37             was disabled at compile time",
38        )),
39
40        #[cfg(feature = "gc-null")]
41        Some(Collector::Null) => Ok(Box::new(null::NullCompiler::default())),
42        #[cfg(not(feature = "gc-null"))]
43        Some(Collector::Null) => Err(wasm_unsupported!(
44            "the null collector is unavailable because the `gc-null` feature \
45             was disabled at compile time",
46        )),
47
48        #[cfg(any(feature = "gc-drc", feature = "gc-null"))]
49        None => Err(wasm_unsupported!(
50            "support for GC types disabled at configuration time"
51        )),
52        #[cfg(not(any(feature = "gc-drc", feature = "gc-null")))]
53        None => Err(wasm_unsupported!(
54            "support for GC types disabled because no collector implementation \
55             was selected at compile time; enable one of the `gc-drc` or \
56             `gc-null` features",
57        )),
58    }
59}

89/// Emit code to read a struct field or array element from its raw address in
90/// the GC heap.
91///
92/// The given address MUST have already been bounds-checked via
93/// `prepare_gc_ref_access`.
94fn read_field_at_addr(
95    func_env: &mut FuncEnvironment<'_>,
96    builder: &mut FunctionBuilder<'_>,
97    ty: WasmStorageType,
98    addr: ir::Value,
99    extension: Option<Extension>,
100) -> WasmResult<ir::Value> {
101    assert_eq!(extension.is_none(), matches!(ty, WasmStorageType::Val(_)));
102    assert_eq!(
103        extension.is_some(),
104        matches!(ty, WasmStorageType::I8 | WasmStorageType::I16)
105    );
106
107    // Data inside GC objects is always little endian.
108    let flags = ir::MemFlags::trusted().with_endianness(ir::Endianness::Little);
109
110    let value = match ty {
111        WasmStorageType::I8 => builder.ins().load(ir::types::I8, flags, addr, 0),
112        WasmStorageType::I16 => builder.ins().load(ir::types::I16, flags, addr, 0),
113        WasmStorageType::Val(v) => match v {
114            WasmValType::I32 => builder.ins().load(ir::types::I32, flags, addr, 0),
115            WasmValType::I64 => builder.ins().load(ir::types::I64, flags, addr, 0),
116            WasmValType::F32 => builder.ins().load(ir::types::F32, flags, addr, 0),
117            WasmValType::F64 => builder.ins().load(ir::types::F64, flags, addr, 0),
118            WasmValType::V128 => builder.ins().load(ir::types::I8X16, flags, addr, 0),
119            WasmValType::Ref(r) => match r.heap_type.top() {
120                WasmHeapTopType::Any | WasmHeapTopType::Extern => gc_compiler(func_env)?
121                    .translate_read_gc_reference(func_env, builder, r, addr, flags)?,
122                WasmHeapTopType::Func => {
123                    let expected_ty = match r.heap_type {
124                        WasmHeapType::Func => ModuleInternedTypeIndex::reserved_value(),
125                        WasmHeapType::ConcreteFunc(ty) => ty.unwrap_module_type_index(),
126                        WasmHeapType::NoFunc => {
127                            let null = builder.ins().iconst(func_env.pointer_type(), 0);
128                            if !r.nullable {
129                                // Because `nofunc` is uninhabited, and this
130                                // reference is non-null, this is unreachable
131                                // code. Unconditionally trap via conditional
132                                // trap instructions to avoid inserting block
133                                // terminators in the middle of this block.
134                                builder.ins().trapz(null, TRAP_INTERNAL_ASSERT);
135                            }
136                            return Ok(null);
137                        }
138                        _ => unreachable!("not a function heap type"),
139                    };
140                    let expected_ty = builder
141                        .ins()
142                        .iconst(ir::types::I32, i64::from(expected_ty.as_bits()));
143
144                    let vmctx = func_env.vmctx_val(&mut builder.cursor());
145
146                    let func_ref_id = builder.ins().load(ir::types::I32, flags, addr, 0);
147                    let get_interned_func_ref = func_env
148                        .builtin_functions
149                        .get_interned_func_ref(builder.func);
150
151                    let call_inst = builder
152                        .ins()
153                        .call(get_interned_func_ref, &[vmctx, func_ref_id, expected_ty]);
154                    builder.func.dfg.first_result(call_inst)
155                }
156                WasmHeapTopType::Cont => {
157                    // TODO(#10248) GC integration for stack switching
158                    return Err(wasmtime_environ::WasmError::Unsupported(
159                        "Stack switching feature not compatbile with GC, yet".to_string(),
160                    ));
161                }
162            },
163        },
164    };
165
166    let value = match extension {
167        Some(Extension::Sign) => builder.ins().sextend(ir::types::I32, value),
168        Some(Extension::Zero) => builder.ins().uextend(ir::types::I32, value),
169        None => value,
170    };
171
172    Ok(value)
173}

175fn write_func_ref_at_addr(
176    func_env: &mut FuncEnvironment<'_>,
177    builder: &mut FunctionBuilder<'_>,
178    ref_type: WasmRefType,
179    flags: ir::MemFlags,
180    field_addr: ir::Value,
181    func_ref: ir::Value,
182) -> WasmResult<()> {
183    assert_eq!(ref_type.heap_type.top(), WasmHeapTopType::Func);
184
185    let vmctx = func_env.vmctx_val(&mut builder.cursor());
186
187    let intern_func_ref_for_gc_heap = func_env
188        .builtin_functions
189        .intern_func_ref_for_gc_heap(builder.func);
190
191    let func_ref = if ref_type.heap_type == WasmHeapType::NoFunc {
192        let null = builder.ins().iconst(func_env.pointer_type(), 0);
193        if !ref_type.nullable {
194            // Because `nofunc` is uninhabited, and this reference is
195            // non-null, this is unreachable code. Unconditionally trap
196            // via conditional trap instructions to avoid inserting
197            // block terminators in the middle of this block.
198            builder.ins().trapz(null, TRAP_INTERNAL_ASSERT);
199        }
200        null
201    } else {
202        func_ref
203    };
204
205    // Convert the raw `funcref` into a `FuncRefTableId` for use in the
206    // GC heap.
207    let call_inst = builder
208        .ins()
209        .call(intern_func_ref_for_gc_heap, &[vmctx, func_ref]);
210    let func_ref_id = builder.func.dfg.first_result(call_inst);
211    let func_ref_id = builder.ins().ireduce(ir::types::I32, func_ref_id);
212
213    // Store the id in the field.
214    builder.ins().store(flags, func_ref_id, field_addr, 0);
215
216    Ok(())
217}

219fn write_field_at_addr(
220    func_env: &mut FuncEnvironment<'_>,
221    builder: &mut FunctionBuilder<'_>,
222    field_ty: WasmStorageType,
223    field_addr: ir::Value,
224    new_val: ir::Value,
225) -> WasmResult<()> {
226    // Data inside GC objects is always little endian.
227    let flags = ir::MemFlags::trusted().with_endianness(ir::Endianness::Little);
228
229    match field_ty {
230        WasmStorageType::I8 => {
231            builder.ins().istore8(flags, new_val, field_addr, 0);
232        }
233        WasmStorageType::I16 => {
234            builder.ins().istore16(flags, new_val, field_addr, 0);
235        }
236        WasmStorageType::Val(WasmValType::Ref(r)) if r.heap_type.top() == WasmHeapTopType::Func => {
237            write_func_ref_at_addr(func_env, builder, r, flags, field_addr, new_val)?;
238        }
239        WasmStorageType::Val(WasmValType::Ref(r)) => {
240            gc_compiler(func_env)?
241                .translate_write_gc_reference(func_env, builder, r, field_addr, new_val, flags)?;
242        }
243        WasmStorageType::Val(_) => {
244            assert_eq!(
245                builder.func.dfg.value_type(new_val).bytes(),
246                wasmtime_environ::byte_size_of_wasm_ty_in_gc_heap(&field_ty)
247            );
248            builder.ins().store(flags, new_val, field_addr, 0);
249        }
250    }
251    Ok(())
252}

263fn default_value(
264    cursor: &mut FuncCursor,
265    func_env: &FuncEnvironment<'_>,
266    ty: &WasmStorageType,
267) -> ir::Value {
268    match ty {
269        WasmStorageType::I8 | WasmStorageType::I16 => cursor.ins().iconst(ir::types::I32, 0),
270        WasmStorageType::Val(v) => match v {
271            WasmValType::I32 => cursor.ins().iconst(ir::types::I32, 0),
272            WasmValType::I64 => cursor.ins().iconst(ir::types::I64, 0),
273            WasmValType::F32 => cursor.ins().f32const(0.0),
274            WasmValType::F64 => cursor.ins().f64const(0.0),
275            WasmValType::V128 => {
276                let c = cursor.func.dfg.constants.insert(vec![0; 16].into());
277                cursor.ins().vconst(ir::types::I8X16, c)
278            }
279            WasmValType::Ref(r) => {
280                assert!(r.nullable);
281                let (ty, needs_stack_map) = func_env.reference_type(r.heap_type);
282
283                // NB: The collector doesn't need to know about null references.
284                let _ = needs_stack_map;
285
286                cursor.ins().iconst(ty, 0)
287            }
288        },
289    }
290}

307pub fn translate_struct_get(
308    func_env: &mut FuncEnvironment<'_>,
309    builder: &mut FunctionBuilder<'_>,
310    struct_type_index: TypeIndex,
311    field_index: u32,
312    struct_ref: ir::Value,
313    extension: Option<Extension>,
314) -> WasmResult<ir::Value> {
315    log::trace!(
316        "translate_struct_get({struct_type_index:?}, {field_index:?}, {struct_ref:?}, {extension:?})"
317    );
318
319    // TODO: If we know we have a `(ref $my_struct)` here, instead of maybe a
320    // `(ref null $my_struct)`, we could omit the `trapz`. But plumbing that
321    // type info from `wasmparser` and through to here is a bit funky.
322    func_env.trapz(builder, struct_ref, crate::TRAP_NULL_REFERENCE);
323
324    let field_index = usize::try_from(field_index).unwrap();
325    let interned_type_index = func_env.module.types[struct_type_index].unwrap_module_type_index();
326
327    let struct_layout = func_env.struct_layout(interned_type_index);
328    let struct_size = struct_layout.size;
329
330    let field_offset = struct_layout.fields[field_index].offset;
331    let field_ty = &func_env.types.unwrap_struct(interned_type_index)?.fields[field_index];
332    let field_size = wasmtime_environ::byte_size_of_wasm_ty_in_gc_heap(&field_ty.element_type);
333    assert!(field_offset + field_size <= struct_size);
334
335    let field_addr = func_env.prepare_gc_ref_access(
336        builder,
337        struct_ref,
338        BoundsCheck::StaticObjectField {
339            offset: field_offset,
340            access_size: u8::try_from(field_size).unwrap(),
341            object_size: struct_size,
342        },
343    );
344
345    let result = read_field_at_addr(
346        func_env,
347        builder,
348        field_ty.element_type,
349        field_addr,
350        extension,
351    );
352    log::trace!("translate_struct_get(..) -> {result:?}");
353    result
354}

460impl ArrayInit<'_> {
461    /// Get the length (as an `i32`-typed `ir::Value`) of these array elements.
462    #[cfg_attr(not(any(feature = "gc-drc", feature = "gc-null")), allow(dead_code))]
463    fn len(self, pos: &mut FuncCursor) -> ir::Value {
464        match self {
465            ArrayInit::Fill { len, .. } => len,
466            ArrayInit::Elems(e) => {
467                let len = u32::try_from(e.len()).unwrap();
468                pos.ins().iconst(ir::types::I32, i64::from(len))
469            }
470        }
471    }
472
473    /// Initialize a newly-allocated array's elements.
474    #[cfg_attr(not(any(feature = "gc-drc", feature = "gc-null")), allow(dead_code))]
475    fn initialize(
476        self,
477        func_env: &mut FuncEnvironment<'_>,
478        builder: &mut FunctionBuilder<'_>,
479        interned_type_index: ModuleInternedTypeIndex,
480        base_size: u32,
481        size: ir::Value,
482        elems_addr: ir::Value,
483        mut init_field: impl FnMut(
484            &mut FuncEnvironment<'_>,
485            &mut FunctionBuilder<'_>,
486            WasmStorageType,
487            ir::Value,
488            ir::Value,
489        ) -> WasmResult<()>,
490    ) -> WasmResult<()> {
491        log::trace!(
492            "initialize_array({interned_type_index:?}, {base_size:?}, {size:?}, {elems_addr:?})"
493        );
494
495        assert!(!func_env.types[interned_type_index].composite_type.shared);
496        let array_ty = func_env.types[interned_type_index]
497            .composite_type
498            .inner
499            .unwrap_array();
500        let elem_ty = array_ty.0.element_type;
501        let elem_size = wasmtime_environ::byte_size_of_wasm_ty_in_gc_heap(&elem_ty);
502        let pointer_type = func_env.pointer_type();
503        let elem_size = builder.ins().iconst(pointer_type, i64::from(elem_size));
504        match self {
505            ArrayInit::Elems(elems) => {
506                let mut elem_addr = elems_addr;
507                for val in elems {
508                    init_field(func_env, builder, elem_ty, elem_addr, *val)?;
509                    elem_addr = builder.ins().iadd(elem_addr, elem_size);
510                }
511            }
512            ArrayInit::Fill { elem, len: _ } => {
513                // Compute the end address of the elements.
514                let base_size = builder.ins().iconst(pointer_type, i64::from(base_size));
515                let array_addr = builder.ins().isub(elems_addr, base_size);
516                let size = uextend_i32_to_pointer_type(builder, pointer_type, size);
517                let elems_end = builder.ins().iadd(array_addr, size);
518
519                emit_array_fill_impl(
520                    func_env,
521                    builder,
522                    elems_addr,
523                    elem_size,
524                    elems_end,
525                    |func_env, builder, elem_addr| {
526                        init_field(func_env, builder, elem_ty, elem_addr, elem)
527                    },
528                )?;
529            }
530        }
531        log::trace!("initialize_array: finished");
532        Ok(())
533    }
534}

536fn emit_array_fill_impl(
537    func_env: &mut FuncEnvironment<'_>,
538    builder: &mut FunctionBuilder<'_>,
539    elem_addr: ir::Value,
540    elem_size: ir::Value,
541    fill_end: ir::Value,
542    mut emit_elem_write: impl FnMut(
543        &mut FuncEnvironment<'_>,
544        &mut FunctionBuilder<'_>,
545        ir::Value,
546    ) -> WasmResult<()>,
547) -> WasmResult<()> {
548    log::trace!(
549        "emit_array_fill_impl(elem_addr: {elem_addr:?}, elem_size: {elem_size:?}, fill_end: {fill_end:?})"
550    );
551
552    let pointer_ty = func_env.pointer_type();
553
554    assert_eq!(builder.func.dfg.value_type(elem_addr), pointer_ty);
555    assert_eq!(builder.func.dfg.value_type(elem_size), pointer_ty);
556    assert_eq!(builder.func.dfg.value_type(fill_end), pointer_ty);
557
558    // Loop to fill the elements, emitting the equivalent of the following
559    // pseudo-CLIF:
560    //
561    // current_block:
562    //     ...
563    //     jump loop_header_block(elem_addr)
564    //
565    // loop_header_block(elem_addr: i32):
566    //     done = icmp eq elem_addr, fill_end
567    //     brif done, continue_block, loop_body_block
568    //
569    // loop_body_block:
570    //     emit_elem_write()
571    //     next_elem_addr = iadd elem_addr, elem_size
572    //     jump loop_header_block(next_elem_addr)
573    //
574    // continue_block:
575    //     ...
576
577    let current_block = builder.current_block().unwrap();
578    let loop_header_block = builder.create_block();
579    let loop_body_block = builder.create_block();
580    let continue_block = builder.create_block();
581
582    builder.ensure_inserted_block();
583    builder.insert_block_after(loop_header_block, current_block);
584    builder.insert_block_after(loop_body_block, loop_header_block);
585    builder.insert_block_after(continue_block, loop_body_block);
586
587    // Current block: jump to the loop header block with the first element's
588    // address.
589    builder.ins().jump(loop_header_block, &[elem_addr.into()]);
590
591    // Loop header block: check if we're done, then jump to either the continue
592    // block or the loop body block.
593    builder.switch_to_block(loop_header_block);
594    builder.append_block_param(loop_header_block, pointer_ty);
595    log::trace!("emit_array_fill_impl: loop header");
596    let elem_addr = builder.block_params(loop_header_block)[0];
597    let done = builder.ins().icmp(IntCC::Equal, elem_addr, fill_end);
598    builder
599        .ins()
600        .brif(done, continue_block, &[], loop_body_block, &[]);
601
602    // Loop body block: write the value to the current element, compute the next
603    // element's address, and then jump back to the loop header block.
604    builder.switch_to_block(loop_body_block);
605    log::trace!("emit_array_fill_impl: loop body");
606    emit_elem_write(func_env, builder, elem_addr)?;
607    let next_elem_addr = builder.ins().iadd(elem_addr, elem_size);
608    builder
609        .ins()
610        .jump(loop_header_block, &[next_elem_addr.into()]);
611
612    // Continue...
613    builder.switch_to_block(continue_block);
614    log::trace!("emit_array_fill_impl: finished");
615    builder.seal_block(loop_header_block);
616    builder.seal_block(loop_body_block);
617    builder.seal_block(continue_block);
618    Ok(())
619}

621pub fn translate_array_fill(
622    func_env: &mut FuncEnvironment<'_>,
623    builder: &mut FunctionBuilder<'_>,
624    array_type_index: TypeIndex,
625    array_ref: ir::Value,
626    index: ir::Value,
627    value: ir::Value,
628    n: ir::Value,
629) -> WasmResult<()> {
630    log::trace!(
631        "translate_array_fill({array_type_index:?}, {array_ref:?}, {index:?}, {value:?}, {n:?})"
632    );
633
634    let len = translate_array_len(func_env, builder, array_ref)?;
635
636    // Check that the full range of elements we want to fill is within bounds.
637    let end_index = func_env.uadd_overflow_trap(builder, index, n, crate::TRAP_ARRAY_OUT_OF_BOUNDS);
638    let out_of_bounds = builder
639        .ins()
640        .icmp(IntCC::UnsignedGreaterThan, end_index, len);
641    func_env.trapnz(builder, out_of_bounds, crate::TRAP_ARRAY_OUT_OF_BOUNDS);
642
643    // Get the address of the first element we want to fill.
644    let interned_type_index = func_env.module.types[array_type_index].unwrap_module_type_index();
645    let ArraySizeInfo {
646        obj_size,
647        one_elem_size,
648        base_size,
649    } = emit_array_size_info(func_env, builder, interned_type_index, len);
650    let offset_in_elems = builder.ins().imul(index, one_elem_size);
651    let obj_offset = builder.ins().iadd(base_size, offset_in_elems);
652    let elem_addr = func_env.prepare_gc_ref_access(
653        builder,
654        array_ref,
655        BoundsCheck::DynamicObjectField {
656            offset: obj_offset,
657            object_size: obj_size,
658        },
659    );
660
661    // Calculate the end address, just after the filled region.
662    let fill_size = builder.ins().imul(n, one_elem_size);
663    let fill_size = uextend_i32_to_pointer_type(builder, func_env.pointer_type(), fill_size);
664    let fill_end = builder.ins().iadd(elem_addr, fill_size);
665
666    let one_elem_size =
667        uextend_i32_to_pointer_type(builder, func_env.pointer_type(), one_elem_size);
668
669    let result = emit_array_fill_impl(
670        func_env,
671        builder,
672        elem_addr,
673        one_elem_size,
674        fill_end,
675        |func_env, builder, elem_addr| {
676            let elem_ty = func_env
677                .types
678                .unwrap_array(interned_type_index)?
679                .0
680                .element_type;
681            write_field_at_addr(func_env, builder, elem_ty, elem_addr, value)
682        },
683    );
684    log::trace!("translate_array_fill(..) -> {result:?}");
685    result
686}

688pub fn translate_array_len(
689    func_env: &mut FuncEnvironment<'_>,
690    builder: &mut FunctionBuilder,
691    array_ref: ir::Value,
692) -> WasmResult<ir::Value> {
693    log::trace!("translate_array_len({array_ref:?})");
694
695    func_env.trapz(builder, array_ref, crate::TRAP_NULL_REFERENCE);
696
697    let len_offset = gc_compiler(func_env)?.layouts().array_length_field_offset();
698    let len_field = func_env.prepare_gc_ref_access(
699        builder,
700        array_ref,
701        // Note: We can't bounds check the whole array object's size because we
702        // don't know its length yet. Chicken and egg problem.
703        BoundsCheck::StaticOffset {
704            offset: len_offset,
705            access_size: u8::try_from(ir::types::I32.bytes()).unwrap(),
706        },
707    );
708    let result = builder.ins().load(
709        ir::types::I32,
710        ir::MemFlags::trusted().with_readonly(),
711        len_field,
712        0,
713    );
714    log::trace!("translate_array_len(..) -> {result:?}");
715    Ok(result)
716}

718struct ArraySizeInfo {
719    /// The `i32` size of the whole array object, in bytes.
720    obj_size: ir::Value,
721
722    /// The `i32` size of each one of the array's elements, in bytes.
723    one_elem_size: ir::Value,
724
725    /// The `i32` size of the array's base object, in bytes. This is also the
726    /// offset from the start of the array object to its elements.
727    base_size: ir::Value,
728}
729
730/// Emit code to get the dynamic size (in bytes) of a whole array object, along
731/// with some other related bits.
732fn emit_array_size_info(
733    func_env: &mut FuncEnvironment<'_>,
734    builder: &mut FunctionBuilder<'_>,
735    array_type_index: ModuleInternedTypeIndex,
736    // `i32` value containing the array's length.
737    array_len: ir::Value,
738) -> ArraySizeInfo {
739    let array_layout = func_env.array_layout(array_type_index);
740
741    // Note that we check for overflow below because we can't trust the array's
742    // length: it came from inside the GC heap.
743    //
744    // We check for 32-bit multiplication overflow by performing a 64-bit
745    // multiplication and testing the high bits.
746    let one_elem_size = builder
747        .ins()
748        .iconst(ir::types::I64, i64::from(array_layout.elem_size));
749    let array_len = builder.ins().uextend(ir::types::I64, array_len);
750    let all_elems_size = builder.ins().imul(one_elem_size, array_len);
751
752    let high_bits = builder.ins().ushr_imm(all_elems_size, 32);
753    builder.ins().trapnz(high_bits, TRAP_INTERNAL_ASSERT);
754
755    let all_elems_size = builder.ins().ireduce(ir::types::I32, all_elems_size);
756    let base_size = builder
757        .ins()
758        .iconst(ir::types::I32, i64::from(array_layout.base_size));
759    let obj_size =
760        builder
761            .ins()
762            .uadd_overflow_trap(all_elems_size, base_size, TRAP_INTERNAL_ASSERT);
763
764    let one_elem_size = builder.ins().ireduce(ir::types::I32, one_elem_size);
765
766    ArraySizeInfo {
767        obj_size,
768        one_elem_size,
769        base_size,
770    }
771}

773/// Get the bounds-checked address of an element in an array.
774///
775/// The emitted code will trap if `index >= array.length`.
776///
777/// Returns the `ir::Value` containing the address of the `index`th element in
778/// the array. You may read or write a value of the array's element type at this
779/// address. You may not use it for any other kind of access, nor reuse this
780/// value across GC safepoints.
781fn array_elem_addr(
782    func_env: &mut FuncEnvironment<'_>,
783    builder: &mut FunctionBuilder<'_>,
784    array_type_index: ModuleInternedTypeIndex,
785    array_ref: ir::Value,
786    index: ir::Value,
787) -> ir::Value {
788    // First, assert that `index < array.length`.
789    //
790    // This check is visible at the Wasm-semantics level.
791    //
792    // TODO: We should emit spectre-safe bounds checks for array accesses (if
793    // configured) but we don't currently have a great way to do that here. The
794    // proper solution is to use linear memories to back GC heaps and reuse the
795    // code in `bounds_check.rs` to implement these bounds checks. That is all
796    // planned, but not yet implemented.
797
798    let len = translate_array_len(func_env, builder, array_ref).unwrap();
799
800    let in_bounds = builder.ins().icmp(IntCC::UnsignedLessThan, index, len);
801    func_env.trapz(builder, in_bounds, crate::TRAP_ARRAY_OUT_OF_BOUNDS);
802
803    // Compute the size (in bytes) of the whole array object.
804    let ArraySizeInfo {
805        obj_size,
806        one_elem_size,
807        base_size,
808    } = emit_array_size_info(func_env, builder, array_type_index, len);
809
810    // Compute the offset of the `index`th element within the array object.
811    //
812    // NB: no need to check for overflow here, since at this point we know that
813    // `len * elem_size + base_size` did not overflow and `i < len`.
814    let offset_in_elems = builder.ins().imul(index, one_elem_size);
815    let offset_in_array = builder.ins().iadd(offset_in_elems, base_size);
816
817    // Finally, use the object size and element offset we just computed to
818    // perform our implementation-internal bounds checks.
819    //
820    // Checking the whole object's size, rather than the `index`th element's
821    // size allows these bounds checks to be deduplicated across repeated
822    // accesses to the same array at different indices.
823    //
824    // This check should not be visible to Wasm, and serve to protect us from
825    // our own implementation bugs. The goal is to keep any potential widgets
826    // confined within the GC heap, and turn what would otherwise be a security
827    // vulnerability into a simple bug.
828    //
829    // TODO: Ideally we should fold the first Wasm-visible bounds check into
830    // this internal bounds check, so that we aren't performing multiple,
831    // redundant bounds checks. But we should figure out how to do this in a way
832    // that doesn't defeat the object-size bounds checking's deduplication
833    // mentioned above.
834    func_env.prepare_gc_ref_access(
835        builder,
836        array_ref,
837        BoundsCheck::DynamicObjectField {
838            offset: offset_in_array,
839            object_size: obj_size,
840        },
841    )
842}

887pub fn translate_ref_test(
888    func_env: &mut FuncEnvironment<'_>,
889    builder: &mut FunctionBuilder<'_>,
890    test_ty: WasmRefType,
891    val: ir::Value,
892    val_ty: WasmRefType,
893) -> WasmResult<ir::Value> {
894    log::trace!("translate_ref_test({test_ty:?}, {val:?})");
895
896    // First special case: testing for references to bottom types.
897    if test_ty.heap_type.is_bottom() {
898        let result = if test_ty.nullable {
899            // All null references (within the same type hierarchy) match null
900            // references to the bottom type.
901            func_env.translate_ref_is_null(builder.cursor(), val, val_ty)?
902        } else {
903            // `ref.test` is always false for non-nullable bottom types, as the
904            // bottom types are uninhabited.
905            builder.ins().iconst(ir::types::I32, 0)
906        };
907        log::trace!("translate_ref_test(..) -> {result:?}");
908        return Ok(result);
909    }
910
911    // And because `ref.test heap_ty` is only valid on operands whose type is in
912    // the same type hierarchy as `heap_ty`, if `heap_ty` is its hierarchy's top
913    // type, we only need to worry about whether we are testing for nullability
914    // or not.
915    if test_ty.heap_type.is_top() {
916        let result = if test_ty.nullable {
917            builder.ins().iconst(ir::types::I32, 1)
918        } else {
919            let is_null = func_env.translate_ref_is_null(builder.cursor(), val, val_ty)?;
920            let zero = builder.ins().iconst(ir::types::I32, 0);
921            let one = builder.ins().iconst(ir::types::I32, 1);
922            builder.ins().select(is_null, zero, one)
923        };
924        log::trace!("translate_ref_test(..) -> {result:?}");
925        return Ok(result);
926    }
927
928    // `i31ref`s are a little interesting because they don't point to GC
929    // objects; we test the bit pattern of the reference itself.
930    if test_ty.heap_type == WasmHeapType::I31 {
931        let i31_mask = builder.ins().iconst(
932            ir::types::I32,
933            i64::from(wasmtime_environ::I31_DISCRIMINANT),
934        );
935        let is_i31 = builder.ins().band(val, i31_mask);
936        let result = if test_ty.nullable {
937            let is_null = func_env.translate_ref_is_null(builder.cursor(), val, val_ty)?;
938            builder.ins().bor(is_null, is_i31)
939        } else {
940            is_i31
941        };
942        log::trace!("translate_ref_test(..) -> {result:?}");
943        return Ok(result);
944    }
945
946    // Otherwise, in the general case, we need to inspect our given object's
947    // actual type, which also requires null-checking and i31-checking it.
948
949    let is_any_hierarchy = test_ty.heap_type.top() == WasmHeapTopType::Any;
950
951    let non_null_block = builder.create_block();
952    let non_null_non_i31_block = builder.create_block();
953    let continue_block = builder.create_block();
954
955    // Current block: check if the reference is null and branch appropriately.
956    let is_null = func_env.translate_ref_is_null(builder.cursor(), val, val_ty)?;
957    let result_when_is_null = builder
958        .ins()
959        .iconst(ir::types::I32, test_ty.nullable as i64);
960    builder.ins().brif(
961        is_null,
962        continue_block,
963        &[result_when_is_null.into()],
964        non_null_block,
965        &[],
966    );
967
968    // Non-null block: We know the GC ref is non-null, but we need to also check
969    // for `i31` references that don't point to GC objects.
970    builder.switch_to_block(non_null_block);
971    log::trace!("translate_ref_test: non-null ref block");
972    if is_any_hierarchy {
973        let i31_mask = builder.ins().iconst(
974            ir::types::I32,
975            i64::from(wasmtime_environ::I31_DISCRIMINANT),
976        );
977        let is_i31 = builder.ins().band(val, i31_mask);
978        // If it is an `i31`, then create the result value based on whether we
979        // want `i31`s to pass the test or not.
980        let result_when_is_i31 = builder.ins().iconst(
981            ir::types::I32,
982            matches!(
983                test_ty.heap_type,
984                WasmHeapType::Any | WasmHeapType::Eq | WasmHeapType::I31
985            ) as i64,
986        );
987        builder.ins().brif(
988            is_i31,
989            continue_block,
990            &[result_when_is_i31.into()],
991            non_null_non_i31_block,
992            &[],
993        );
994    } else {
995        // If we aren't testing the `any` hierarchy, the reference cannot be an
996        // `i31ref`. Jump directly to the non-null and non-i31 block; rely on
997        // branch folding during lowering to clean this up.
998        builder.ins().jump(non_null_non_i31_block, &[]);
999    }
1000
1001    // Non-null and non-i31 block: Read the actual `VMGcKind` or
1002    // `VMSharedTypeIndex` out of the object's header and check whether it
1003    // matches the expected type.
1004    builder.switch_to_block(non_null_non_i31_block);
1005    log::trace!("translate_ref_test: non-null and non-i31 ref block");
1006    let check_header_kind = |func_env: &mut FuncEnvironment<'_>,
1007                             builder: &mut FunctionBuilder,
1008                             val: ir::Value,
1009                             expected_kind: VMGcKind|
1010     -> ir::Value {
1011        let kind_addr = func_env.prepare_gc_ref_access(
1012            builder,
1013            val,
1014            BoundsCheck::StaticObjectField {
1015                offset: wasmtime_environ::VM_GC_HEADER_KIND_OFFSET,
1016                access_size: wasmtime_environ::VM_GC_KIND_SIZE,
1017                object_size: wasmtime_environ::VM_GC_HEADER_SIZE,
1018            },
1019        );
1020        let actual_kind = builder.ins().load(
1021            ir::types::I32,
1022            ir::MemFlags::trusted().with_readonly(),
1023            kind_addr,
1024            0,
1025        );
1026        let expected_kind = builder
1027            .ins()
1028            .iconst(ir::types::I32, i64::from(expected_kind.as_u32()));
1029        // Inline version of `VMGcKind::matches`.
1030        let and = builder.ins().band(actual_kind, expected_kind);
1031        let kind_matches = builder
1032            .ins()
1033            .icmp(ir::condcodes::IntCC::Equal, and, expected_kind);
1034        builder.ins().uextend(ir::types::I32, kind_matches)
1035    };
1036    let result = match test_ty.heap_type {
1037        WasmHeapType::Any
1038        | WasmHeapType::None
1039        | WasmHeapType::Extern
1040        | WasmHeapType::NoExtern
1041        | WasmHeapType::Func
1042        | WasmHeapType::NoFunc
1043        | WasmHeapType::Cont
1044        | WasmHeapType::NoCont
1045        | WasmHeapType::I31 => unreachable!("handled top, bottom, and i31 types above"),
1046
1047        // For these abstract but non-top and non-bottom types, we check the
1048        // `VMGcKind` that is in the object's header.
1049        WasmHeapType::Eq => check_header_kind(func_env, builder, val, VMGcKind::EqRef),
1050        WasmHeapType::Struct => check_header_kind(func_env, builder, val, VMGcKind::StructRef),
1051        WasmHeapType::Array => check_header_kind(func_env, builder, val, VMGcKind::ArrayRef),
1052
1053        // For concrete types, we need to do a full subtype check between the
1054        // `VMSharedTypeIndex` in the object's header and the
1055        // `ModuleInternedTypeIndex` we have here.
1056        //
1057        // TODO: This check should ideally be done inline, but we don't have a
1058        // good way to access the `TypeRegistry`'s supertypes arrays from Wasm
1059        // code at the moment.
1060        WasmHeapType::ConcreteArray(ty) | WasmHeapType::ConcreteStruct(ty) => {
1061            let expected_interned_ty = ty.unwrap_module_type_index();
1062            let expected_shared_ty =
1063                func_env.module_interned_to_shared_ty(&mut builder.cursor(), expected_interned_ty);
1064
1065            let ty_addr = func_env.prepare_gc_ref_access(
1066                builder,
1067                val,
1068                BoundsCheck::StaticOffset {
1069                    offset: wasmtime_environ::VM_GC_HEADER_TYPE_INDEX_OFFSET,
1070                    access_size: func_env.offsets.size_of_vmshared_type_index(),
1071                },
1072            );
1073            let actual_shared_ty = builder.ins().load(
1074                ir::types::I32,
1075                ir::MemFlags::trusted().with_readonly(),
1076                ty_addr,
1077                0,
1078            );
1079
1080            func_env.is_subtype(builder, actual_shared_ty, expected_shared_ty)
1081        }
1082
1083        // Same as for concrete arrays and structs except that a `VMFuncRef`
1084        // doesn't begin with a `VMGcHeader` and is a raw pointer rather than GC
1085        // heap index.
1086        WasmHeapType::ConcreteFunc(ty) => {
1087            let expected_interned_ty = ty.unwrap_module_type_index();
1088            let expected_shared_ty =
1089                func_env.module_interned_to_shared_ty(&mut builder.cursor(), expected_interned_ty);
1090
1091            let actual_shared_ty = func_env.load_funcref_type_index(
1092                &mut builder.cursor(),
1093                ir::MemFlags::trusted().with_readonly(),
1094                val,
1095            );
1096
1097            func_env.is_subtype(builder, actual_shared_ty, expected_shared_ty)
1098        }
1099        WasmHeapType::ConcreteCont(_) => {
1100            // TODO(#10248) GC integration for stack switching
1101            return Err(wasmtime_environ::WasmError::Unsupported(
1102                "Stack switching feature not compatbile with GC, yet".to_string(),
1103            ));
1104        }
1105    };
1106    builder.ins().jump(continue_block, &[result.into()]);
1107
1108    // Control flow join point with the result.
1109    builder.switch_to_block(continue_block);
1110    let result = builder.append_block_param(continue_block, ir::types::I32);
1111    log::trace!("translate_ref_test(..) -> {result:?}");
1112
1113    builder.seal_block(non_null_block);
1114    builder.seal_block(non_null_non_i31_block);
1115    builder.seal_block(continue_block);
1116
1117    Ok(result)
1118}

1135///
1136/// Traps if the size overflows.
1137#[cfg_attr(not(any(feature = "gc-drc", feature = "gc-null")), allow(dead_code))]
1138fn emit_array_size(
1139    func_env: &mut FuncEnvironment<'_>,
1140    builder: &mut FunctionBuilder<'_>,
1141    array_layout: &GcArrayLayout,
1142    len: ir::Value,
1143) -> ir::Value {
1144    let base_size = builder
1145        .ins()
1146        .iconst(ir::types::I32, i64::from(array_layout.base_size));
1147
1148    // `elems_size = len * elem_size`
1149    //
1150    // Check for multiplication overflow and trap if it occurs, since that
1151    // means Wasm is attempting to allocate an array that is larger than our
1152    // implementation limits. (Note: there is no standard implementation
1153    // limit for array length beyond `u32::MAX`.)
1154    //
1155    // We implement this check by encoding our logically-32-bit operands as
1156    // i64 values, doing a 64-bit multiplication, and then checking the high
1157    // 32 bits of the multiplication's result. If the high 32 bits are not
1158    // all zeros, then the multiplication overflowed.
1159    debug_assert_eq!(builder.func.dfg.value_type(len), ir::types::I32);
1160    let len = builder.ins().uextend(ir::types::I64, len);
1161    let elems_size_64 = builder
1162        .ins()
1163        .imul_imm(len, i64::from(array_layout.elem_size));
1164    let high_bits = builder.ins().ushr_imm(elems_size_64, 32);
1165    func_env.trapnz(builder, high_bits, crate::TRAP_ALLOCATION_TOO_LARGE);
1166    let elems_size = builder.ins().ireduce(ir::types::I32, elems_size_64);
1167
1168    // And if adding the base size and elements size overflows, then the
1169    // allocation is too large.
1170    let size = func_env.uadd_overflow_trap(
1171        builder,
1172        base_size,
1173        elems_size,
1174        crate::TRAP_ALLOCATION_TOO_LARGE,
1175    );
1176
1177    size
1178}

1180/// Common helper for struct-field initialization that can be reused across
1181/// collectors.
1182#[cfg_attr(not(any(feature = "gc-drc", feature = "gc-null")), allow(dead_code))]
1183fn initialize_struct_fields(
1184    func_env: &mut FuncEnvironment<'_>,
1185    builder: &mut FunctionBuilder<'_>,
1186    struct_ty: ModuleInternedTypeIndex,
1187    raw_ptr_to_struct: ir::Value,
1188    field_values: &[ir::Value],
1189    mut init_field: impl FnMut(
1190        &mut FuncEnvironment<'_>,
1191        &mut FunctionBuilder<'_>,
1192        WasmStorageType,
1193        ir::Value,
1194        ir::Value,
1195    ) -> WasmResult<()>,
1196) -> WasmResult<()> {
1197    let struct_layout = func_env.struct_layout(struct_ty);
1198    let struct_size = struct_layout.size;
1199    let field_offsets: SmallVec<[_; 8]> = struct_layout.fields.iter().map(|f| f.offset).collect();
1200    assert_eq!(field_offsets.len(), field_values.len());
1201
1202    assert!(!func_env.types[struct_ty].composite_type.shared);
1203    let struct_ty = func_env.types[struct_ty]
1204        .composite_type
1205        .inner
1206        .unwrap_struct();
1207    let field_types: SmallVec<[_; 8]> = struct_ty.fields.iter().cloned().collect();
1208    assert_eq!(field_types.len(), field_values.len());
1209
1210    for ((ty, val), offset) in field_types.into_iter().zip(field_values).zip(field_offsets) {
1211        let size_of_access = wasmtime_environ::byte_size_of_wasm_ty_in_gc_heap(&ty.element_type);
1212        assert!(offset + size_of_access <= struct_size);
1213        let field_addr = builder.ins().iadd_imm(raw_ptr_to_struct, i64::from(offset));
1214        init_field(func_env, builder, ty.element_type, field_addr, *val)?;
1215    }
1216
1217    Ok(())
1218}

1325    /// Get the raw pointer of `gc_ref[offset]` bounds checked for an access of
1326    /// `size` bytes.
1327    ///
1328    /// The given `gc_ref` must be a non-null, non-i31 GC reference.
1329    ///
1330    /// If `check` is a `BoundsCheck::Object`, then it is the callers
1331    /// responsibility to ensure that `offset + access_size <= object_size`.
1332    ///
1333    /// Returns a raw pointer to `gc_ref[offset]` -- not a raw pointer to the GC
1334    /// object itself (unless `offset` happens to be `0`). This raw pointer may
1335    /// be used to read or write up to as many bytes as described by `bound`. Do
1336    /// NOT attempt accesses bytes outside of `bound`; that may lead to
1337    /// unchecked out-of-bounds accesses.
1338    ///
1339    /// This method is collector-agnostic.
1340    fn prepare_gc_ref_access(
1341        &mut self,
1342        builder: &mut FunctionBuilder,
1343        gc_ref: ir::Value,
1344        bounds_check: BoundsCheck,
1345    ) -> ir::Value {
1346        log::trace!("prepare_gc_ref_access({gc_ref:?}, {bounds_check:?})");
1347        assert_eq!(builder.func.dfg.value_type(gc_ref), ir::types::I32);
1348
1349        let gc_heap = self.get_gc_heap(&mut builder.func);
1350        let gc_heap = self.heaps[gc_heap].clone();
1351        let result = match crate::bounds_checks::bounds_check_and_compute_addr(
1352            builder,
1353            self,
1354            &gc_heap,
1355            gc_ref,
1356            bounds_check,
1357            crate::TRAP_INTERNAL_ASSERT,
1358        ) {
1359            Reachability::Reachable(v) => v,
1360            Reachability::Unreachable => {
1361                // We are now in unreachable code, but we don't want to plumb
1362                // through a bunch of `Reachability` through all of our callers,
1363                // so just assert we won't reach here and return `null`
1364                let null = builder.ins().iconst(self.pointer_type(), 0);
1365                builder.ins().trapz(null, crate::TRAP_INTERNAL_ASSERT);
1366                null
1367            }
1368        };
1369        log::trace!("prepare_gc_ref_access(..) -> {result:?}");
1370        result
1371    }

1373    /// Emit checks (if necessary) for whether the given `gc_ref` is null or is
1374    /// an `i31ref`.
1375    ///
1376    /// Takes advantage of static information based on `ty` as to whether the GC
1377    /// reference is nullable or can ever be an `i31`.
1378    ///
1379    /// Returns an `ir::Value` that is an `i32` will be non-zero if the GC
1380    /// reference is null or is an `i31ref`; otherwise, it will be zero.
1381    ///
1382    /// This method is collector-agnostic.
1383    #[cfg_attr(not(feature = "gc-drc"), allow(dead_code))]
1384    fn gc_ref_is_null_or_i31(
1385        &mut self,
1386        builder: &mut FunctionBuilder,
1387        ty: WasmRefType,
1388        gc_ref: ir::Value,
1389    ) -> ir::Value {
1390        assert_eq!(builder.func.dfg.value_type(gc_ref), ir::types::I32);
1391        assert!(ty.is_vmgcref_type_and_not_i31());
1392
1393        let might_be_i31 = match ty.heap_type {
1394            // If we are definitely dealing with an i31, we shouldn't be
1395            // emitting dynamic checks for it, and the caller shouldn't call
1396            // this function. Should have been caught by the assertion at the
1397            // start of the function.
1398            WasmHeapType::I31 => unreachable!(),
1399
1400            // Could potentially be an i31.
1401            WasmHeapType::Any | WasmHeapType::Eq => true,
1402
1403            // If it is definitely a struct, array, or uninhabited type, then it
1404            // is definitely not an i31.
1405            WasmHeapType::Array
1406            | WasmHeapType::ConcreteArray(_)
1407            | WasmHeapType::Struct
1408            | WasmHeapType::ConcreteStruct(_)
1409            | WasmHeapType::None => false,
1410
1411            // Despite being a different type hierarchy, this *could* be an
1412            // `i31` if it is the result of
1413            //
1414            //     (extern.convert_any (ref.i31 ...))
1415            WasmHeapType::Extern => true,
1416
1417            // Can only ever be `null`.
1418            WasmHeapType::NoExtern => false,
1419
1420            // Wrong type hierarchy, and also funcrefs are not GC-managed
1421            // types. Should have been caught by the assertion at the start of
1422            // the function.
1423            WasmHeapType::Func | WasmHeapType::ConcreteFunc(_) | WasmHeapType::NoFunc => {
1424                unreachable!()
1425            }
1426            WasmHeapType::Cont | WasmHeapType::ConcreteCont(_) | WasmHeapType::NoCont => {
1427                unreachable!()
1428            }
1429        };
1430
1431        match (ty.nullable, might_be_i31) {
1432            // This GC reference statically cannot be null nor an i31. (Let
1433            // Cranelift's optimizer const-propagate this value and erase any
1434            // unnecessary control flow resulting from branching on this value.)
1435            (false, false) => builder.ins().iconst(ir::types::I32, 0),
1436
1437            // This GC reference is always non-null, but might be an i31.
1438            (false, true) => builder.ins().band_imm(gc_ref, i64::from(I31_DISCRIMINANT)),
1439
1440            // This GC reference might be null, but can never be an i31.
1441            (true, false) => builder.ins().icmp_imm(IntCC::Equal, gc_ref, 0),
1442
1443            // Fully general case: this GC reference could be either null or an
1444            // i31.
1445            (true, true) => {
1446                let is_i31 = builder.ins().band_imm(gc_ref, i64::from(I31_DISCRIMINANT));
1447                let is_null = builder.ins().icmp_imm(IntCC::Equal, gc_ref, 0);
1448                let is_null = builder.ins().uextend(ir::types::I32, is_null);
1449                builder.ins().bor(is_i31, is_null)
1450            }
1451        }
1452    }

1454    // Emit code to check whether `a <: b` for two `VMSharedTypeIndex`es.
1455    pub(crate) fn is_subtype(
1456        &mut self,
1457        builder: &mut FunctionBuilder<'_>,
1458        a: ir::Value,
1459        b: ir::Value,
1460    ) -> ir::Value {
1461        log::trace!("is_subtype({a:?}, {b:?})");
1462
1463        let diff_tys_block = builder.create_block();
1464        let continue_block = builder.create_block();
1465
1466        // Current block: fast path for when `a == b`.
1467        log::trace!("is_subtype: fast path check for exact same types");
1468        let same_ty = builder.ins().icmp(IntCC::Equal, a, b);
1469        let same_ty = builder.ins().uextend(ir::types::I32, same_ty);
1470        builder.ins().brif(
1471            same_ty,
1472            continue_block,
1473            &[same_ty.into()],
1474            diff_tys_block,
1475            &[],
1476        );
1477
1478        // Different types block: fall back to the `is_subtype` libcall.
1479        builder.switch_to_block(diff_tys_block);
1480        log::trace!("is_subtype: slow path to do full `is_subtype` libcall");
1481        let is_subtype = self.builtin_functions.is_subtype(builder.func);
1482        let vmctx = self.vmctx_val(&mut builder.cursor());
1483        let call_inst = builder.ins().call(is_subtype, &[vmctx, a, b]);
1484        let result = builder.func.dfg.first_result(call_inst);
1485        builder.ins().jump(continue_block, &[result.into()]);
1486
1487        // Continue block: join point for the result.
1488        builder.switch_to_block(continue_block);
1489        let result = builder.append_block_param(continue_block, ir::types::I32);
1490        log::trace!("is_subtype(..) -> {result:?}");
1491
1492        builder.seal_block(diff_tys_block);
1493        builder.seal_block(continue_block);
1494
1495        result
1496    }
