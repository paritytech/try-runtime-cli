5macro_rules! arr_newtype_fmt_impl {
6    ($ty:ident, $bytes:expr $(, $gen:ident: $gent:ident)*) => {
7        impl<$($gen: $gent),*> $crate::_export::_core::fmt::LowerHex for $ty<$($gen),*> {
8            #[inline]
9            fn fmt(&self, f: &mut $crate::_export::_core::fmt::Formatter) -> $crate::_export::_core::fmt::Result {
10                #[allow(unused)]
11                use crate::Hash as _;
12                let case = $crate::hex::Case::Lower;
13                if <$ty<$($gen),*>>::DISPLAY_BACKWARD {
14                    $crate::hex::fmt_hex_exact!(f, $bytes, self.0.iter().rev(), case)
15                } else {
16                    $crate::hex::fmt_hex_exact!(f, $bytes, self.0.iter(), case)
17                }
18            }
19        }
20
21        impl<$($gen: $gent),*> $crate::_export::_core::fmt::UpperHex for $ty<$($gen),*> {
22            #[inline]
23            fn fmt(&self, f: &mut $crate::_export::_core::fmt::Formatter) -> $crate::_export::_core::fmt::Result {
24                #[allow(unused)]
25                use crate::Hash as _;
26                let case = $crate::hex::Case::Upper;
27                if <$ty<$($gen),*>>::DISPLAY_BACKWARD {
28                    $crate::hex::fmt_hex_exact!(f, $bytes, self.0.iter().rev(), case)
29                } else {
30                    $crate::hex::fmt_hex_exact!(f, $bytes, self.0.iter(), case)
31                }
32            }
33        }
34
35        impl<$($gen: $gent),*> $crate::_export::_core::fmt::Display for $ty<$($gen),*> {
36            #[inline]
37            fn fmt(&self, f: &mut $crate::_export::_core::fmt::Formatter) -> $crate::_export::_core::fmt::Result {
38                $crate::_export::_core::fmt::LowerHex::fmt(self, f)
39            }
40        }
41
42        impl<$($gen: $gent),*> $crate::_export::_core::fmt::Debug for $ty<$($gen),*> {
43            #[inline]
44            fn fmt(&self, f: &mut $crate::_export::_core::fmt::Formatter) -> $crate::_export::_core::fmt::Result {
45                write!(f, "{:#}", self)
46            }
47        }
48    }
49}

52/// Adds trait impls to the type called `Hash` in the current scope.
53///
54/// Implpements various conversion traits as well as the [`crate::Hash`] trait.
55/// Arguments:
56///
57/// * `$bits` - number of bits this hash type has
58/// * `$reverse` - `bool`  - `true` if the hash type should be displayed backwards, `false`
59///    otherwise.
60/// * `$gen: $gent` - generic type(s) and trait bound(s)
61///
62/// Restrictions on usage:
63///
64/// * There must be a free-standing `fn from_engine(HashEngine) -> Hash` in the scope
65/// * `fn internal_new([u8; $bits / 8]) -> Self` must exist on `Hash`
66/// * `fn internal_engine() -> HashEngine` must exist on `Hash`
67///
68/// `from_engine` obviously implements the finalization algorithm.
69/// `internal_new` is required so that types with more than one field are constructible.
70/// `internal_engine` is required to initialize the engine for given hash type.
71macro_rules! hash_trait_impls {
72    ($bits:expr, $reverse:expr $(, $gen:ident: $gent:ident)*) => {
73        impl<$($gen: $gent),*> Hash<$($gen),*> {
74            /// Zero cost conversion between a fixed length byte array shared reference and
75            /// a shared reference to this Hash type.
76            pub fn from_bytes_ref(bytes: &[u8; $bits / 8]) -> &Self {
77                // Safety: Sound because Self is #[repr(transparent)] containing [u8; $bits / 8]
78                unsafe { &*(bytes as *const _ as *const Self) }
79            }
80
81            /// Zero cost conversion between a fixed length byte array exclusive reference and
82            /// an exclusive reference to this Hash type.
83            pub fn from_bytes_mut(bytes: &mut [u8; $bits / 8]) -> &mut Self {
84                // Safety: Sound because Self is #[repr(transparent)] containing [u8; $bits / 8]
85                unsafe { &mut *(bytes as *mut _ as *mut Self) }
86            }
87        }
88
89        impl<$($gen: $gent),*> str::FromStr for Hash<$($gen),*> {
90            type Err = $crate::hex::HexToArrayError;
91            fn from_str(s: &str) -> $crate::_export::_core::result::Result<Self, Self::Err> {
92                use $crate::{Hash, hex::{FromHex}};
93
94                let mut bytes = <[u8; $bits / 8]>::from_hex(s)?;
95                if $reverse {
96                    bytes.reverse();
97                }
98                Ok(Self::from_byte_array(bytes))
99            }
100        }
101
102        $crate::internal_macros::arr_newtype_fmt_impl!(Hash, $bits / 8 $(, $gen: $gent)*);
103        serde_impl!(Hash, $bits / 8 $(, $gen: $gent)*);
104        borrow_slice_impl!(Hash $(, $gen: $gent)*);
105
106        impl<$($gen: $gent),*> $crate::_export::_core::convert::AsRef<[u8; $bits / 8]> for Hash<$($gen),*> {
107            fn as_ref(&self) -> &[u8; $bits / 8] {
108                &self.0
109            }
110        }
111
112        impl<I: SliceIndex<[u8]> $(, $gen: $gent)*> Index<I> for Hash<$($gen),*> {
113            type Output = I::Output;
114
115            #[inline]
116            fn index(&self, index: I) -> &Self::Output {
117                &self.0[index]
118            }
119        }
120
121        impl<$($gen: $gent),*> crate::Hash for Hash<$($gen),*> {
122            type Engine = HashEngine;
123            type Bytes = [u8; $bits / 8];
124
125            const LEN: usize = $bits / 8;
126            const DISPLAY_BACKWARD: bool = $reverse;
127
128            fn engine() -> Self::Engine {
129                Self::internal_engine()
130            }
131
132            fn from_engine(e: HashEngine) -> Hash<$($gen),*> {
133                from_engine(e)
134            }
135
136            fn from_slice(sl: &[u8]) -> $crate::_export::_core::result::Result<Hash<$($gen),*>, FromSliceError> {
137                if sl.len() != $bits / 8 {
138                    Err(FromSliceError{expected: Self::LEN, got: sl.len()})
139                } else {
140                    let mut ret = [0; $bits / 8];
141                    ret.copy_from_slice(sl);
142                    Ok(Self::internal_new(ret))
143                }
144            }
145
146            fn to_byte_array(self) -> Self::Bytes {
147                self.0
148            }
149
150            fn as_byte_array(&self) -> &Self::Bytes {
151                &self.0
152            }
153
154            fn from_byte_array(bytes: Self::Bytes) -> Self {
155                Self::internal_new(bytes)
156            }
157
158            fn all_zeros() -> Self {
159                Hash::internal_new([0x00; $bits / 8])
160            }
161        }
162    }
163}
164pub(crate) use hash_trait_impls;

166/// Creates a type called `Hash` and implements standard interface for it.
167///
168/// The created type will have all standard derives, `Hash` impl and implementation of
169/// `internal_engine` returning default. The created type has a single field.
170///
171/// Arguments:
172///
173/// * `$bits` - the number of bits of the hash type
174/// * `$reverse` - `true` if the hash should be displayed backwards, `false` otherwise
175/// * `$doc` - doc string to put on the type
176/// * `$schemars` - a literal that goes into `schema_with`.
177///
178/// The `from_engine` free-standing function is still required with this macro. See the doc of
179/// [`hash_trait_impls`].
180macro_rules! hash_type {
181    ($bits:expr, $reverse:expr, $doc:literal) => {
182        #[doc = $doc]
183        #[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
184        #[repr(transparent)]
185        pub struct Hash([u8; $bits / 8]);
186
187        impl Hash {
188            fn internal_new(arr: [u8; $bits / 8]) -> Self { Hash(arr) }
189
190            fn internal_engine() -> HashEngine { Default::default() }
191        }
192
193        #[cfg(feature = "schemars")]
194        impl schemars::JsonSchema for Hash {
195            fn schema_name() -> String { "Hash".to_owned() }
196
197            fn json_schema(gen: &mut schemars::gen::SchemaGenerator) -> schemars::schema::Schema {
198                let len = $bits / 8;
199                let mut schema: schemars::schema::SchemaObject = <String>::json_schema(gen).into();
200                schema.string = Some(Box::new(schemars::schema::StringValidation {
201                    max_length: Some(len * 2),
202                    min_length: Some(len * 2),
203                    pattern: Some("[0-9a-fA-F]+".to_owned()),
204                }));
205                schema.into()
206            }
207        }
208
209        crate::internal_macros::hash_trait_impls!($bits, $reverse);
210    };
211}
212pub(crate) use hash_type;
